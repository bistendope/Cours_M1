  87         int partial;

  91         union {

  93                         /* object id of the root directory */

 191  *
 192  * The state starts at the root directory; each directory is a singly

 197  * The identity field is within the union structure because file and

 611 fsdir_add_entry(struct fsobject *dir, struct fsdirent *nde)
 612 {
 613         struct fsdirent *ode;
 614 
 615         assert(dir->isdir);
 616         for (ode = dir->obj_dir.entries; ode != NULL; ode = ode->next) {
 617                 if (ode->name == nde->name) {
 618                         die("In directory %u, %s already existed",
 619                             dir->obj_dir.identity, name_get(nde->name));

 621         }
 622         nde->next = dir->obj_dir.entries;
 623         dir->obj_dir.entries = nde;

 675 static
 676 struct fsobject *
 677 findsub(struct fsobject *obj, unsigned isdir, unsigned id)
 678 {
 679         struct fsobject *ret;
 680         struct fsdirent *de;
 681         unsigned objid;
 682 
 683         /* Are we on the object we're looking for? */
 684         objid = obj->isdir ? obj->obj_dir.identity : obj->obj_file.identity;
 685         if (obj->isdir == isdir && objid == id) {
 686                 return obj;
 687         }
 688 
 689         /* If the object we're on isn't a dir, can't search any further */
 690         if (!obj->isdir) {
 691                 return NULL;
 692         }
 693 
 694         for (de = obj->obj_dir.entries; de != NULL; de = de->next) {
 695                 ret = findsub(de->obj, isdir, id);
 696                 if (ret != NULL) {
 697                         return ret;
 698                 }
 699         }
 700         return NULL;
 701 }

 738 /*
 739  * Apply change CHANGE to the volume state encoded in/under ROOTDIR.
 740  */
 741 static
 742 void
 743 apply_change(struct fsobject *rootdir, struct fschange *change)
 744 {
 745         struct fsobject *obj1, *obj2;
 746         struct fsdirent *de;
 747         off_t endpos;
 748 
 749         assert(rootdir->isdir);
 750 
 751         switch (change->type) {
 752             case FC_NEWFS:
 753                 /*
 754                  * Creating the volume; crosscheck the root directory.
 755                  */
 756                 assert(rootdir->isdir);
 757                 assert(rootdir->refcount == 1);
 758                 assert(rootdir->obj_dir.identity ==
 759                        change->fc_newfs.rootdirnum);
 760                 assert(rootdir->obj_dir.entries == NULL);
 761                 assert(rootdir->obj_dir.parent == rootdir);
 762                 break;
 763             case FC_TRUNCATE:
 764                 /*
 765                  * Truncate a file. (Only files, not directories.)
 766                  *
 767                  * Because truncates can and do get posted after a
 768                  * file is unlinked, we have to tolerate not finding
 769                  * the file. We don't model unlinked files, because
 770                  * they aren't visible and our goal is to check the
 771                  * visible/observable state.
 772                  */
 773                 obj1 = findfile_maybe(rootdir, change->fc_truncate.file);
 774                 if (obj1 != NULL) {
 775                         obj1->obj_file.len = change->fc_truncate.len;
 776                 }
 777                 break;
 778             case FC_WRITE:
 779                 /*
 780                  * Write to a file. (Only files, not directories.)
 781                  * All this actually does is update the file size if
 782                  * needed.
 783                  *
 784                  * We also have to tolerate writes to unlinked files.
 785                  */
 786                 obj1 = findfile_maybe(rootdir, change->fc_write.file);
 787                 if (obj1 != NULL) {
 788                         endpos = change->fc_write.pos + change->fc_write.len;
 789                         if (obj1->obj_file.len < endpos) {
 790                                 obj1->obj_file.len = endpos;
 791                         }
 792                 }
 793                 break;
 794             case FC_CREAT:
 795                 /*
 796                  * Create a file. This creates a new object and a
 797                  * directory entry to hold it, then inserts the
 798                  * directory entry in the containing directory.
 799                  */
 800                 obj1 = finddir(rootdir, change->fc_creat.dir);
 801                 obj2 = fsobject_create_file(change->fc_creat.newfile);
 802                 de = fsdirent_create(change->fc_creat.name, obj2);
 803                 fsdir_add_entry(obj1, de);
 804                 break;
 805             case FC_MKDIR:
 806                 /*
 807                  * Create a directory. The same as creating a file,
 808                  * except the new object is a directory.
 809                  */
 810                 obj1 = finddir(rootdir, change->fc_mkdir.dir);
 811                 obj2 = fsobject_create_dir(change->fc_mkdir.newdir, obj1);
 812                 de = fsdirent_create(change->fc_mkdir.name, obj2);
 813                 fsdir_add_entry(obj1, de);
 814                 break;
 815             case FC_RMDIR:
 816                 /*
 817                  * Remove a directory. First take out the directory
 818                  * entry (by name); then examine the object found;
 819                  * then delete everything.
 820                  *
 821                  * XXX: why do these checks use assert vs. testing and
 822                  * calling die()?
 823                  */
 824                 obj1 = finddir(rootdir, change->fc_rmdir.dir);
 825                 de = fsdir_remove_entry(obj1, change->fc_rmdir.name);
 826                 obj2 = de->obj;
 827                 de->obj = NULL;
 828                 assert(obj2->isdir);
 829                 assert(obj2->obj_dir.entries == NULL);
 830                 assert(obj2->obj_dir.identity == change->fc_rmdir.victimdir);
 831                 assert(obj2->obj_dir.parent == obj1);
 832                 fsdirent_destroy(de);
 833                 fsobject_decref(obj2);
 834                 break;
 835             case FC_UNLINK:
 836                 /*
 837                  * Remove a file. Much the same as removing a directory.
 838                  */
 839                 obj1 = finddir(rootdir, change->fc_unlink.dir);
 840                 de = fsdir_remove_entry(obj1, change->fc_unlink.name);
 841                 obj2 = de->obj;
 842                 de->obj = NULL;
 843                 assert(!obj2->isdir);
 844                 assert(obj2->obj_file.identity ==
 845                        change->fc_unlink.victimfile);
 846                 fsdirent_destroy(de);
 847                 fsobject_decref(obj2);
 848                 break;
 849             case FC_LINK:
 850                 /*
 851                  * Hard-link a file.
 852                  */
 853                 obj1 = finddir(rootdir, change->fc_link.fromdir);
 854                 de = fsdir_find_entry(obj1, change->fc_link.fromname, 1);
 855                 obj2 = de->obj;
 856                 assert(!obj2->isdir);
 857                 assert(obj2->obj_file.identity == change->fc_link.file);
 858                 obj1 = finddir(rootdir, change->fc_link.todir);
 859                 fsobject_incref(obj2);
 860                 de = fsdirent_create(change->fc_link.toname, obj2);
 861                 fsdir_add_entry(obj1, de);
 862                 break;
 863             case FC_RENAMEFILE:
 864                 /*
 865                  * Rename a file. XXX: this appears to do the wrong
 866                  * thing if you rename one file over another.
 867                  */
 868                 obj1 = finddir(rootdir, change->fc_renamefile.fromdir);
 869                 de = fsdir_remove_entry(obj1, change->fc_renamefile.fromname);
 870                 obj2 = de->obj;
 871                 assert(!obj2->isdir);
 872                 assert(obj2->obj_file.identity ==
 873                        change->fc_renamefile.movedfile);
 874                 obj1 = finddir(rootdir, change->fc_renamefile.todir);
 875                 de->name = change->fc_renamefile.toname;
 876                 fsdir_add_entry(obj1, de);
 877                 break;
 878             case FC_RENAMEDIR:
 879                 /*
 880                  * Rename a directory. Same as renaming a file, except
 881                  * that we update the parent of the directory being
 882                  * moved. XXX: also seems to do the wrong thing if you
 883                  * rename one directory over another. And, XXX: should
 884                  * this be updating the refcount as the parent
 885                  * changes? Shouldn't the refcount be the same as the
 886                  * on-disk linkcount, and shouldn't the on-disk
 887                  * linkcount be part of the state we examine and
 888                  * verify?
 889                  */
 890                 obj1 = finddir(rootdir, change->fc_renamedir.fromdir);
 891                 de = fsdir_remove_entry(obj1, change->fc_renamedir.fromname);
 892                 obj2 = de->obj;
 893                 assert(obj2->isdir);
 894                 assert(obj2->obj_dir.identity ==
 895                        change->fc_renamedir.moveddir);
 896                 assert(obj2->obj_dir.parent == obj1);
 897                 obj1 = finddir(rootdir, change->fc_renamedir.todir);
 898                 de->name = change->fc_renamedir.toname;
 899                 obj2->obj_dir.parent = obj1;
 900                 fsdir_add_entry(obj1, de);
 901                 break;
 902         }
 903 }

 933  * however, the change record should otherwise be fully filled in
 934  * already.
 935  *
 936  * usage: fc_attach(newrecord);
 937  */
 938 static
 939 void
 940 fc_attach(struct fschange *new)
 941 {
 942         struct fschange *prev;
 943 
 944         prev = changes;
 945 
 946         assert(prev->next == NULL);
 947         assert(new->prev == NULL);
 948         assert(new->next == NULL);
 949         prev->next = new;

 957 /*

 959  */

 963 {
 964         /* If we already have a state, throw it away. */
 965         if (state != NULL) {
 966                 fsobject_destroytree(state);
 967         }

1886 /*
1887  * Inspect a directory. PARENTOBJ is the *containing* directory;
1888  * PARENTINO is its inode number, and DIRNAMESTR is the name of the
1889  * directory to inspect. The arguments are set up this way (rather
1890  * than e.g. passing the fsobject of the directory to inspect) to
1891  * improve the error reporting, and to allow checking .., and so
1892  * forth.
1893  */
1894 static
1895 struct fsobject *
1896 inspectdir(struct fsobject *parentobj, ino_t parentino, const char *dirnamestr)
1897 {
1898         char subnamestr[NAME_MAX+1];
1899         size_t subnamelen;
1900         struct stat dirstat, dotstat, substat;
1901         int dirfd;
1902         struct fsobject *subobj, *ret;
1903         struct fsdirent *contents, *de;
1904 
1905         /*
1906          * Stat the target, and cd into it.
1907          */
1908 
1909         if (xstat(dirnamestr, &dirstat)) {
1910                 err(1, "%s: stat", dirnamestr);
1911         }
1912 
1913         assert(S_ISDIR(dirstat.st_mode));
1914         if (chdir(dirnamestr)) {
1915                 err(1, "%s: chdir", dirnamestr);
1916         }
1917 
1918         /*
1919          * Check that . is correct
1920          */
1921 
1922         if (xstat(".", &dotstat)) {
1923                 err(1, "In %s: .: stat", dirnamestr);
1924         }
1925         if (dotstat.st_dev != dirstat.st_dev) {
1926                 errx(1, "in %s: .: wrong volume id; found %u, expected %u",
1927                      dirnamestr, dotstat.st_dev, dirstat.st_dev);
1928         }
1929         if (dotstat.st_ino != dirstat.st_ino) {
1930                 errx(1, "%s/.: wrong inode number; found %u, expected %u",
1931                      dirnamestr, dotstat.st_ino, dirstat.st_ino);
1932         }
1933 
1934         /*
1935          * Check that .. leads back
1936          */
1937 
1938         if (xstat("..", &dotstat)) {
1939                 err(1, "In %s: ..: stat", dirnamestr);
1940         }
1941         if (dotstat.st_dev != dirstat.st_dev) {
1942                 errx(1, "In %s: ..: wrong volume id; found %u, expected %u",
1943                      dirnamestr, dotstat.st_dev, dirstat.st_dev);
1944         }
1945         if (dotstat.st_ino != parentino) {
1946                 errx(1, "In %s: ..: wrong inode number; found %u, expected %u",
1947                      dirnamestr, dotstat.st_ino, parentino);
1948         }
1949 
1950         /*
1951          * Create a directory fsobject.
1952          */
1953 
1954         dirfd = open(".", O_RDONLY);
1955         if (dirfd < 0) {
1956                 err(1, "In %s: .: open", dirnamestr);
1957         }
1958 
1959         ret = fsobject_create_dir(UNKNOWN_ID, parentobj);
1960 
1961         /*
1962          * Read the contents of the target directory and create
1963          * directory entries for them. Recurse for ones that are
1964          * themselves directories.
1965          */
1966         contents = NULL;
1967         while (1) {
1968                 subnamelen = getdirentry(dirfd, subnamestr,
1969                                          sizeof(subnamestr)-1);
1970                 if (subnamelen == 0) {
1971                         break;
1972                 }
1973                 subnamestr[subnamelen] = 0;
1974 
1975                 if (!strcmp(subnamestr, ".") || !strcmp(subnamestr, "..")) {
1976                         continue;
1977                 }
1978                 if (xstat(subnamestr, &substat)) {
1979                         err(1, "In %s: %s: stat", dirnamestr, subnamestr);
1980                 }
1981                 if (S_ISDIR(substat.st_mode)) {
1982                         subobj = inspectdir(ret, dirstat.st_ino, subnamestr);
1983                         found_subdirs++;
1984                 }
1985                 else {
1986                         subobj = fsobject_create_file(UNKNOWN_ID);
1987                         subobj->obj_file.len = substat.st_size;
1988                         found_files++;
1989                 }
1990                 de = fsdirent_create(name_find(subnamestr), subobj);
1991                 de->next = contents;
1992                 contents = de;
1993         }
1994 
1995         /*
1996          * Done scanning; cd out, save the contents, and return the
1997          * new object.
1998          */
1999 
2000         close(dirfd);
2001         if (chdir("..")) {
2002                 err(1, "In %s; ..: chdir", dirnamestr);
2003         }
2004 
2005         ret->obj_dir.entries = contents;
2006 
2007         return ret;
2008 }

2047 /*
2048  * Compare two fsobjects. Return the matching score. (lower scores are
2049  * better matches)
2050  */
2051 static
2052 unsigned
2053 compare_objects(struct fsobject *obja, struct fsobject *objb)
2054 {
2055         struct fsdirent *enta, *entb;
2056         unsigned ret, found;
2057 
2058         if (obja->isdir != objb->isdir) {
2059                 /*
2060                  * One object's a file, the other's a directory.
2061                  *
2062                  * Return one point for each name in the missing
2063                  * subtree. This includes one point for the top dir,
2064                  * which is mismatched rather than missing.
2065                  */
2066                 if (obja->isdir) {
2067                         return count_subtree(obja);
2068                 }
2069                 assert(objb->isdir);
2070                 return count_subtree(objb);
2071         }
2072 
2073         if (!obja->isdir) {
2074                 /* 
2075                  * Both objects are files
2076                  */
2077                 assert(!objb->isdir);
2078                 if (obja->obj_file.len != objb->obj_file.len) {
2079                         /* one point for the size being different */
2080                         return 1;
2081                 }
2082                 return 0;
2083         }
2084 
2085         /* 
2086          * Both objects are directories. Recurse on all pairs of
2087          * entries that have the same name. Return one point for each
2088          * name that's present (recursively) in one object but not the
2089          * other.
2090          *
2091          * XXX: sort the entries first or something instead of naively
2092          * searching 2*N^2 times.
2093          */
2094         assert(obja->isdir);
2095         assert(objb->isdir);
2096 
2097         ret = 0;
2098         for (enta = obja->obj_dir.entries; enta != NULL; enta = enta->next) {
2099                 found = 0;
2100                 for (entb = objb->obj_dir.entries; entb != NULL;
2101                      entb = entb->next) {
2102                         if (enta->name == entb->name) {
2103                                 ret += compare_objects(enta->obj, entb->obj);
2104                                 found = 1;
2105                                 break;
2106                         }
2107                 }
2108                 if (!found) {
2109                         if (enta->obj->isdir) {
2110                                 /* whole subtree is missing */
2111                                 ret += count_subtree(enta->obj);
2112                         }
2113                         /* one file is missing */
2114                         ret += 1;
2115                 }
2116         }
2117 
2118 
2119         for (entb = objb->obj_dir.entries; entb != NULL; entb = entb->next) {
2120                 found = 0;
2121                 for (enta = obja->obj_dir.entries; enta != NULL;
2122                      enta = enta->next) {
2123                         if (enta->name == entb->name) {
2124                                 found = 1;
2125                                 break;
2126                         }
2127                 }
2128                 if (!found) {
2129                         if (entb->obj->isdir) {
2130                                 /* whole subtree is missing */
2131                                 ret += count_subtree(entb->obj);
2132                         }
2133                         /* one file is missing */
2134                         ret += 1;
2135                 }
2136         }
2137 
2138         return ret;
2139 }

2453 /*
2454  * Check a range of the file FD named NAMESTR, from START to END,
2455  * against the model state expected as of CHANGE.
2456  */
2457 static
2458 void
2459 checkfilerange(int fd, const char *namestr, struct fschange *change,
2460                off_t start, off_t end)
2461 {
2462         assert(start < end);
2463         if (change->type == FC_TRUNCATE) {
2464                 off_t tlen;
2465                 struct fschange *prev;
2466 
2467                 /*
2468                  * The most recent change was a truncate. Anything
2469                  * beyond the truncate length should read as zeroes;
2470                  * recurse on the part before (if any) using the
2471                  * previous change affecting this file.
2472                  *
2473                  * We might be checking that a chunk of the file
2474                  * beyond the truncate length is zero (even though as
2475                  * of this change it's past EOF) if a later change
2476                  * made that region into a hole in a sparse file.
2477                  */
2478 
2479                 tlen = change->fc_truncate.len;
2480                 prev = change->fc_truncate.prev_thisfile;
2481 
2482                 if (tlen < start) {
2483                         checkfilezeros(fd, namestr, start, end);
2484                 }
2485                 else if (tlen < end) {
2486                         checkfilerange(fd, namestr, prev, start, tlen);
2487                         checkfilezeros(fd, namestr, tlen, end);
2488                 }
2489                 else {
2490                         checkfilerange(fd, namestr, prev, start, end);
2491                 }
2492         }
2493         else if (change->type == FC_WRITE) {
2494                 off_t wstart, wend;
2495                 struct fschange *prev;
2496                 unsigned code, seq;
2497                 off_t oldfilesize, zerostart;
2498 
2499                 /*
2500                  * The most recent change was a write.
2501                  */
2502 
2503                 wstart = change->fc_write.pos;
2504                 wend = change->fc_write.pos + change->fc_write.len;
2505                 prev = change->fc_write.prev_thisfile;
2506                 code = change->fc_write.code;
2507                 seq = change->fc_write.seq;
2508                 oldfilesize = change->fc_write.oldfilesize;
2509 
2510                 /*
2511                  * ZEROSTART (where we begin to allow the file to
2512                  * contain zeros) should be the point at the write
2513                  * where we began to extend the file, if any.
2514                  */
2515                 if (oldfilesize < wstart) {
2516                         zerostart = wstart;
2517                 }
2518                 else if (oldfilesize < wend) {
2519                         zerostart = oldfilesize;
2520                 }
2521                 else {
2522                         zerostart = wend;
2523                 }
2524 
2525                 /*
2526                  * Six cases for the range overlap:
2527                  *     (1)    (2)    (3)    (4)   (5)    (6)
2528                  *    **     ***    ******   *     ***      **
2529                  *       ***   ***    ***   ***  ***    ***
2530                  *
2531                  * We call checkfilerange recursively using the
2532                  * previous change for this file for the
2533                  * non-overlapping parts, because this write didn't
2534                  * touch those so they must reflect the previous file
2535                  * state; and checkfiledata for the overlapping parts
2536                  * that this write did touch.
2537                  */
2538 
2539                 if (end <= wstart || start >= wend) {
2540                         /* cases 1 and 6 */
2541                         checkfilerange(fd, namestr, prev, start, end);
2542                 }
2543                 else {
2544                         /* cases 2-5 */
2545                         if (start < wstart) {
2546                                 /* case 2 or 3 */
2547                                 checkfilerange(fd, namestr, prev,
2548                                                start, wstart);
2549                         }
2550                         checkfiledata(fd, namestr, code, seq, zerostart,
2551                                       wstart, start, end, wend);
2552                         if (end > wend) {
2553                                 /* cases 3 or 5 */
2554                                 checkfilerange(fd, namestr, prev, wend, end);
2555                         }
2556                 }
2557         }
2558         else if (change->type == FC_RENAMEFILE) {
2559                 struct fschange *prev;
2560 
2561                 /*
2562                  * The most recent change was a rename. As rename
2563                  * doesn't affect contents, recurse on the previous
2564                  * change affecting the file.
2565                  */
2566 
2567                 prev = change->fc_renamefile.prev_movedfile;
2568                 checkfilerange(fd, namestr, prev, start, end);
2569         }
2570         else {
2571                 assert(change->type == FC_CREAT);
2572 
2573                 /*
2574                  * The most recent change was the file being created.
2575                  * Like with truncate, check that the range is zero
2576                  * even though it's past EOF, as a later write might
2577                  * have converted past-EOF space into a hole in a
2578                  * sparse file.
2579                  */
2580                 checkfilezeros(fd, namestr, start, end);
2581         }
2582 }

2586  * FD, whose name is NAMESTR and size is FILESIZE.
2587  *
2588  * For writes it checks the data, so should be accurate; for truncate
2589  * all it can check is the size, so it can readily be fooled by
2590  * sequences of truncates or multiple truncates to the same length.
2591  */
2592 static

2599

2601             case FC_TRUNCATE:
2602                 return filesize == change->fc_truncate.len;
2603             case FC_WRITE:
2604                 pos = change->fc_write.pos;
2605                 len = change->fc_write.len;
2606                 code = change->fc_write.code;
2607                 seq = change->fc_write.seq;
2608                 oldfilesize = change->fc_write.oldfilesize;
2609                 if (oldfilesize < pos) {

2762          * length?

2804 /*
2805  * Compare the on-disk state we see to the model we've built.
2806  */
2807 void
2808 checkfs(void)
2809 {
2810         struct fschange *change, *best;
2811         unsigned score, bestscore;
2812 
2813         /*
2814          * We just built the model; talk about it.
2815          */
2816         printf("Established %u versions across %u directories and %u files\n",
2817                changes->version + 1, nextdirnum, nextfilenum);
2818 
2819         /*
2820          * Inspect the volume state we've got. Initializes the global
2821          * FOUND holding the found volume state.
2822          */

2825                found_subdirs, found_files);
2826 
2827         /*
2828          * Rewind the model state to the beginning.
2829          */
2830         rewindstate();
2831 
2832         /*
2833          * Loop through all the changes; apply each one to the model
2834          * state, and score the model state against the found state.
2835          * Keep track of the best matching version.
2836          *
2837          * XXX: there might be several versions with the same score,
2838          * in which case we'll blindly take the most recent one. If
2839          * there are several versions that match exactly (which can
2840          * easily happen for workloads that don't explicitly avoid it)
2841          * we'll also blindly take the most recent one, and then the
2842          * matching of file contents can blow up. We should collect
2843          * all the versions with the same score, and score each one
2844          * on file contents too. Or something like that.
2845          */
2846 
2847         change = firstchange;
2848         assert(change->type == FC_NEWFS);
2849         best = NULL;
2850         bestscore = 0;
2851 
2852         while (change != NULL) {
2853                 apply_change(state, change);
2854                 score = compare_objects(state, found);
2855                 if (best == NULL || score <= bestscore) {
2856                         best = change;
2857                         bestscore = score;
2858                 }
2859                 //printf("version %u score %u\n", change->version, score);
2860                 change = change->next;
2861         }
2862         assert(best != NULL);
2863 
2864         /*
2865          * Set the model state to the best matching state we found.
2866          */
2867
