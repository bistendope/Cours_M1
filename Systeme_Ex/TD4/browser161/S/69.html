  44  * enough to struggle off the ground. You should replace all of this
  45  * code while doing the VM assignment. In fact, starting in that
  46  * assignment, this file is not included in your kernel!

  48  * NOTE: it's been found over the years that students often begin on
  49  * the VM assignment by copying dumbvm.c and trying to improve it.

  54  * what a VM system is supposed to do, and understanding what corners

  73 /*
  74  * Check if we're in a context that can sleep. While most of the
  75  * operations in dumbvm don't in fact sleep, in a real VM system many
  76  * of them would. In those, assert that sleeping is ok. This helps
  77  * avoid the situation where syscall-layer code that works ok with
  78  * dumbvm starts blowing up during the VM assignment.
  79  */
  80 static
  81 void
  82 dumbvm_can_sleep(void)
  83 {
  84         if (CURCPU_EXISTS()) {
  85                 /* must not hold spinlocks */
  86                 KASSERT(curcpu->c_spinlocks == 0);
  87 
  88                 /* must not be in an interrupt handler */
  89                 KASSERT(curthread->t_in_interrupt == 0);
  90         }
  91 }

  93 static
  94 paddr_t
  95 getppages(unsigned long npages)
  96 {
  97         paddr_t addr;
  98 
  99         spinlock_acquire(&stealmem_lock);
 100 
 101         addr = ram_stealmem(npages);
 102 
 103         spinlock_release(&stealmem_lock);
 104         return addr;
 105 }

 107 /* Allocate/free some kernel-space virtual pages */
 108 vaddr_t
 109 alloc_kpages(unsigned npages)
 110 {
 111         paddr_t pa;
 112 
 113         dumbvm_can_sleep();
 114         pa = getppages(npages);
 115         if (pa==0) {
 116                 return 0;
 117         }
 118         return PADDR_TO_KVADDR(pa);
 119 }

 136 int
 137 vm_fault(int faulttype, vaddr_t faultaddress)
 138 {
 139         vaddr_t vbase1, vtop1, vbase2, vtop2, stackbase, stacktop;
 140         paddr_t paddr;
 141         int i;
 142         uint32_t ehi, elo;
 143         struct addrspace *as;
 144         int spl;
 145 
 146         faultaddress &= PAGE_FRAME;
 147 
 148         DEBUG(DB_VM, "dumbvm: fault: 0x%x\n", faultaddress);
 149 
 150         switch (faulttype) {
 151             case VM_FAULT_READONLY:
 152                 /* We always create pages read-write, so we can't get this */
 153                 panic("dumbvm: got VM_FAULT_READONLY\n");
 154             case VM_FAULT_READ:
 155             case VM_FAULT_WRITE:
 156                 break;
 157             default:
 158                 return EINVAL;
 159         }
 160 
 161         if (curproc == NULL) {
 162                 /*
 163                  * No process. This is probably a kernel fault early
 164                  * in boot. Return EFAULT so as to panic instead of
 165                  * getting into an infinite faulting loop.
 166                  */
 167                 return EFAULT;
 168         }
 169 
 170         as = proc_getas();
 171         if (as == NULL) {
 172                 /*
 173                  * No address space set up. This is probably also a
 174                  * kernel fault early in boot.
 175                  */
 176                 return EFAULT;
 177         }
 178 
 179         /* Assert that the address space has been set up properly. */
 180         KASSERT(as->as_vbase1 != 0);
 181         KASSERT(as->as_pbase1 != 0);
 182         KASSERT(as->as_npages1 != 0);
 183         KASSERT(as->as_vbase2 != 0);
 184         KASSERT(as->as_pbase2 != 0);
 185         KASSERT(as->as_npages2 != 0);
 186         KASSERT(as->as_stackpbase != 0);
 187         KASSERT((as->as_vbase1 & PAGE_FRAME) == as->as_vbase1);
 188         KASSERT((as->as_pbase1 & PAGE_FRAME) == as->as_pbase1);
 189         KASSERT((as->as_vbase2 & PAGE_FRAME) == as->as_vbase2);
 190         KASSERT((as->as_pbase2 & PAGE_FRAME) == as->as_pbase2);
 191         KASSERT((as->as_stackpbase & PAGE_FRAME) == as->as_stackpbase);
 192 
 193         vbase1 = as->as_vbase1;
 194         vtop1 = vbase1 + as->as_npages1 * PAGE_SIZE;
 195         vbase2 = as->as_vbase2;
 196         vtop2 = vbase2 + as->as_npages2 * PAGE_SIZE;
 197         stackbase = USERSTACK - DUMBVM_STACKPAGES * PAGE_SIZE;
 198         stacktop = USERSTACK;
 199 
 200         if (faultaddress >= vbase1 && faultaddress < vtop1) {
 201                 paddr = (faultaddress - vbase1) + as->as_pbase1;
 202         }
 203         else if (faultaddress >= vbase2 && faultaddress < vtop2) {
 204                 paddr = (faultaddress - vbase2) + as->as_pbase2;
 205         }
 206         else if (faultaddress >= stackbase && faultaddress < stacktop) {
 207                 paddr = (faultaddress - stackbase) + as->as_stackpbase;
 208         }
 209         else {
 210                 return EFAULT;
 211         }
 212 
 213         /* make sure it's page-aligned */
 214         KASSERT((paddr & PAGE_FRAME) == paddr);
 215 
 216         /* Disable interrupts on this CPU while frobbing the TLB. */
 217         spl = splhigh();
 218 
 219         for (i=0; i<NUM_TLB; i++) {
 220                 tlb_read(&ehi, &elo, i);
 221                 if (elo & TLBLO_VALID) {
 222                         continue;
 223                 }
 224                 ehi = faultaddress;
 225                 elo = paddr | TLBLO_DIRTY | TLBLO_VALID;
 226                 DEBUG(DB_VM, "dumbvm: 0x%x -> 0x%x\n", faultaddress, paddr);
 227                 tlb_write(ehi, elo, i);
 228                 splx(spl);
 229                 return 0;
 230         }
 231 
 232         kprintf("dumbvm: Ran out of TLB entries - cannot handle page fault\n");
 233         splx(spl);
 234         return EFAULT;
 235 }

 239 {

 241         if (as==NULL) {
 242                 return NULL;

 263 void
 264 as_activate(void)
 265 {
 266         int i, spl;
 267         struct addrspace *as;
 268 
 269         as = proc_getas();
 270         if (as == NULL) {
 271                 return;
 272         }
 273 
 274         /* Disable interrupts on this CPU while frobbing the TLB. */
 275         spl = splhigh();
 276 
 277         for (i=0; i<NUM_TLB; i++) {
 278                 tlb_write(TLBHI_INVALID(i), TLBLO_INVALID(), i);
 279         }
 280 
 281         splx(spl);
 282 }

 290 int
 291 as_define_region(struct addrspace *as, vaddr_t vaddr, size_t sz,
 292                  int readable, int writeable, int executable)
 293 {
 294         size_t npages;
 295 
 296         dumbvm_can_sleep();
 297 
 298         /* Align the region. First, the base... */
 299         sz += vaddr & ~(vaddr_t)PAGE_FRAME;
 300         vaddr &= PAGE_FRAME;
 301 
 302         /* ...and now the length. */
 303         sz = (sz + PAGE_SIZE - 1) & PAGE_FRAME;
 304 
 305         npages = sz / PAGE_SIZE;
 306 
 307         /* We don't use these - all pages are read-write */
 308         (void)readable;
 309         (void)writeable;
 310         (void)executable;
 311 
 312         if (as->as_vbase1 == 0) {
 313                 as->as_vbase1 = vaddr;
 314                 as->as_npages1 = npages;
 315                 return 0;
 316         }
 317 
 318         if (as->as_vbase2 == 0) {
 319                 as->as_vbase2 = vaddr;
 320                 as->as_npages2 = npages;
 321                 return 0;
 322         }
 323 
 324         /*
 325          * Support for more than two regions is not available.
 326          */
 327         kprintf("dumbvm: Warning: too many regions\n");
 328         return ENOSYS;
 329 }

 338 int
 339 as_prepare_load(struct addrspace *as)
 340 {
 341         KASSERT(as->as_pbase1 == 0);
 342         KASSERT(as->as_pbase2 == 0);
 343         KASSERT(as->as_stackpbase == 0);
 344 
 345         dumbvm_can_sleep();
 346 
 347         as->as_pbase1 = getppages(as->as_npages1);
 348         if (as->as_pbase1 == 0) {
 349                 return ENOMEM;
 350         }
 351 
 352         as->as_pbase2 = getppages(as->as_npages2);
 353         if (as->as_pbase2 == 0) {
 354                 return ENOMEM;
 355         }
 356 
 357         as->as_stackpbase = getppages(DUMBVM_STACKPAGES);
 358         if (as->as_stackpbase == 0) {
 359                 return ENOMEM;
 360         }
 361 
 362         as_zero_region(as->as_pbase1, as->as_npages1);
 363         as_zero_region(as->as_pbase2, as->as_npages2);
 364         as_zero_region(as->as_stackpbase, DUMBVM_STACKPAGES);
 365 
 366         return 0;
 367 }

 386 int
 387 as_copy(struct addrspace *old, struct addrspace **ret)
 388 {
 389         struct addrspace *new;
 390 
 391         dumbvm_can_sleep();
 392 
 393         new = as_create();
 394         if (new==NULL) {
 395                 return ENOMEM;
 396         }
 397 
 398         new->as_vbase1 = old->as_vbase1;
 399         new->as_npages1 = old->as_npages1;
 400         new->as_vbase2 = old->as_vbase2;
 401         new->as_npages2 = old->as_npages2;
 402 
 403         /* (Mis)use as_prepare_load to allocate some physical memory. */
 404         if (as_prepare_load(new)) {
 405                 as_destroy(new);
 406                 return ENOMEM;
 407         }
 408 
 409         KASSERT(new->as_pbase1 != 0);
 410         KASSERT(new->as_pbase2 != 0);
 411         KASSERT(new->as_stackpbase != 0);
 412 
 413         memmove((void *)PADDR_TO_KVADDR(new->as_pbase1),
 414                 (const void *)PADDR_TO_KVADDR(old->as_pbase1),
 415                 old->as_npages1*PAGE_SIZE);
 416 
 417         memmove((void *)PADDR_TO_KVADDR(new->as_pbase2),
 418                 (const void *)PADDR_TO_KVADDR(old->as_pbase2),
 419                 old->as_npages2*PAGE_SIZE);
 420 
 421         memmove((void *)PADDR_TO_KVADDR(new->as_stackpbase),
 422                 (const void *)PADDR_TO_KVADDR(old->as_stackpbase),
 423                 DUMBVM_STACKPAGES*PAGE_SIZE);
 424 
 425         *ret = new;
 426         return 0;
 427 }
