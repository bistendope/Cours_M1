  30 #ifndef _SFS_H_
  31 #define _SFS_H_
  32 
  33 
  34 /*
  35  * Header for SFS, the Simple File System.
  36  */
  37 
  38 
  39 /*
  40  * Get abstract structure definitions
  41  */
  42 #include <fs.h>
  43 #include <vnode.h>
  44 
  45 /*
  46  * Get on-disk structures and constants that are made available to
  47  * userland for the benefit of mksfs, dumpsfs, etc.
  48  */
  49 #include <kern/sfs.h>
  50 
  51 /*
  52  * In-memory inode
  53  */
  54 struct sfs_vnode {
  55         struct vnode sv_absvn;          /* abstract vnode structure */
  56         struct sfs_dinode sv_i;         /* copy of on-disk inode */
  57         uint32_t sv_ino;                /* inode number */
  58         bool sv_dirty;                  /* true if sv_i modified */
  59 };
  60 
  61 /*
  62  * In-memory info for a whole fs volume
  63  */
  64 struct sfs_fs {
  65         struct fs sfs_absfs;            /* abstract filesystem structure */
  66         struct sfs_superblock sfs_sb;   /* copy of on-disk superblock */
  67         bool sfs_superdirty;            /* true if superblock modified */
  68         struct device *sfs_device;      /* device mounted on */
  69         struct vnodearray *sfs_vnodes;  /* vnodes loaded into memory */
  70         struct bitmap *sfs_freemap;     /* blocks in use are marked 1 */
  71         bool sfs_freemapdirty;          /* true if freemap modified */
  72 };
  73 
  74 /*
  75  * Function for mounting a sfs (calls vfs_mount)
  76  */
  77 int sfs_mount(const char *device);
  78 
  79 
  80 #endif /* _SFS_H_ */
