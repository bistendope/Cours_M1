  46 /*
  47  * Called for each open().
  48  *
  49  * We reject O_APPEND.
  50  */
  51 static
  52 int
  53 dev_eachopen(struct vnode *v, int flags)
  54 {
  55         struct device *d = v->vn_data;
  56 
  57         if (flags & (O_CREAT | O_TRUNC | O_EXCL | O_APPEND)) {
  58                 return EINVAL;
  59         }
  60 
  61         return DEVOP_EACHOPEN(d, flags);
  62 }

  77 /*
  78  * Check a seek position.
  79  *
  80  * For block devices, require block alignment.
  81  * For character devices, we should prohibit seeking entirely, but
  82  * for the moment we need to accept any position. (XXX)
  83  */
  84 static
  85 int
  86 dev_tryseek(struct device *d, off_t pos)
  87 {
  88         if (d->d_blocks > 0) {
  89                 if ((pos % d->d_blocksize)!=0) {
  90                         /* not block-aligned */
  91                         return EINVAL;
  92                 }
  93                 if (pos / d->d_blocksize >= d->d_blocks) {
  94                         /* off the end */
  95                         return EINVAL;
  96                 }
  97         }
  98         else {
  99                 //return ESPIPE;
 100         }
 101         return 0;
 102 }

 104 /*
 105  * Called for read. Hand off to DEVOP_IO.
 106  */
 107 static
 108 int
 109 dev_read(struct vnode *v, struct uio *uio)
 110 {
 111         struct device *d = v->vn_data;
 112         int result;
 113 
 114         result = dev_tryseek(d, uio->uio_offset);
 115         if (result) {
 116                 return result;
 117         }
 118 
 119         KASSERT(uio->uio_rw == UIO_READ);
 120         return DEVOP_IO(d, uio);
 121 }

 153 /*
 154  * Called for stat().
 155  * Set the type and the size (block devices only).
 156  * The link count for a device is always 1.
 157  */
 158 static
 159 int
 160 dev_stat(struct vnode *v, struct stat *statbuf)
 161 {
 162         struct device *d = v->vn_data;
 163         int result;
 164 
 165         bzero(statbuf, sizeof(struct stat));
 166 
 167         if (d->d_blocks > 0) {
 168                 statbuf->st_size = d->d_blocks * d->d_blocksize;
 169                 statbuf->st_blksize = d->d_blocksize;
 170         }
 171         else {
 172                 statbuf->st_size = 0;
 173         }
 174 
 175         result = VOP_GETTYPE(v, &statbuf->st_mode);
 176         if (result) {
 177                 return result;
 178         }
 179         /* Make up some plausible default permissions. */
 180         statbuf->st_mode |= 0600;
 181 
 182         statbuf->st_nlink = 1;
 183         statbuf->st_blocks = d->d_blocks;
 184 
 185         /* The device number this device sits on (in OS/161, it doesn't) */
 186         statbuf->st_dev = 0;
 187 
 188         /* The device number this device *is* */
 189         statbuf->st_rdev = d->d_devnumber;
 190 
 191         return 0;
 192 }

 194 /*
 195  * Return the type. A device is a "block device" if it has a known
 196  * length. A device that generates data in a stream is a "character
 197  * device".
 198  */
 199 static
 200 int
 201 dev_gettype(struct vnode *v, mode_t *ret)
 202 {
 203         struct device *d = v->vn_data;
 204         if (d->d_blocks > 0) {
 205                 *ret = S_IFBLK;
 206         }
 207         else {
 208                 *ret = S_IFCHR;
 209         }
 210         return 0;
 211 }

 251 /*
 252  * For ftruncate().
 253  */
 254 static
 255 int
 256 dev_truncate(struct vnode *v, off_t len)
 257 {
 258         struct device *d = v->vn_data;
 259 
 260         /*
 261          * Allow truncating to the object's own size, if it has one.
 262          */
 263         if (d->d_blocks > 0 && (off_t)(d->d_blocks*d->d_blocksize) == len) {
 264                 return 0;
 265         }
 266 
 267         return EINVAL;
 268 }

 292 /*
 293  * Name lookup.
 294  *
 295  * One interesting feature of device:name pathname syntax is that you
 296  * can implement pathnames on arbitrary devices. For instance, if you
 297  * had a graphics device that supported multiple resolutions (which we
 298  * don't), you might arrange things so that you could open it with
 299  * pathnames like "video:800x600/24bpp" in order to select the operating
 300  * mode.
 301  *
 302  * However, we have no support for this in the base system.
 303  */
 304 static
 305 int
 306 dev_lookup(struct vnode *dir,
 307            char *pathname, struct vnode **result)
 308 {
 309         /*
 310          * If the path was "device:", we get "". For that, return self.
 311          * Anything else is an error.
 312          * Increment the ref count of the vnode before returning it.
 313          */
 314         if (strlen(pathname)>0) {
 315                 return ENOENT;
 316         }
 317         VOP_INCREF(dir);
 318         *result = dir;
 319         return 0;
 320 }

 353 /*
 354  * Function to create a vnode for a VFS device.
 355  */
 356 struct vnode *
 357 dev_create_vnode(struct device *dev)
 358 {
 359         int result;
 360         struct vnode *v;
 361 
 362         v = kmalloc(sizeof(struct vnode));
 363         if (v==NULL) {
 364                 return NULL;
 365         }
 366 
 367         result = vnode_init(v, &dev_vnode_ops, NULL, dev);
 368         if (result != 0) {
 369                 panic("While creating vnode for device: vnode_init: %s\n",
 370                       strerror(result));
 371         }
 372 
 373         return v;
 374 }

 376 /*
 377  * Undo dev_create_vnode.
 378  *
 379  * Note: this is only used in failure paths; we don't support
 380  * hotpluggable devices, so once a device is attached it's permanent.
 381  */
 382 void
 383 dev_uncreate_vnode(struct vnode *vn)
 384 {
 385         KASSERT(vn->vn_ops == &dev_vnode_ops);
 386         vnode_cleanup(vn);
 387         kfree(vn);
 388 }
