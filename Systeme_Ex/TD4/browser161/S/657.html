 171 static
 172 unsigned
 173 sizeblocks(enum sizes sz)
 174 {
 175         /*
 176          * XXX for now hardwire things we know about SFS
 177          */
 178 #define BLOCKSIZE /*SFS_BLOCKSIZE*/ 512
 179         static const unsigned ndb = /*SFS_NDIRECT*/ 15;
 180         static const unsigned dbperidb = /*SFS_DBPERIDB*/ 128;
 181 
 182         switch (sz) {
 183             case SIZE_ONE:
 184                 /* one block; 512 bytes */
 185                 return 1;
 186             case SIZE_SMALL:
 187                 /* fits in direct blocks only; 7.5K */
 188                 return ndb;
 189             case SIZE_MEDIUM:
 190                 /* uses an indirect block; ~40K */
 191                 return ndb + dbperidb / 2;
 192             case SIZE_LARGE:
 193                 /* uses a double-indirect block; 4.2M */
 194                 return ndb + dbperidb + dbperidb * dbperidb / 2;
 195             case SIZE_LARGEPLUS:
 196                 /* requires a triple-indirect block; 8.5M */
 197                 return ndb + dbperidb + dbperidb * dbperidb + dbperidb / 2;
 198         }
 199         assert(0);
 200         return 12;
 201 }

 204 unsigned
 205 sizebytes(enum sizes sz)

 208 }

 213 static
 214 void
 215 file_randomwrite(struct file *f, enum sizes sz,
 216                  unsigned startskip, unsigned endskip)
 217 {
 218         unsigned nblocks, nwrites, i;
 219         unsigned blocknum;
 220         off_t pos;
 221 
 222         nblocks = sizeblocks(sz);
 223         assert(nblocks > startskip + endskip);
 224 
 225         nwrites = nblocks/6;
 226         if (nwrites < 2) {
 227                 nwrites = 2;
 228         }
 229 
 230         nblocks -= startskip + endskip;
 231         for (i=0; i<nwrites; i++) {
 232                 blocknum = startskip + random() % nblocks;
 233                 pos = (off_t)BLOCKSIZE * blocknum;
 234                 op_write(f, pos, BLOCKSIZE);
 235         }
 236 }

1586 /*
1587  * Sequence of randomly chosen arbitrary directory operations.
1588  */
1589 
1590 #define ISNT    0
1591 #define IS_FILE 1
1592 #define IS_DIR  2
1593 
1594 static
1595 bool
1596 randop(unsigned testcode, unsigned *exists, unsigned numfiles,
1597        bool dofileops, struct file **files)
1598 {
1599         enum sizes sz;
1600         unsigned filenum, filenum2;
1601 
1602         switch (random() % (dofileops ? 8 : 6)) {
1603             case 0:
1604                 /* create a file */
1605                 filenum = random() % numfiles;
1606                 if (exists[filenum] == ISNT) {
1607                         if (dofileops) {
1608                                 sz = randsize();
1609                         }
1610                         else {
1611                                 sz = SIZE_ONE;
1612                         }
1613                         writenewfile(testcode, filenum, sz);
1614                         exists[filenum] = IS_FILE;
1615                         return true;
1616                 }
1617                 break;
1618             case 1:
1619                 /* make a directory */
1620                 filenum = random() % numfiles;
1621                 if (exists[filenum] == ISNT) {
1622                         op_mkdir(filenum);
1623                         exists[filenum] = IS_DIR;
1624                         return true;
1625                 }
1626                 break;
1627             case 2:
1628 #if 0 /* XXX currently no link() in OS/161 */
1629                 /* hardlink a file */
1630                 filenum = random() % numfiles;
1631                 filenum2 = random() % numfiles;
1632                 if (exists[filenum] == IS_FILE &&
1633                     exists[filenum2] == ISNT) {
1634                         op_link(filenum, filenum2);
1635                         exists[filenum2] = IS_FILE;
1636                         return true;
1637                 }
1638 #endif
1639                 break;
1640             case 3:
1641                 /* unlink a file */
1642                 filenum = random() % numfiles;
1643                 if (exists[filenum] == IS_FILE) {
1644                         op_unlink(filenum);
1645                         exists[filenum] = ISNT;
1646                         if (files != NULL && files[filenum] != NULL) {
1647                                 op_close(files[filenum]);
1648                                 files[filenum] = NULL;
1649                         }
1650                         return true;
1651                 }
1652                 break;
1653             case 4:
1654                 /* rmdir a dir */
1655                 filenum = random() % numfiles;
1656                 if (exists[filenum] == IS_DIR) {
1657                         op_rmdir(filenum);
1658                         exists[filenum] = ISNT;
1659                         return true;
1660                 }
1661                 break;
1662             case 5:
1663                 /* rename something */
1664                 filenum = random() % numfiles;
1665                 filenum2 = random() % numfiles;
1666 
1667                 /* XXX currently something in check.c horks on this case */
1668                 if (filenum == filenum2) {
1669                         break;
1670                 }
1671 
1672                 if (exists[filenum] != ISNT &&
1673                     (exists[filenum2] == ISNT ||
1674                      exists[filenum2] == exists[filenum])) {
1675                         op_rename(filenum, filenum2);
1676                         if (filenum != filenum2) {
1677                                 exists[filenum2] = exists[filenum];
1678                                 exists[filenum] = ISNT;
1679                         }
1680                         return true;
1681                 }
1682                 break;
1683             case 6:
1684                 /* truncate something */
1685 
1686                 assert(dofileops);
1687                 assert(files != NULL);
1688 
1689                 filenum = random() % numfiles;
1690                 if (exists[filenum] == IS_FILE) {
1691                         if (files[filenum] == NULL) {
1692                                 files[filenum] = op_open(testcode, filenum, 0);
1693                         }
1694                         randtruncate(files[filenum]);
1695                         return true;
1696                 }
1697                 break;
1698             case 7:
1699                 /* write to something */
1700 
1701                 assert(dofileops);
1702                 assert(files != NULL);
1703 
1704                 filenum = random() % numfiles;
1705                 if (exists[filenum] == IS_FILE) {
1706                         if (files[filenum] == NULL) {
1707                                 files[filenum] = op_open(testcode, filenum, 0);
1708                         }
1709                         randwrite(files[filenum]);
1710                         return true;
1711                 }
1712                 break;
1713         }
1714         return false;
1715 }
