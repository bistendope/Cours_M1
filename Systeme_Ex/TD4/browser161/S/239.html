  55 ////////////////////////////////////////////////////////////
  56 //
  57 // Pool-based subpage allocator.
  58 //
  59 // It works like this:
  60 //
  61 //    We allocate one page at a time and fill it with objects of size k,
  62 //    for various k. Each page has its own freelist, maintained by a
  63 //    linked list in the first word of each object. Each page also has a
  64 //    freecount, so we know when the page is completely free and can
  65 //    release it.
  66 //
  67 //    No assumptions are made about the sizes k; they need not be
  68 //    powers of two. Note, however, that malloc must always return
  69 //    pointers aligned to the maximum alignment requirements of the
  70 //    platform; thus block sizes must at least be multiples of 4,
  71 //    preferably 8. They must also be at least sizeof(struct
  72 //    freelist). It is only worth defining an additional block size if
  73 //    more blocks would fit on a page than with the existing block
  74 //    sizes, and large numbers of items of the new size are allocated.
  75 //
  76 //    The free counts and addresses of the pages are maintained in
  77 //    another list.  Maintaining this table is a nuisance, because it
  78 //    cannot recursively use the subpage allocator. (We could probably
  79 //    make that work, but it would be painful.)
  80 //
  81 
  82 ////////////////////////////////////////
  83 
  84 /*
  85  * Debugging modes.
  86  *
  87  * SLOW enables consistency checks; this will check the integrity of
  88  * kernel heap pages that kmalloc touches in the course of ordinary
  89  * operations.
  90  *
  91  * SLOWER enables SLOW and also checks the integrity of all heap pages
  92  * at strategic points.
  93  *
  94  * GUARDS enables the use of guard bands on subpage allocations, so as
  95  * to catch simple off-the-end accesses. By default the guard bands
  96  * are checked only at kfree() time. This is independent of SLOW and
  97  * SLOWER. Note that the extra space used by the guard bands increases
  98  * memory usage (possibly by a lot depending on the sizes allocated) and
  99  * will likely produce a different heap layout, so it's likely to cause
 100  * malloc-related bugs to manifest differently.
 101  *
 102  * LABELS records the allocation site and a generation number for each
 103  * allocation and is useful for tracking down memory leaks.
 104  *
 105  * On top of these one can enable the following:
 106  *
 107  * CHECKBEEF checks that free blocks still contain 0xdeadbeef when
 108  * checking kernel heap pages with SLOW and SLOWER. This is quite slow
 109  * in its own right.
 110  *
 111  * CHECKGUARDS checks that allocated blocks' guard bands are intact
 112  * when checking kernel heap pages with SLOW and SLOWER. This is also
 113  * quite slow in its own right.
 114  */

 215 /*
 216  * Allocate a page to hold pagerefs.
 217  */
 218 static
 219 void
 220 allocpagerefpage(struct kheap_root *root)
 221 {
 222         vaddr_t va;
 223 
 224         KASSERT(root->page == NULL);
 225 
 226         /*
 227          * We release the spinlock while calling alloc_kpages. This
 228          * avoids deadlock if alloc_kpages needs to come back here.
 229          * Note that this means things can change behind our back...
 230          */
 231         spinlock_release(&kmalloc_spinlock);
 232         va = alloc_kpages(1);
 233         spinlock_acquire(&kmalloc_spinlock);
 234         if (va == 0) {
 235                 kprintf("kmalloc: Couldn't get a pageref page\n");
 236                 return;
 237         }
 238         KASSERT(va % PAGE_SIZE == 0);
 239 
 240         if (root->page != NULL) {
 241                 /* Oops, somebody else allocated it. */
 242                 spinlock_release(&kmalloc_spinlock);
 243                 free_kpages(va);
 244                 spinlock_acquire(&kmalloc_spinlock);
 245                 /* Once allocated it isn't ever freed. */
 246                 KASSERT(root->page != NULL);
 247                 return;
 248         }
 249 
 250         root->page = (struct pagerefpage *)va;
 251 }

 253 /*
 254  * Allocate a pageref structure.
 255  */
 256 static
 257 struct pageref *
 258 allocpageref(void)
 259 {
 260         unsigned i,j;
 261         uint32_t k;
 262         unsigned whichroot;
 263         struct kheap_root *root;
 264 
 265         for (whichroot=0; whichroot < NUM_PAGEREFPAGES; whichroot++) {
 266                 root = &kheaproots[whichroot];
 267                 if (root->numinuse >= NPAGEREFS_PER_PAGE) {
 268                         continue;
 269                 }
 270 
 271                 /*
 272                  * This should probably not be a linear search.
 273                  */
 274                 for (i=0; i<INUSE_WORDS; i++) {
 275                         if (root->pagerefs_inuse[i]==0xffffffff) {
 276                                 /* full */
 277                                 continue;
 278                         }
 279                         for (k=1,j=0; k!=0; k<<=1,j++) {
 280                                 if ((root->pagerefs_inuse[i] & k)==0) {
 281                                         root->pagerefs_inuse[i] |= k;
 282                                         root->numinuse++;
 283                                         if (root->page == NULL) {
 284                                                 allocpagerefpage(root);
 285                                         }
 286                                         if (root->page == NULL) {
 287                                                 return NULL;
 288                                         }
 289                                         return &root->page->refs[i*32 + j];
 290                                 }
 291                         }
 292                         KASSERT(0);
 293                 }
 294         }
 295 
 296         /* ran out */
 297         return NULL;
 298 }

 300 /*
 301  * Release a pageref structure.
 302  */
 303 static
 304 void
 305 freepageref(struct pageref *p)
 306 {
 307         size_t i, j;
 308         uint32_t k;
 309         unsigned whichroot;
 310         struct kheap_root *root;
 311         struct pagerefpage *page;
 312 
 313         for (whichroot=0; whichroot < NUM_PAGEREFPAGES; whichroot++) {
 314                 root = &kheaproots[whichroot];
 315 
 316                 page = root->page;
 317                 if (page == NULL) {
 318                         KASSERT(root->numinuse == 0);
 319                         continue;
 320                 }
 321 
 322                 j = p-page->refs;
 323                 /* note: j is unsigned, don't test < 0 */
 324                 if (j < NPAGEREFS_PER_PAGE) {
 325                         /* on this page */
 326                         i = j/32;
 327                         k = ((uint32_t)1) << (j%32);
 328                         KASSERT((root->pagerefs_inuse[i] & k) != 0);
 329                         root->pagerefs_inuse[i] &= ~k;
 330                         KASSERT(root->numinuse > 0);
 331                         root->numinuse--;
 332                         return;
 333                 }
 334         }
 335         /* pageref wasn't on any of the pages */
 336         KASSERT(0);
 337 }

 350 #ifdef GUARDS
 351 
 352 /* Space returned to the client is filled with GUARD_RETBYTE */
 353 #define GUARD_RETBYTE 0xa9
 354 /* Padding space (internal fragmentation loss) is filled with GUARD_FILLBYTE */
 355 #define GUARD_FILLBYTE 0xba
 356 /* The guard bands on an allocated block should contain GUARD_HALFWORD */
 357 #define GUARD_HALFWORD 0xb0b0
 358 
 359 /* The guard scheme uses 8 bytes per block. */
 360 #define GUARD_OVERHEAD 8
 361 
 362 /* Pointers are offset by 4 bytes when guards are in use. */
 363 #define GUARD_PTROFFSET 4
 364 
 365 /*
 366  * Set up the guard values in a block we're about to return.
 367  */
 368 static
 369 void *
 370 establishguardband(void *block, size_t clientsize, size_t blocksize)
 371 {
 372         vaddr_t lowguard, lowsize, data, enddata, highguard, highsize, i;
 373 
 374         KASSERT(clientsize + GUARD_OVERHEAD <= blocksize);
 375         KASSERT(clientsize < 65536U);
 376 
 377         lowguard = (vaddr_t)block;
 378         lowsize = lowguard + 2;
 379         data = lowsize + 2;
 380         enddata = data + clientsize;
 381         highguard = lowguard + blocksize - 4;
 382         highsize = highguard + 2;
 383 
 384         *(uint16_t *)lowguard = GUARD_HALFWORD;
 385         *(uint16_t *)lowsize = clientsize;
 386         for (i=data; i<enddata; i++) {
 387                 *(uint8_t *)i = GUARD_RETBYTE;
 388         }
 389         for (i=enddata; i<highguard; i++) {
 390                 *(uint8_t *)i = GUARD_FILLBYTE;
 391         }
 392         *(uint16_t *)highguard = GUARD_HALFWORD;
 393         *(uint16_t *)highsize = clientsize;
 394 
 395         return (void *)data;
 396 }

 398 /*
 399  * Validate the guard values in an existing block.
 400  */
 401 static
 402 void
 403 checkguardband(vaddr_t blockaddr, size_t smallerblocksize, size_t blocksize)
 404 {
 405         /*
 406          * The first two bytes of the block are the lower guard band.
 407          * The next two bytes are the real size (the size of the
 408          * client data). The last four bytes of the block duplicate
 409          * this info. In between the real data and the upper guard
 410          * band should be filled with GUARD_FILLBYTE.
 411          *
 412          * If the guard values are wrong, or the low and high sizes
 413          * don't match, or the size is out of range, by far the most
 414          * likely explanation is that something ran past the bounds of
 415          * its memory block.
 416          */
 417         vaddr_t lowguard, lowsize, data, enddata, highguard, highsize, i;
 418         unsigned clientsize;
 419 
 420         lowguard = blockaddr;
 421         lowsize = lowguard + 2;
 422         data = lowsize + 2;
 423         highguard = blockaddr + blocksize - 4;
 424         highsize = highguard + 2;
 425 
 426         KASSERT(*(uint16_t *)lowguard == GUARD_HALFWORD);
 427         KASSERT(*(uint16_t *)highguard == GUARD_HALFWORD);
 428         clientsize = *(uint16_t *)lowsize;
 429         KASSERT(clientsize == *(uint16_t *)highsize);
 430         KASSERT(clientsize + GUARD_OVERHEAD > smallerblocksize);
 431         KASSERT(clientsize + GUARD_OVERHEAD <= blocksize);
 432         enddata = data + clientsize;
 433         for (i=enddata; i<highguard; i++) {
 434                 KASSERT(*(uint8_t *)i == GUARD_FILLBYTE);
 435         }
 436 }

 453 #ifdef CHECKBEEF
 454 /*
 455  * Check that a (free) block contains deadbeef as it should.
 456  *
 457  * The first word of the block is a freelist pointer and should not be
 458  * deadbeef; the rest of the block should be only deadbeef.
 459  */
 460 static
 461 void
 462 checkdeadbeef(void *block, size_t blocksize)
 463 {
 464         uint32_t *ptr = block;
 465         size_t i;
 466 
 467         for (i=1; i < blocksize/sizeof(uint32_t); i++) {
 468                 KASSERT(ptr[i] == 0xdeadbeef);
 469         }
 470 }
 471 #endif /* CHECKBEEF */
 472 
 473 #ifdef SLOW
 474 /*
 475  * Check that a particular heap page (the one managed by the argument
 476  * PR) is valid.
 477  *
 478  * This checks:
 479  *    - that the page is within MIPS_KSEG0 (for mips)
 480  *    - that the freelist starting point in PR is valid
 481  *    - that the number of free blocks is consistent with the freelist
 482  *    - that each freelist next pointer points within the page
 483  *    - that no freelist pointer points to the middle of a block
 484  *    - that free blocks are still deadbeefed (if CHECKBEEF)
 485  *    - that the freelist is not circular
 486  *    - that the guard bands are intact on all allocated blocks (if
 487  *      CHECKGUARDS)
 488  *
 489  * Note that if CHECKGUARDS is set, a circular freelist will cause an
 490  * assertion as a bit in isfree is set twice; if not, a circular
 491  * freelist will cause an infinite loop.
 492  */
 493 static
 494 void
 495 checksubpage(struct pageref *pr)
 496 {
 497         vaddr_t prpage, fla;
 498         struct freelist *fl;
 499         int blktype;
 500         int nfree=0;
 501         size_t blocksize;
 502 #ifdef CHECKGUARDS
 503         const unsigned maxblocks = PAGE_SIZE / SMALLEST_SUBPAGE_SIZE;
 504         const unsigned numfreewords = DIVROUNDUP(maxblocks, 32);
 505         uint32_t isfree[numfreewords], mask;
 506         unsigned numblocks, blocknum, i;
 507         size_t smallerblocksize;
 508 #endif
 509 
 510         KASSERT(spinlock_do_i_hold(&kmalloc_spinlock));
 511 
 512         if (pr->freelist_offset == INVALID_OFFSET) {
 513                 KASSERT(pr->nfree==0);
 514                 return;
 515         }
 516 
 517         prpage = PR_PAGEADDR(pr);
 518         blktype = PR_BLOCKTYPE(pr);
 519         KASSERT(blktype >= 0 && blktype < NSIZES);
 520         blocksize = sizes[blktype];
 521 
 522 #ifdef CHECKGUARDS
 523         smallerblocksize = blktype > 0 ? sizes[blktype - 1] : 0;
 524         for (i=0; i<numfreewords; i++) {
 525                 isfree[i] = 0;
 526         }
 527 #endif
 528 
 529 #ifdef __mips__
 530         KASSERT(prpage >= MIPS_KSEG0);
 531         KASSERT(prpage < MIPS_KSEG1);
 532 #endif
 533 
 534         KASSERT(pr->freelist_offset < PAGE_SIZE);
 535         KASSERT(pr->freelist_offset % blocksize == 0);
 536 
 537         fla = prpage + pr->freelist_offset;
 538         fl = (struct freelist *)fla;
 539 
 540         for (; fl != NULL; fl = fl->next) {
 541                 fla = (vaddr_t)fl;
 542                 KASSERT(fla >= prpage && fla < prpage + PAGE_SIZE);
 543                 KASSERT((fla-prpage) % blocksize == 0);
 544 #ifdef CHECKBEEF
 545                 checkdeadbeef(fl, blocksize);
 546 #endif
 547 #ifdef CHECKGUARDS
 548                 blocknum = (fla-prpage) / blocksize;
 549                 mask = 1U << (blocknum % 32);
 550                 KASSERT((isfree[blocknum / 32] & mask) == 0);
 551                 isfree[blocknum / 32] |= mask;
 552 #endif
 553                 KASSERT(fl->next != fl);
 554                 nfree++;
 555         }
 556         KASSERT(nfree==pr->nfree);
 557 
 558 #ifdef CHECKGUARDS
 559         numblocks = PAGE_SIZE / blocksize;
 560         for (i=0; i<numblocks; i++) {
 561                 mask = 1U << (i % 32);
 562                 if ((isfree[i / 32] & mask) == 0) {
 563                         checkguardband(prpage + i * blocksize,
 564                                        smallerblocksize, blocksize);
 565                 }
 566         }
 567 #endif
 568 }
 569 #else
 570 #define checksubpage(pr) ((void)(pr))
 571 #endif
 572 
 573 #ifdef SLOWER
 574 /*
 575  * Run checksubpage on all heap pages. This also checks that the
 576  * linked lists of pagerefs are more or less intact.
 577  */
 578 static
 579 void
 580 checksubpages(void)
 581 {
 582         struct pageref *pr;
 583         int i;
 584         unsigned sc=0, ac=0;
 585 
 586         KASSERT(spinlock_do_i_hold(&kmalloc_spinlock));
 587 
 588         for (i=0; i<NSIZES; i++) {
 589                 for (pr = sizebases[i]; pr != NULL; pr = pr->next_samesize) {
 590                         checksubpage(pr);
 591                         KASSERT(sc < TOTAL_PAGEREFS);
 592                         sc++;
 593                 }
 594         }
 595 
 596         for (pr = allbase; pr != NULL; pr = pr->next_all) {
 597                 checksubpage(pr);
 598                 KASSERT(ac < TOTAL_PAGEREFS);
 599                 ac++;
 600         }
 601 
 602         KASSERT(sc==ac);
 603 }
 604 #else
 605 #define checksubpages()
 606 #endif

 638 static
 639 void
 640 dump_subpage(struct pageref *pr, unsigned generation)
 641 {
 642         unsigned blocksize = sizes[PR_BLOCKTYPE(pr)];
 643         unsigned numblocks = PAGE_SIZE / blocksize;
 644         unsigned numfreewords = DIVROUNDUP(numblocks, 32);
 645         uint32_t isfree[numfreewords], mask;
 646         vaddr_t prpage;
 647         struct freelist *fl;
 648         vaddr_t blockaddr;
 649         struct malloclabel *ml;
 650         unsigned i;
 651 
 652         for (i=0; i<numfreewords; i++) {
 653                 isfree[i] = 0;
 654         }
 655 
 656         prpage = PR_PAGEADDR(pr);
 657         fl = (struct freelist *)(prpage + pr->freelist_offset);
 658         for (; fl != NULL; fl = fl->next) {
 659                 i = ((vaddr_t)fl - prpage) / blocksize;
 660                 mask = 1U << (i % 32);
 661                 isfree[i / 32] |= mask;
 662         }
 663 
 664         for (i=0; i<numblocks; i++) {
 665                 mask = 1U << (i % 32);
 666                 if (isfree[i / 32] & mask) {
 667                         continue;
 668                 }
 669                 blockaddr = prpage + i * blocksize;
 670                 ml = (struct malloclabel *)blockaddr;
 671                 if (ml->generation != generation) {
 672                         continue;
 673                 }
 674                 kprintf("%5zu bytes at %p, allocated at %p\n",
 675                         blocksize, (void *)blockaddr, (void *)ml->label);
 676         }
 677 }

 725 {
 726 #ifdef LABELS
 727         unsigned i;
 728 
 729         /* print the whole thing with interrupts off */
 730         spinlock_acquire(&kmalloc_spinlock);
 731         for (i=0; i<=mallocgeneration; i++) {
 732                 dump_subpages(i);
 733         }
 734         spinlock_release(&kmalloc_spinlock);
 735 #else
 736         kprintf("Enable LABELS in kmalloc.c to use this functionality.\n");
 737 #endif

 742 /*
 743  * Print the allocated/freed map of a single kernel heap page.
 744  */
 745 static
 746 void
 747 subpage_stats(struct pageref *pr)
 748 {
 749         vaddr_t prpage, fla;
 750         struct freelist *fl;
 751         int blktype;
 752         unsigned i, n, index;
 753         uint32_t freemap[PAGE_SIZE / (SMALLEST_SUBPAGE_SIZE*32)];
 754 
 755         checksubpage(pr);
 756         KASSERT(spinlock_do_i_hold(&kmalloc_spinlock));
 757 
 758         /* clear freemap[] */
 759         for (i=0; i<ARRAYCOUNT(freemap); i++) {
 760                 freemap[i] = 0;
 761         }
 762 
 763         prpage = PR_PAGEADDR(pr);
 764         blktype = PR_BLOCKTYPE(pr);
 765         KASSERT(blktype >= 0 && blktype < NSIZES);
 766 
 767         /* compute how many bits we need in freemap and assert we fit */
 768         n = PAGE_SIZE / sizes[blktype];
 769         KASSERT(n <= 32 * ARRAYCOUNT(freemap));
 770 
 771         if (pr->freelist_offset != INVALID_OFFSET) {
 772                 fla = prpage + pr->freelist_offset;
 773                 fl = (struct freelist *)fla;
 774 
 775                 for (; fl != NULL; fl = fl->next) {
 776                         fla = (vaddr_t)fl;
 777                         index = (fla-prpage) / sizes[blktype];
 778                         KASSERT(index<n);
 779                         freemap[index/32] |= (1<<(index%32));
 780                 }
 781         }
 782 
 783         kprintf("at 0x%08lx: size %-4lu  %u/%u free\n",
 784                 (unsigned long)prpage, (unsigned long) sizes[blktype],
 785                 (unsigned) pr->nfree, n);
 786         kprintf("   ");
 787         for (i=0; i<n; i++) {
 788                 int val = (freemap[i/32] & (1<<(i%32)))!=0;
 789                 kprintf("%c", val ? '.' : '*');
 790                 if (i%64==63 && i<n-1) {
 791                         kprintf("\n   ");
 792                 }
 793         }
 794         kprintf("\n");
 795 }

 797 /*
 798  * Print the whole heap.
 799  */
 800 void
 801 kheap_printstats(void)
 802 {
 803         struct pageref *pr;
 804 
 805         /* print the whole thing with interrupts off */
 806         spinlock_acquire(&kmalloc_spinlock);
 807 
 808         kprintf("Subpage allocator status:\n");
 809 
 810         for (pr = allbase; pr != NULL; pr = pr->next_all) {
 811                 subpage_stats(pr);
 812         }

 819 /*
 820  * Remove a pageref from both lists that it's on.
 821  */
 822 static
 823 void
 824 remove_lists(struct pageref *pr, int blktype)
 825 {
 826         struct pageref **guy;
 827 
 828         KASSERT(blktype>=0 && blktype<NSIZES);
 829 
 830         for (guy = &sizebases[blktype]; *guy; guy = &(*guy)->next_samesize) {
 831                 checksubpage(*guy);
 832                 if (*guy == pr) {
 833                         *guy = pr->next_samesize;
 834                         break;
 835                 }
 836         }
 837 
 838         for (guy = &allbase; *guy; guy = &(*guy)->next_all) {
 839                 checksubpage(*guy);
 840                 if (*guy == pr) {
 841                         *guy = pr->next_all;
 842                         break;
 843                 }
 844         }
 845 }

 847 /*
 848  * Given a requested client size, return the block type, that is, the
 849  * index into the sizes[] array for the block size to use.
 850  */
 851 static
 852 inline
 853 int blocktype(size_t clientsz)
 854 {
 855         unsigned i;
 856         for (i=0; i<NSIZES; i++) {
 857                 if (clientsz <= sizes[i]) {
 858                         return i;
 859                 }
 860         }
 861 
 862         panic("Subpage allocator cannot handle allocation of size %zu\n",
 863               clientsz);
 864 
 865         // keep compiler happy
 866         return 0;
 867 }

 869 /*
 870  * Allocate a block of size SZ, where SZ is not large enough to
 871  * warrant a whole-page allocation.
 872  */
 873 static
 874 void *
 875 subpage_kmalloc(size_t sz
 876 #ifdef LABELS
 877                 , vaddr_t label
 878 #endif
 879         )
 880 {
 881         unsigned blktype;       // index into sizes[] that we're using
 882         struct pageref *pr;     // pageref for page we're allocating from
 883         vaddr_t prpage;         // PR_PAGEADDR(pr)
 884         vaddr_t fla;            // free list entry address
 885         struct freelist *volatile fl;   // free list entry
 886         void *retptr;           // our result
 887 
 888         volatile int i;
 889 
 890 #ifdef GUARDS
 891         size_t clientsz;
 892 #endif
 893 
 894 #ifdef GUARDS
 895         clientsz = sz;
 896         sz += GUARD_OVERHEAD;
 897 #endif
 898 #ifdef LABELS
 899 #ifdef GUARDS
 900         /* Include the label in what GUARDS considers the client data. */
 901         clientsz += LABEL_PTROFFSET;
 902 #endif
 903         sz += LABEL_PTROFFSET;
 904 #endif
 905         blktype = blocktype(sz);
 906 #ifdef GUARDS
 907         sz = sizes[blktype];
 908 #endif
 909 
 910         spinlock_acquire(&kmalloc_spinlock);
 911 
 912         checksubpages();
 913 
 914         for (pr = sizebases[blktype]; pr != NULL; pr = pr->next_samesize) {
 915 
 916                 /* check for corruption */
 917                 KASSERT(PR_BLOCKTYPE(pr) == blktype);
 918                 checksubpage(pr);
 919 
 920                 if (pr->nfree > 0) {
 921 
 922                 doalloc: /* comes here after getting a whole fresh page */
 923 
 924                         KASSERT(pr->freelist_offset < PAGE_SIZE);
 925                         prpage = PR_PAGEADDR(pr);
 926                         fla = prpage + pr->freelist_offset;
 927                         fl = (struct freelist *)fla;
 928 
 929                         retptr = fl;
 930                         fl = fl->next;
 931                         pr->nfree--;
 932 
 933                         if (fl != NULL) {
 934                                 KASSERT(pr->nfree > 0);
 935                                 fla = (vaddr_t)fl;
 936                                 KASSERT(fla - prpage < PAGE_SIZE);
 937                                 pr->freelist_offset = fla - prpage;
 938                         }
 939                         else {
 940                                 KASSERT(pr->nfree == 0);
 941                                 pr->freelist_offset = INVALID_OFFSET;
 942                         }
 943 #ifdef GUARDS
 944                         retptr = establishguardband(retptr, clientsz, sz);
 945 #endif
 946 #ifdef LABELS
 947                         retptr = establishlabel(retptr, label);
 948 #endif
 949 
 950                         checksubpages();
 951 
 952                         spinlock_release(&kmalloc_spinlock);
 953                         return retptr;
 954                 }
 955         }
 956 
 957         /*
 958          * No page of the right size available.
 959          * Make a new one.
 960          *
 961          * We release the spinlock while calling alloc_kpages. This
 962          * avoids deadlock if alloc_kpages needs to come back here.
 963          * Note that this means things can change behind our back...
 964          */
 965 
 966         spinlock_release(&kmalloc_spinlock);
 967         prpage = alloc_kpages(1);
 968         if (prpage==0) {
 969                 /* Out of memory. */
 970                 kprintf("kmalloc: Subpage allocator couldn't get a page\n");
 971                 return NULL;
 972         }
 973         KASSERT(prpage % PAGE_SIZE == 0);
 974 #ifdef CHECKBEEF
 975         /* deadbeef the whole page, as it probably starts zeroed */
 976         fill_deadbeef((void *)prpage, PAGE_SIZE);
 977 #endif
 978         spinlock_acquire(&kmalloc_spinlock);
 979 
 980         pr = allocpageref();
 981         if (pr==NULL) {
 982                 /* Couldn't allocate accounting space for the new page. */
 983                 spinlock_release(&kmalloc_spinlock);
 984                 free_kpages(prpage);
 985                 kprintf("kmalloc: Subpage allocator couldn't get pageref\n");
 986                 return NULL;
 987         }
 988 
 989         pr->pageaddr_and_blocktype = MKPAB(prpage, blktype);
 990         pr->nfree = PAGE_SIZE / sizes[blktype];
 991 
 992         /*
 993          * Note: fl is volatile because the MIPS toolchain we were
 994          * using in spring 2001 attempted to optimize this loop and
 995          * blew it. Making fl volatile inhibits the optimization.
 996          */
 997 
 998         fla = prpage;
 999         fl = (struct freelist *)fla;
1000         fl->next = NULL;
1001         for (i=1; i<pr->nfree; i++) {
1002                 fl = (struct freelist *)(fla + i*sizes[blktype]);
1003                 fl->next = (struct freelist *)(fla + (i-1)*sizes[blktype]);
1004                 KASSERT(fl != fl->next);
1005         }
1006         fla = (vaddr_t) fl;
1007         pr->freelist_offset = fla - prpage;
1008         KASSERT(pr->freelist_offset == (pr->nfree-1)*sizes[blktype]);
1009 
1010         pr->next_samesize = sizebases[blktype];
1011         sizebases[blktype] = pr;
1012 
1013         pr->next_all = allbase;
1014         allbase = pr;
1015 
1016         /* This is kind of cheesy, but avoids duplicating the alloc code. */
1017         goto doalloc;
1018 }

1020 /*
1021  * Free a pointer previously returned from subpage_kmalloc. If the
1022  * pointer is not on any heap page we recognize, return -1.
1023  */
1024 static
1025 int
1026 subpage_kfree(void *ptr)
1027 {
1028         int blktype;            // index into sizes[] that we're using
1029         vaddr_t ptraddr;        // same as ptr
1030         struct pageref *pr;     // pageref for page we're freeing in
1031         vaddr_t prpage;         // PR_PAGEADDR(pr)
1032         vaddr_t fla;            // free list entry address
1033         struct freelist *fl;    // free list entry
1034         vaddr_t offset;         // offset into page
1035 #ifdef GUARDS
1036         size_t blocksize, smallerblocksize;
1037 #endif
1038 
1039         ptraddr = (vaddr_t)ptr;
1040 #ifdef GUARDS
1041         if (ptraddr % PAGE_SIZE == 0) {
1042                 /*
1043                  * With guard bands, all client-facing subpage
1044                  * pointers are offset by GUARD_PTROFFSET (which is 4)
1045                  * from the underlying blocks and are therefore not
1046                  * page-aligned. So a page-aligned pointer is not one
1047                  * of ours. Catch this up front, as otherwise
1048                  * subtracting GUARD_PTROFFSET could give a pointer on
1049                  * a page we *do* own, and then we'll panic because
1050                  * it's not a valid one.
1051                  */
1052                 return -1;
1053         }
1054         ptraddr -= GUARD_PTROFFSET;
1055 #endif
1056 #ifdef LABELS
1057         if (ptraddr % PAGE_SIZE == 0) {
1058                 /* ditto */
1059                 return -1;
1060         }
1061         ptraddr -= LABEL_PTROFFSET;
1062 #endif
1063 
1064         spinlock_acquire(&kmalloc_spinlock);
1065 
1066         checksubpages();
1067 
1068         /* Silence warnings with gcc 4.8 -Og (but not -O2) */
1069         prpage = 0;
1070         blktype = 0;
1071 
1072         for (pr = allbase; pr; pr = pr->next_all) {
1073                 prpage = PR_PAGEADDR(pr);
1074                 blktype = PR_BLOCKTYPE(pr);
1075                 KASSERT(blktype >= 0 && blktype < NSIZES);
1076 
1077                 /* check for corruption */
1078                 KASSERT(blktype>=0 && blktype<NSIZES);
1079                 checksubpage(pr);
1080 
1081                 if (ptraddr >= prpage && ptraddr < prpage + PAGE_SIZE) {
1082                         break;
1083                 }
1084         }
1085 
1086         if (pr==NULL) {
1087                 /* Not on any of our pages - not a subpage allocation */
1088                 spinlock_release(&kmalloc_spinlock);
1089                 return -1;
1090         }
1091 
1092         offset = ptraddr - prpage;
1093 
1094         /* Check for proper positioning and alignment */
1095         if (offset >= PAGE_SIZE || offset % sizes[blktype] != 0) {
1096                 panic("kfree: subpage free of invalid addr %p\n", ptr);
1097         }
1098 
1099 #ifdef GUARDS
1100         blocksize = sizes[blktype];
1101         smallerblocksize = blktype > 0 ? sizes[blktype - 1] : 0;
1102         checkguardband(ptraddr, smallerblocksize, blocksize);
1103 #endif
1104 
1105         /*
1106          * Clear the block to 0xdeadbeef to make it easier to detect
1107          * uses of dangling pointers.
1108          */
1109         fill_deadbeef((void *)ptraddr, sizes[blktype]);
1110 
1111         /*
1112          * We probably ought to check for free twice by seeing if the block
1113          * is already on the free list. But that's expensive, so we don't.
1114          */
1115 
1116         fla = prpage + offset;
1117         fl = (struct freelist *)fla;
1118         if (pr->freelist_offset == INVALID_OFFSET) {
1119                 fl->next = NULL;
1120         } else {
1121                 fl->next = (struct freelist *)(prpage + pr->freelist_offset);
1122 
1123                 /* this block should not already be on the free list! */
1124 #ifdef SLOW
1125                 {
1126                         struct freelist *fl2;
1127 
1128                         for (fl2 = fl->next; fl2 != NULL; fl2 = fl2->next) {
1129                                 KASSERT(fl2 != fl);
1130                         }
1131                 }
1132 #else
1133                 /* check just the head */
1134                 KASSERT(fl != fl->next);
1135 #endif
1136         }
1137         pr->freelist_offset = offset;
1138         pr->nfree++;
1139 
1140         KASSERT(pr->nfree <= PAGE_SIZE / sizes[blktype]);
1141         if (pr->nfree == PAGE_SIZE / sizes[blktype]) {
1142                 /* Whole page is free. */
1143                 remove_lists(pr, blktype);
1144                 freepageref(pr);
1145                 /* Call free_kpages without kmalloc_spinlock. */
1146                 spinlock_release(&kmalloc_spinlock);
1147                 free_kpages(prpage);
1148         }
1149         else {
1150                 spinlock_release(&kmalloc_spinlock);
1151         }
1152 
1153 #ifdef SLOWER /* Don't get the lock unless checksubpages does something. */
1154         spinlock_acquire(&kmalloc_spinlock);
1155         checksubpages();
1156         spinlock_release(&kmalloc_spinlock);
1157 #endif
1158 
1159         return 0;
1160 }

1165 /*
1166  * Allocate a block of size SZ. Redirect either to subpage_kmalloc or
1167  * alloc_kpages depending on how big SZ is.
1168  */
1169 void *
1170 kmalloc(size_t sz)
1171 {
1172         size_t checksz;
1173 #ifdef LABELS
1174         vaddr_t label;
1175 #endif
1176 
1177 #ifdef LABELS
1178 #ifdef __GNUC__
1179         label = (vaddr_t)__builtin_return_address(0);
1180 #else
1181 #error "Don't know how to get return address with this compiler"
1182 #endif /* __GNUC__ */
1183 #endif /* LABELS */
1184 
1185         checksz = sz + GUARD_OVERHEAD + LABEL_OVERHEAD;
1186         if (checksz >= LARGEST_SUBPAGE_SIZE) {
1187                 unsigned long npages;
1188                 vaddr_t address;
1189 
1190                 /* Round up to a whole number of pages. */
1191                 npages = (sz + PAGE_SIZE - 1)/PAGE_SIZE;
1192                 address = alloc_kpages(npages);
1193                 if (address==0) {
1194                         return NULL;
1195                 }
1196                 KASSERT(address % PAGE_SIZE == 0);
1197 
1198                 return (void *)address;
1199         }
1200 
1201 #ifdef LABELS
1202         return subpage_kmalloc(sz, label);
1203 #else
1204         return subpage_kmalloc(sz);
1205 #endif
1206 }

1208 /*
1209  * Free a block previously returned from kmalloc.
1210  */
1211 void
1212 kfree(void *ptr)
1213 {
1214         /*
1215          * Try subpage first; if that fails, assume it's a big allocation.
1216          */
1217         if (ptr == NULL) {
1218                 return;
1219         } else if (subpage_kfree(ptr)) {
1220                 KASSERT((vaddr_t)ptr%PAGE_SIZE==0);
1221                 free_kpages((vaddr_t)ptr);
1222         }
1223 }
