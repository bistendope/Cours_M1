  41 /*
  42  * Functions provided for debug hacking:
  43  *   ltrace_on:    turns on the trace161 tracing flag CODE.
  44  *   ltrace_off:   turns off the trace161 tracing flag CODE.
  45  *   ltrace_debug: causes sys161/trace161 to print a message with CODE.
  46  *   ltrace_dump:  causes trace161 to do a complete state dump, tagged CODE.
  47  *   ltrace_stop:  causes sys161/trace161 to drop to the debugger.
  48  *   ltrace_setprof: turn on and off trace161 profile collection.
  49  *   ltrace_eraseprof: discard trace161 profile collected so far.
  50  *
  51  * The flags for ltrace_on/off are the characters used to control
  52  * tracing on the trace161 command line. See the System/161 manual for
  53  * more information.
  54  *
  55  * ltrace_debug is for printing simple indications that a certain
  56  * piece of code has been reached, like one might use kprintf, except
  57  * that it is less invasive than kprintf. Think of it as setting the
  58  * value of a readout on the system's front panel. (In real life,
  59  * since computers don't have front panels with blinking lights any
  60  * more, people often use the speaker or the top left corner of the
  61  * screen for this purpose.)
  62  *
  63  * ltrace_dump dumps the entire system state and is primarily intended
  64  * for regression testing of System/161. It might or might not prove
  65  * useful for debugging as well.
  66  *
  67  * Calling ltrace_stop behaves similarly to hardwiring a breakpoint
  68  * instruction in your code, except that debuggers have trouble
  69  * stepping past hardwired breakpoints. Currently the value of the
  70  * code is ignored.
  71  *
  72  * ltrace_setprof can be used to dynamically turn profiling on and
  73  * off, if trace161 is collecting a profile. (Otherwise it does
  74  * nothing.) This can be used to e.g. profile only code that executes
  75  * while holding a given lock.
  76  *
  77  * ltrace_eraseprof can be used to clear the accumulated profile
  78  * data, if trace161 is collecting a profile. (Otherwise it does
  79  * nothing.) This can be used to e.g. exclude bootup actions from your
  80  * profile.
  81  */
  82 void ltrace_on(uint32_t code);
  83 void ltrace_off(uint32_t code);
  84 void ltrace_debug(uint32_t code);
  85 void ltrace_dump(uint32_t code);
  86 void ltrace_stop(uint32_t code);
  87 void ltrace_setprof(uint32_t onoff);
  88 void ltrace_eraseprof(void);
  89 
  90 #endif /* _LAMEBUS_LTRACE_H_ */
