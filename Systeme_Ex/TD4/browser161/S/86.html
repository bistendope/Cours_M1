  30 /*
  31  * Machine (and hardware) independent console driver.
  32  *
  33  * We expose a simple interface to the rest of the kernel: "putch" to
  34  * print a character, "getch" to read one.
  35  *
  36  * As long as the device we're connected to does, we allow printing in
  37  * an interrupt handler or with interrupts off (by polling),
  38  * transparently to the caller. Note that getch by polling is not
  39  * supported, although such support could be added without undue
  40  * difficulty.
  41  *
  42  * Note that nothing happens until we have a device to write to. A
  43  * buffer of size DELAYBUFSIZE is used to hold output that is
  44  * generated before this point. This means that (1) using kprintf for
  45  * debugging problems that occur early in initialization is awkward,
  46  * and (2) if the system crashes before we find a console, no output
  47  * at all may appear.
  48  */
  49

  77 /*
  78  * This is for accumulating characters printed before the
  79  * console is set up. Upon console setup they are dumped
  80  * to the actual console; thenceforth this space is unused.
  81  */
  82 #define DELAYBUFSIZE  1024
  83 static char delayed_outbuf[DELAYBUFSIZE];
  84 static size_t delayed_outbuf_pos=0;
  85 
  86 static
  87 void
  88 putch_delayed(int ch)
  89 {
  90         /*
  91          * No synchronization needed: called only during system startup
  92          * by main thread.
  93          */
  94 
  95         KASSERT(delayed_outbuf_pos < sizeof(delayed_outbuf));
  96         delayed_outbuf[delayed_outbuf_pos++] = ch;
  97 }
  98 
  99 static
 100 void
 101 flush_delay_buf(void)
 102 {
 103         size_t i;
 104         for (i=0; i<delayed_outbuf_pos; i++) {
 105                 putch(delayed_outbuf[i]);
 106         }
 107         delayed_outbuf_pos = 0;
 108 }

 136 /*
 137  * Read a character, using interrupts to wait for I/O completion.
 138  */
 139 static
 140 int
 141 getch_intr(struct con_softc *cs)
 142 {
 143         unsigned char ret;
 144 
 145         P(cs->cs_rsem);
 146         ret = cs->cs_gotchars[cs->cs_gotchars_tail];
 147         cs->cs_gotchars_tail =
 148                 (cs->cs_gotchars_tail + 1) % CONSOLE_INPUT_BUFFER_SIZE;
 149         return ret;
 150 }
 151 
 152 /*
 153  * Called from underlying device when a read-ready interrupt occurs.
 154  *
 155  * Note: if gotchars_head == gotchars_tail, the buffer is empty. Thus
 156  * if gotchars_head+1 == gotchars_tail, the buffer is full. A slightly
 157  * tidier way to implement this check (that avoids wasting a slot,
 158  * too) would be with a second semaphore used with a nonblocking P,
 159  * but we don't have that in OS/161.
 160  */
 161 void
 162 con_input(void *vcs, int ch)
 163 {
 164         struct con_softc *cs = vcs;
 165         unsigned nexthead;
 166 
 167         nexthead = (cs->cs_gotchars_head + 1) % CONSOLE_INPUT_BUFFER_SIZE;
 168         if (nexthead == cs->cs_gotchars_tail) {
 169                 /* overflow; drop character */
 170                 return;
 171         }
 172 
 173         cs->cs_gotchars[cs->cs_gotchars_head] = ch;
 174         cs->cs_gotchars_head = nexthead;
 175 
 176         V(cs->cs_rsem);
 177 }

 179 /*
 180  * Called from underlying device when a write-done interrupt occurs.
 181  */
 182 void
 183 con_start(void *vcs)
 184 {
 185         struct con_softc *cs = vcs;

 187         V(cs->cs_wsem);
 188 }

 192 /*
 193  * Exported interface.
 194  *
 195  * Warning: putch must work even in an interrupt handler or with
 196  * interrupts disabled, and before the console is probed. getch need
 197  * not, and does not.
 198  */
 199 
 200 void
 201 putch(int ch)
 202 {
 203         struct con_softc *cs = the_console;
 204 
 205         if (cs==NULL) {
 206                 putch_delayed(ch);
 207         }
 208         else if (curthread->t_in_interrupt ||
 209                  curthread->t_curspl > 0 ||
 210                  curcpu->c_spinlocks > 0) {
 211                 putch_polled(cs, ch);
 212         }
 213         else {
 214                 putch_intr(cs, ch);
 215         }
 216 }
 217 
 218 int
 219 getch(void)
 220 {
 221         struct con_softc *cs = the_console;
 222         KASSERT(cs != NULL);
 223         KASSERT(!curthread->t_in_interrupt && curthread->t_iplhigh_count == 0);
 224 
 225         return getch_intr(cs);
 226 }

 243 static
 244 int
 245 con_io(struct device *dev, struct uio *uio)
 246 {
 247         int result;
 248         char ch;
 249         struct lock *lk;
 250 
 251         (void)dev;  // unused
 252 
 253         if (uio->uio_rw==UIO_READ) {
 254                 lk = con_userlock_read;
 255         }
 256         else {
 257                 lk = con_userlock_write;
 258         }
 259 
 260         KASSERT(lk != NULL);
 261         lock_acquire(lk);
 262 
 263         while (uio->uio_resid > 0) {
 264                 if (uio->uio_rw==UIO_READ) {
 265                         ch = getch();
 266                         if (ch=='\r') {
 267                                 ch = '\n';
 268                         }
 269                         result = uiomove(&ch, 1, uio);
 270                         if (result) {
 271                                 lock_release(lk);
 272                                 return result;
 273                         }
 274                         if (ch=='\n') {
 275                                 break;
 276                         }
 277                 }
 278                 else {
 279                         result = uiomove(&ch, 1, uio);
 280                         if (result) {
 281                                 lock_release(lk);
 282                                 return result;
 283                         }
 284                         if (ch=='\n') {
 285                                 putch('\r');
 286                         }
 287                         putch(ch);
 288                 }
 289         }
 290         lock_release(lk);
 291         return 0;
 292 }

 311 static
 312 int
 313 attach_console_to_vfs(struct con_softc *cs)
 314 {
 315         struct device *dev;
 316         int result;
 317 
 318         dev = kmalloc(sizeof(*dev));
 319         if (dev==NULL) {
 320                 return ENOMEM;
 321         }
 322 
 323         dev->d_ops = &console_devops;
 324         dev->d_blocks = 0;
 325         dev->d_blocksize = 1;
 326         dev->d_data = cs;
 327 
 328         result = vfs_adddev("con", dev, 0);
 329         if (result) {
 330                 kfree(dev);
 331                 return result;
 332         }
 333 
 334         return 0;
 335 }

 339 /*
 340  * Config routine called by autoconf.c after we are attached to something.
 341  */
 342 
 343 int
 344 config_con(struct con_softc *cs, int unit)
 345 {
 346         struct semaphore *rsem, *wsem;
 347         struct lock *rlk, *wlk;
 348 
 349         /*
 350          * Only allow one system console.
 351          * Further devices that could be the system console are ignored.
 352          *
 353          * Do not hardwire the console to be "con1" instead of "con0",
 354          * or these asserts will go off.
 355          */
 356         if (unit>0) {
 357                 KASSERT(the_console!=NULL);
 358                 return ENODEV;
 359         }
 360         KASSERT(the_console==NULL);
 361 
 362         rsem = sem_create("console read", 0);
 363         if (rsem == NULL) {
 364                 return ENOMEM;
 365         }
 366         wsem = sem_create("console write", 1);
 367         if (wsem == NULL) {
 368                 sem_destroy(rsem);
 369                 return ENOMEM;
 370         }
 371         rlk = lock_create("console-lock-read");
 372         if (rlk == NULL) {
 373                 sem_destroy(rsem);
 374                 sem_destroy(wsem);
 375                 return ENOMEM;
 376         }
 377         wlk = lock_create("console-lock-write");
 378         if (wlk == NULL) {
 379                 lock_destroy(rlk);
 380                 sem_destroy(rsem);
 381                 sem_destroy(wsem);
 382                 return ENOMEM;
 383         }
 384 
 385         cs->cs_rsem = rsem;
 386         cs->cs_wsem = wsem;
 387         cs->cs_gotchars_head = 0;
 388         cs->cs_gotchars_tail = 0;
 389 
 390         the_console = cs;
 391         con_userlock_read = rlk;
 392         con_userlock_write = wlk;
 393 
 394         flush_delay_buf();
 395 
 396         return attach_console_to_vfs(cs);
 397 }
