  46 static
  47 void
  48 pipe_badptr(void *ptr, const char *desc)
  49 {
  50         int rv;
  51 
  52         report_begin("%s", desc);
  53         rv = pipe(ptr);
  54         report_check(rv, errno, EFAULT);
  55 }

  57 static
  58 void
  59 pipe_unaligned(void)
  60 {
  61         int fds[3], rv;
  62         char *ptr;
  63 
  64         report_begin("pipe with unaligned pointer");
  65 
  66         ptr = (char *)&fds[0];
  67         ptr++;
  68 
  69         rv = pipe((int *)ptr);
  70         report_survival(rv, errno);
  71         if (rv == 0) {
  72                 memmove(fds, ptr, 2*sizeof(int));
  73                 close(fds[0]);
  74                 close(fds[1]);
  75         }
  76 }
