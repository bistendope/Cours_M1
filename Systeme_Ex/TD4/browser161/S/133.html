  52 int
  53 sfs_eachopen(struct vnode *v, int openflags)
  54 {
  55         /*
  56          * At this level we do not need to handle O_CREAT, O_EXCL,
  57          * O_TRUNC, or O_APPEND.
  58          *
  59          * Any of O_RDONLY, O_WRONLY, and O_RDWR are valid, so we don't need
  60          * to check that either.
  61          */
  62 
  63         (void)v;
  64         (void)openflags;
  65 
  66         return 0;
  67 }

  69 /*
  70  * This is called on *each* open() of a directory.
  71  * Directories may only be open for read.
  72  */
  73 static
  74 int
  75 sfs_eachopendir(struct vnode *v, int openflags)
  76 {
  77         switch (openflags & O_ACCMODE) {
  78             case O_RDONLY:
  79                 break;
  80             case O_WRONLY:
  81             case O_RDWR:
  82             default:
  83                 return EISDIR;
  84         }
  85         if (openflags & O_APPEND) {
  86                 return EISDIR;
  87         }
  88 
  89         (void)v;
  90         return 0;
  91 }

 149 /*
 150  * Called for stat/fstat/lstat.
 151  */
 152 static
 153 int
 154 sfs_stat(struct vnode *v, struct stat *statbuf)
 155 {
 156         struct sfs_vnode *sv = v->vn_data;
 157         int result;
 158 
 159         /* Fill in the stat structure */
 160         bzero(statbuf, sizeof(struct stat));
 161 
 162         result = VOP_GETTYPE(v, &statbuf->st_mode);
 163         if (result) {
 164                 return result;
 165         }
 166 
 167         statbuf->st_size = sv->sv_i.sfi_size;
 168         statbuf->st_nlink = sv->sv_i.sfi_linkcount;
 169 
 170         /* We don't support this yet */
 171         statbuf->st_blocks = 0;
 172 
 173         /* Fill in other fields as desired/possible... */
 174 
 175         return 0;
 176 }

 178 /*
 179  * Return the type of the file (types as per kern/stat.h)
 180  */
 181 static
 182 int
 183 sfs_gettype(struct vnode *v, uint32_t *ret)
 184 {
 185         struct sfs_vnode *sv = v->vn_data;
 186         struct sfs_fs *sfs = v->vn_fs->fs_data;
 187 
 188         vfs_biglock_acquire();
 189 
 190         switch (sv->sv_i.sfi_type) {
 191         case SFS_TYPE_FILE:
 192                 *ret = S_IFREG;
 193                 vfs_biglock_release();
 194                 return 0;
 195         case SFS_TYPE_DIR:
 196                 *ret = S_IFDIR;
 197                 vfs_biglock_release();
 198                 return 0;
 199         }
 200         panic("sfs: %s: gettype: Invalid inode type (inode %u, type %u)\n",
 201               sfs->sfs_sb.sb_volname, sv->sv_ino, sv->sv_i.sfi_type);
 202         return EINVAL;
 203 }

 277 /*
 278  * Create a file. If EXCL is set, insist that the filename not already
 279  * exist; otherwise, if it already exists, just open it.
 280  */
 281 static
 282 int
 283 sfs_creat(struct vnode *v, const char *name, bool excl, mode_t mode,
 284           struct vnode **ret)
 285 {
 286         struct sfs_fs *sfs = v->vn_fs->fs_data;
 287         struct sfs_vnode *sv = v->vn_data;
 288         struct sfs_vnode *newguy;
 289         uint32_t ino;
 290         int result;
 291 
 292         vfs_biglock_acquire();
 293 
 294         /* Look up the name */
 295         result = sfs_dir_findname(sv, name, &ino, NULL, NULL);
 296         if (result!=0 && result!=ENOENT) {
 297                 vfs_biglock_release();
 298                 return result;
 299         }
 300 
 301         /* If it exists and we didn't want it to, fail */
 302         if (result==0 && excl) {
 303                 vfs_biglock_release();
 304                 return EEXIST;
 305         }
 306 
 307         if (result==0) {
 308                 /* We got something; load its vnode and return */
 309                 result = sfs_loadvnode(sfs, ino, SFS_TYPE_INVAL, &newguy);
 310                 if (result) {
 311                         vfs_biglock_release();
 312                         return result;
 313                 }
 314                 *ret = &newguy->sv_absvn;
 315                 vfs_biglock_release();
 316                 return 0;
 317         }
 318 
 319         /* Didn't exist - create it */
 320         result = sfs_makeobj(sfs, SFS_TYPE_FILE, &newguy);
 321         if (result) {
 322                 vfs_biglock_release();
 323                 return result;
 324         }
 325 
 326         /* We don't currently support file permissions; ignore MODE */
 327         (void)mode;
 328 
 329         /* Link it into the directory */
 330         result = sfs_dir_link(sv, name, newguy->sv_ino, NULL);
 331         if (result) {
 332                 VOP_DECREF(&newguy->sv_absvn);
 333                 vfs_biglock_release();
 334                 return result;
 335         }
 336 
 337         /* Update the linkcount of the new file */
 338         newguy->sv_i.sfi_linkcount++;
 339 
 340         /* and consequently mark it dirty. */
 341         newguy->sv_dirty = true;
 342 
 343         *ret = &newguy->sv_absvn;
 344 
 345         vfs_biglock_release();
 346         return 0;
 347 }

 349 /*
 350  * Make a hard link to a file.
 351  * The VFS layer should prevent this being called unless both
 352  * vnodes are ours.
 353  */
 354 static
 355 int
 356 sfs_link(struct vnode *dir, const char *name, struct vnode *file)
 357 {
 358         struct sfs_vnode *sv = dir->vn_data;
 359         struct sfs_vnode *f = file->vn_data;
 360         int result;
 361 
 362         KASSERT(file->vn_fs == dir->vn_fs);
 363 
 364         vfs_biglock_acquire();
 365 
 366         /* Hard links to directories aren't allowed. */
 367         if (f->sv_i.sfi_type == SFS_TYPE_DIR) {
 368                 vfs_biglock_release();
 369                 return EINVAL;
 370         }
 371 
 372         /* Create the link */
 373         result = sfs_dir_link(sv, name, f->sv_ino, NULL);
 374         if (result) {
 375                 vfs_biglock_release();
 376                 return result;
 377         }
 378 
 379         /* and update the link count, marking the inode dirty */
 380         f->sv_i.sfi_linkcount++;
 381         f->sv_dirty = true;
 382 
 383         vfs_biglock_release();
 384         return 0;
 385 }

 387 /*
 388  * Delete a file.
 389  */
 390 static
 391 int
 392 sfs_remove(struct vnode *dir, const char *name)
 393 {
 394         struct sfs_vnode *sv = dir->vn_data;
 395         struct sfs_vnode *victim;
 396         int slot;
 397         int result;
 398 
 399         vfs_biglock_acquire();
 400 
 401         /* Look for the file and fetch a vnode for it. */
 402         result = sfs_lookonce(sv, name, &victim, &slot);
 403         if (result) {
 404                 vfs_biglock_release();
 405                 return result;
 406         }
 407 
 408         /* Erase its directory entry. */
 409         result = sfs_dir_unlink(sv, slot);
 410         if (result==0) {
 411                 /* If we succeeded, decrement the link count. */
 412                 KASSERT(victim->sv_i.sfi_linkcount > 0);
 413                 victim->sv_i.sfi_linkcount--;
 414                 victim->sv_dirty = true;
 415         }
 416 
 417         /* Discard the reference that sfs_lookonce got us */
 418         VOP_DECREF(&victim->sv_absvn);
 419 
 420         vfs_biglock_release();
 421         return result;
 422 }

 424 /*
 425  * Rename a file.
 426  *
 427  * Since we don't support subdirectories, assumes that the two
 428  * directories passed are the same.
 429  */
 430 static
 431 int
 432 sfs_rename(struct vnode *d1, const char *n1,
 433            struct vnode *d2, const char *n2)
 434 {
 435         struct sfs_vnode *sv = d1->vn_data;
 436         struct sfs_fs *sfs = sv->sv_absvn.vn_fs->fs_data;
 437         struct sfs_vnode *g1;
 438         int slot1, slot2;
 439         int result, result2;
 440 
 441         vfs_biglock_acquire();
 442 
 443         KASSERT(d1==d2);
 444         KASSERT(sv->sv_ino == SFS_ROOTDIR_INO);
 445 
 446         /* Look up the old name of the file and get its inode and slot number*/
 447         result = sfs_lookonce(sv, n1, &g1, &slot1);
 448         if (result) {
 449                 vfs_biglock_release();
 450                 return result;
 451         }
 452 
 453         /* We don't support subdirectories */
 454         KASSERT(g1->sv_i.sfi_type == SFS_TYPE_FILE);
 455 
 456         /*
 457          * Link it under the new name.
 458          *
 459          * We could theoretically just overwrite the original
 460          * directory entry, except that we need to check to make sure
 461          * the new name doesn't already exist; might as well use the
 462          * existing link routine.
 463          */
 464         result = sfs_dir_link(sv, n2, g1->sv_ino, &slot2);
 465         if (result) {
 466                 goto puke;
 467         }
 468 
 469         /* Increment the link count, and mark inode dirty */
 470         g1->sv_i.sfi_linkcount++;
 471         g1->sv_dirty = true;
 472 
 473         /* Unlink the old slot */
 474         result = sfs_dir_unlink(sv, slot1);
 475         if (result) {
 476                 goto puke_harder;
 477         }
 478 
 479         /*
 480          * Decrement the link count again, and mark the inode dirty again,
 481          * in case it's been synced behind our back.
 482          */
 483         KASSERT(g1->sv_i.sfi_linkcount>0);
 484         g1->sv_i.sfi_linkcount--;
 485         g1->sv_dirty = true;
 486 
 487         /* Let go of the reference to g1 */
 488         VOP_DECREF(&g1->sv_absvn);
 489 
 490         vfs_biglock_release();
 491         return 0;
 492 
 493  puke_harder:
 494         /*
 495          * Error recovery: try to undo what we already did
 496          */
 497         result2 = sfs_dir_unlink(sv, slot2);
 498         if (result2) {
 499                 kprintf("sfs: %s: rename: %s\n",
 500                         sfs->sfs_sb.sb_volname, strerror(result));
 501                 kprintf("sfs: %s: rename: while cleaning up: %s\n",
 502                         sfs->sfs_sb.sb_volname, strerror(result2));
 503                 panic("sfs: %s: rename: Cannot recover\n",
 504                       sfs->sfs_sb.sb_volname);
 505         }
 506         g1->sv_i.sfi_linkcount--;
 507  puke:
 508         /* Let go of the reference to g1 */
 509         VOP_DECREF(&g1->sv_absvn);
 510         vfs_biglock_release();
 511         return result;
 512 }

 514 /*
 515  * lookparent returns the last path component as a string and the
 516  * directory it's in as a vnode.
 517  *
 518  * Since we don't support subdirectories, this is very easy -
 519  * return the root dir and copy the path.
 520  */
 521 static
 522 int
 523 sfs_lookparent(struct vnode *v, char *path, struct vnode **ret,
 524                   char *buf, size_t buflen)
 525 {
 526         struct sfs_vnode *sv = v->vn_data;
 527 
 528         vfs_biglock_acquire();
 529 
 530         if (sv->sv_i.sfi_type != SFS_TYPE_DIR) {
 531                 vfs_biglock_release();
 532                 return ENOTDIR;
 533         }
 534 
 535         if (strlen(path)+1 > buflen) {
 536                 vfs_biglock_release();
 537                 return ENAMETOOLONG;
 538         }
 539         strcpy(buf, path);
 540 
 541         VOP_INCREF(&sv->sv_absvn);
 542         *ret = &sv->sv_absvn;
 543 
 544         vfs_biglock_release();
 545         return 0;
 546 }

 548 /*
 549  * Lookup gets a vnode for a pathname.
 550  *
 551  * Since we don't support subdirectories, it's easy - just look up the
 552  * name.
 553  */
 554 static
 555 int
 556 sfs_lookup(struct vnode *v, char *path, struct vnode **ret)
 557 {
 558         struct sfs_vnode *sv = v->vn_data;
 559         struct sfs_vnode *final;
 560         int result;
 561 
 562         vfs_biglock_acquire();
 563 
 564         if (sv->sv_i.sfi_type != SFS_TYPE_DIR) {
 565                 vfs_biglock_release();
 566                 return ENOTDIR;
 567         }
 568 
 569         result = sfs_lookonce(sv, path, &final, NULL);
 570         if (result) {
 571                 vfs_biglock_release();
 572                 return result;
 573         }
 574 
 575         *ret = &final->sv_absvn;
 576 
 577         vfs_biglock_release();
 578         return 0;
 579 }
