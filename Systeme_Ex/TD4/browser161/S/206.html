  61 /*
  62  * Function for a thread that runs an arbitrary userlevel program by
  63  * name.
  64  *
  65  * Note: this cannot pass arguments to the program. You may wish to
  66  * change it so it can, because that will make testing much easier
  67  * in the future.
  68  *
  69  * It copies the program name because runprogram destroys the copy
  70  * it gets by passing it to vfs_open().
  71  */
  72 static
  73 void
  74 cmd_progthread(void *ptr, unsigned long nargs)
  75 {
  76         char **args = ptr;
  77         char progname[128];
  78         int result;
  79 
  80         KASSERT(nargs >= 1);
  81 
  82         if (nargs > 2) {
  83                 kprintf("Warning: argument passing from menu not supported\n");
  84         }
  85 
  86         /* Hope we fit. */
  87         KASSERT(strlen(args[0]) < sizeof(progname));
  88 
  89         strcpy(progname, args[0]);
  90 
  91         result = runprogram(progname);
  92         if (result) {
  93                 kprintf("Running program %s failed: %s\n", args[0],
  94                         strerror(result));
  95                 return;
  96         }
  97 
  98         /* NOTREACHED: runprogram only returns on error. */
  99 }

 101 /*
 102  * Common code for cmd_prog and cmd_shell.
 103  *
 104  * Note that this does not wait for the subprogram to finish, but
 105  * returns immediately to the menu. This is usually not what you want,
 106  * so you should have it call your system-calls-assignment waitpid
 107  * code after forking.
 108  *
 109  * Also note that because the subprogram's thread uses the "args"
 110  * array and strings, until you do this a race condition exists
 111  * between that code and the menu input code.
 112  */
 113 static
 114 int
 115 common_prog(int nargs, char **args)
 116 {
 117         struct proc *proc;
 118         int result;
 119 
 120         /* Create a process for the new program to run in. */
 121         proc = proc_create_runprogram(args[0] /* name */);
 122         if (proc == NULL) {
 123                 return ENOMEM;
 124         }
 125 
 126         result = thread_fork(args[0] /* thread name */,
 127                         proc /* new process */,
 128                         cmd_progthread /* thread function */,
 129                         args /* thread arg */, nargs /* thread arg */);
 130         if (result) {
 131                 kprintf("thread_fork failed: %s\n", strerror(result));
 132                 proc_destroy(proc);
 133                 return result;
 134         }
 135 
 136         /*
 137          * The new process will be destroyed when the program exits...
 138          * once you write the code for handling that.
 139          */
 140 
 141         return 0;
 142 }

 204         int result;

 206         struct uio ku;

 209         (void)args;
 210 
 211         uio_kinit(&iov, &ku, buf, sizeof(buf)-1, 0, UIO_READ);

 213         if (result) {
 214                 kprintf("vfs_getcwd failed (%s)\n", strerror(result));

 274 struct deadlock {
 275         struct lock *lock1;
 276         struct lock *lock2;

 279 static
 280 void
 281 cmd_deadlockthread(void *ptr, unsigned long num)
 282 {
 283         struct deadlock *dl = ptr;
 284 
 285         (void)num;
 286 
 287         /* If it doesn't wedge right away, keep trying... */
 288         while (1) {
 289                 lock_acquire(dl->lock2);
 290                 lock_acquire(dl->lock1);
 291                 kprintf("+");
 292                 lock_release(dl->lock1);
 293                 lock_release(dl->lock2);
 294         }
 295 }

 297 /*
 298  * Command for doing an intentional deadlock.
 299  */
 300 static
 301 int
 302 cmd_deadlock(int nargs, char **args)
 303 {
 304         struct deadlock dl;
 305         int result;
 306 
 307         (void)nargs;
 308         (void)args;
 309 
 310         dl.lock1 = lock_create("deadlock1");
 311         if (dl.lock1 == NULL) {
 312                 kprintf("lock_create failed\n");
 313                 return ENOMEM;
 314         }
 315         dl.lock2 = lock_create("deadlock2");
 316         if (dl.lock2 == NULL) {
 317                 lock_destroy(dl.lock1);
 318                 kprintf("lock_create failed\n");
 319                 return ENOMEM;
 320         }
 321 
 322         result = thread_fork(args[0] /* thread name */,
 323                         NULL /* kernel thread */,
 324                         cmd_deadlockthread /* thread function */,
 325                         &dl /* thread arg */, 0 /* thread arg */);
 326         if (result) {
 327                 kprintf("thread_fork failed: %s\n", strerror(result));
 328                 lock_release(dl.lock1);
 329                 lock_destroy(dl.lock2);
 330                 lock_destroy(dl.lock1);
 331                 return result;
 332         }
 333 
 334         /* If it doesn't wedge right away, keep trying... */
 335         while (1) {
 336                 lock_acquire(dl.lock1);
 337                 lock_acquire(dl.lock2);
 338                 kprintf(".");
 339                 lock_release(dl.lock2);
 340                 lock_release(dl.lock1);
 341         }
 342         /* NOTREACHED */
 343         return 0;
 344 }

 376 static
 377 int
 378 cmd_mount(int nargs, char **args)
 379 {
 380         char *fstype;
 381         char *device;
 382         unsigned i;
 383 
 384         if (nargs != 3) {
 385                 kprintf("Usage: mount fstype device:\n");
 386                 return EINVAL;
 387         }
 388 
 389         fstype = args[1];
 390         device = args[2];
 391 
 392         /* Allow (but do not require) colon after device name */
 393         if (device[strlen(device)-1]==':') {
 394                 device[strlen(device)-1] = 0;
 395         }
 396 
 397         for (i=0; i<ARRAYCOUNT(mounttable); i++) {
 398                 if (!strcmp(mounttable[i].name, fstype)) {
 399                         return mounttable[i].func(device);
 400                 }
 401         }
 402         kprintf("Unknown filesystem type %s\n", fstype);
 403         return EINVAL;
 404 }

 501 static
 502 void
 503 showmenu(const char *name, const char *x[])
 504 {
 505         int ct, half, i;
 506 
 507         kprintf("\n");
 508         kprintf("%s\n", name);
 509 
 510         for (i=ct=0; x[i]; i++) {
 511                 ct++;
 512         }
 513         half = (ct+1)/2;
 514 
 515         for (i=0; i<half; i++) {
 516                 kprintf("    %-36s", x[i]);
 517                 if (i+half < ct) {
 518                         kprintf("%s", x[i+half]);
 519                 }
 520                 kprintf("\n");
 521         }
 522 
 523         kprintf("\n");
 524 }

 713 /*
 714  * Process a single command.
 715  */
 716 static
 717 int
 718 cmd_dispatch(char *cmd)
 719 {
 720         struct timespec before, after, duration;
 721         char *args[MAXMENUARGS];
 722         int nargs=0;
 723         char *word;
 724         char *context;
 725         int i, result;
 726 
 727         for (word = strtok_r(cmd, " \t", &context);
 728              word != NULL;
 729              word = strtok_r(NULL, " \t", &context)) {
 730 
 731                 if (nargs >= MAXMENUARGS) {
 732                         kprintf("Command line has too many words\n");
 733                         return E2BIG;
 734                 }
 735                 args[nargs++] = word;
 736         }
 737 
 738         if (nargs==0) {
 739                 return 0;
 740         }
 741 
 742         for (i=0; cmdtable[i].name; i++) {
 743                 if (*cmdtable[i].name && !strcmp(args[0], cmdtable[i].name)) {
 744                         KASSERT(cmdtable[i].func!=NULL);
 745 
 746                         gettime(&before);
 747 
 748                         result = cmdtable[i].func(nargs, args);
 749 
 750                         gettime(&after);
 751                         timespec_sub(&after, &before, &duration);
 752 
 753                         kprintf("Operation took %llu.%09lu seconds\n",
 754                                 (unsigned long long) duration.tv_sec,
 755                                 (unsigned long) duration.tv_nsec);
 756 
 757                         return result;
 758                 }
 759         }
 760 
 761         kprintf("%s: Command not found\n", args[0]);
 762         return EINVAL;
 763 }

 765 /*
 766  * Evaluate a command line that may contain multiple semicolon-delimited
 767  * commands.
 768  *
 769  * If "isargs" is set, we're doing command-line processing; print the
 770  * comamnds as we execute them and panic if the command is invalid or fails.
 771  */
 772 static
 773 void
 774 menu_execute(char *line, int isargs)
 775 {
 776         char *command;
 777         char *context;
 778         int result;
 779 
 780         for (command = strtok_r(line, ";", &context);
 781              command != NULL;
 782              command = strtok_r(NULL, ";", &context)) {
 783 
 784                 if (isargs) {
 785                         kprintf("OS/161 kernel: %s\n", command);
 786                 }
 787 
 788                 result = cmd_dispatch(command);
 789                 if (result) {
 790                         kprintf("Menu command failed: %s\n", strerror(result));
 791                         if (isargs) {
 792                                 panic("Failure processing kernel arguments\n");
 793                         }
 794                 }
 795         }
 796 }

 798 /*
 799  * Command menu main loop.
 800  *
 801  * First, handle arguments passed on the kernel's command line from
 802  * the bootloader. Then loop prompting for commands.
 803  *
 804  * The line passed in from the bootloader is treated as if it had been
 805  * typed at the prompt. Semicolons separate commands; spaces and tabs
 806  * separate words (command names and arguments).
 807  *
