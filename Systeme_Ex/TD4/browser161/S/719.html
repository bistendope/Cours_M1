  30 /*
  31  * Simple test for the user-level semaphores provided by semfs, aka
  32  * "sem:".
  33  *
  34  * This should mostly run once you've implemented open, read, write,
  35  * and fork, and run fully once you've also implemented waitpid.
  36  *
  37  * The last part of the test will generally hang, sometimes in fork,
  38  * unless your filetable/open-file locking is just so.
  39  */
  40 
  41 #include <sys/types.h>
  42 #include <sys/wait.h>
  43 #include <stdio.h>
  44 #include <unistd.h>
  45 #include <fcntl.h>
  46 #include <err.h>

  48 #define ONCELOOPS   3
  49 #define TWICELOOPS  2
  50 #define THRICELOOPS 1
  51 #define LOOPS (ONCELOOPS + 2*TWICELOOPS + 3*THRICELOOPS)
  52 #define NUMJOBS 4

  54 /*
  55  * Print to the console, one character at a time to encourage
  56  * interleaving if the semaphores aren't working.
  57  */
  58 static
  59 void
  60 say(const char *str)
  61 {
  62         size_t i;
  63 
  64         for (i=0; str[i]; i++) {
  65                 putchar(str[i]);
  66         }
  67 }

  85 /*
  86  * This should probably be in libtest.
  87  */
  88 static
  89 void
  90 dowait(pid_t pid, unsigned num)
  91 {
  92         pid_t r;
  93         int status;
  94 
  95         r = waitpid(pid, &status, 0);
  96         if (r < 0) {
  97                 warn("waitpid");
  98                 return;
  99         }
 100         if (WIFSIGNALED(status)) {
 101                 warnx("pid %d (subprocess %u): Signal %d", (int)pid,
 102                       num, WTERMSIG(status));
 103         }
 104         else if (WIFEXITED(status) && WEXITSTATUS(status) != 0) {
 105                 warnx("pid %d (subprocess %u): Exit %d", (int)pid,
 106                      num, WEXITSTATUS(status));
 107         }
 108 }

 113 /*
 114  * Semaphore structure.
 115  */
 116 struct usem {
 117         char name[32];
 118         int fd;
 119 };
 120 
 121 static
 122 void
 123 usem_init(struct usem *sem, const char *tag, unsigned num)
 124 {
 125         snprintf(sem->name, sizeof(sem->name), "sem:usemtest.%s%u", tag, num);
 126         sem->fd = open(sem->name, O_RDWR|O_CREAT|O_TRUNC, 0664);
 127         if (sem->fd < 0) {
 128                 err(1, "%s: create", sem->name);
 129         }
 130         close(sem->fd);
 131         sem->fd = -1;
 132 }
 133 
 134 static
 135 void
 136 usem_open(struct usem *sem)
 137 {
 138         sem->fd = open(sem->name, O_RDWR);
 139         if (sem->fd < 0) {
 140                 err(1, "%s: open", sem->name);
 141         }
 142 }
 143 
 144 static
 145 void
 146 usem_close(struct usem *sem)
 147 {
 148         if (close(sem->fd) == -1) {
 149                 warn("%s: close", sem->name);
 150         }
 151 }
 152 
 153 static
 154 void
 155 usem_cleanup(struct usem *sem)
 156 {
 157         (void)remove(sem->name);
 158 }
 159 
 160 static
 161 void
 162 P(struct usem *sem)
 163 {
 164         ssize_t r;
 165         char c;
 166 
 167         r = read(sem->fd, &c, 1);
 168         if (r < 0) {
 169                 err(1, "%s: read", sem->name);
 170         }
 171         if (r == 0) {
 172                 errx(1, "%s: read: unexpected EOF", sem->name);
 173         }
 174 }
 175 
 176 static
 177 void
 178 V(struct usem *sem)
 179 {
 180         ssize_t r;
 181         char c;
 182 
 183         r = write(sem->fd, &c, 1);
 184         if (r < 0) {
 185                 err(1, "%s: write", sem->name);
 186         }
 187         if (r == 0) {
 188                 errx(1, "%s: write: short count", sem->name);
 189         }
 190 }

 195 static
 196 void
 197 child_plain(struct usem *gosem, struct usem *waitsem, unsigned num)
 198 {
 199         static const char *const strings[NUMJOBS] = {
 200                 "Nitwit!",
 201                 "Blubber!",
 202                 "Oddment!",
 203                 "Tweak!",
 204         };
 205 
 206         const char *string;
 207         unsigned i;
 208 
 209         string = strings[num];
 210         for (i=0; i<LOOPS; i++) {
 211                 P(gosem);
 212                 say(string);
 213                 V(waitsem);
 214         }
 215 }

 217 static
 218 void
 219 child_with_own_fd(struct usem *gosem, struct usem *waitsem, unsigned num)
 220 {
 221         usem_open(gosem);
 222         usem_open(waitsem);
 223         child_plain(gosem, waitsem, num);
 224         usem_close(gosem);
 225         usem_close(waitsem);
 226 }

 228 static
 229 void
 230 baseparent(struct usem *gosems, struct usem *waitsems)
 231 {
 232         unsigned i, j;
 233 
 234         for (i=0; i<NUMJOBS; i++) {
 235                 usem_open(&gosems[i]);
 236                 usem_open(&waitsems[i]);
 237         }
 238 
 239         say("Once...\n");
 240         for (j=0; j<ONCELOOPS; j++) {
 241                 for (i=0; i<NUMJOBS; i++) {
 242                         V(&gosems[i]);
 243                         P(&waitsems[i]);
 244                         putchar(' ');
 245                 }
 246                 putchar('\n');
 247         }
 248 
 249         say("Twice...\n");
 250         for (j=0; j<TWICELOOPS; j++) {
 251                 for (i=0; i<NUMJOBS; i++) {
 252                         V(&gosems[i]);
 253                         P(&waitsems[i]);
 254                         putchar(' ');
 255                         V(&gosems[i]);
 256                         P(&waitsems[i]);
 257                         putchar(' ');
 258                 }
 259                 putchar('\n');
 260         }
 261 
 262         say("Three times...\n");
 263         for (j=0; j<THRICELOOPS; j++) {
 264                 for (i=0; i<NUMJOBS; i++) {
 265                         V(&gosems[i]);
 266                         P(&waitsems[i]);
 267                         putchar(' ');
 268                         V(&gosems[i]);
 269                         P(&waitsems[i]);
 270                         putchar(' ');
 271                         V(&gosems[i]);
 272                         P(&waitsems[i]);
 273                         putchar('\n');
 274                 }
 275         }
 276 
 277         for (i=0; i<NUMJOBS; i++) {
 278                 usem_close(&gosems[i]);
 279                 usem_close(&waitsems[i]);
 280         }
 281 }

 283 static
 284 void
 285 basetest(void)
 286 {
 287         unsigned i;
 288         struct usem gosems[NUMJOBS], waitsems[NUMJOBS];
 289         pid_t pids[NUMJOBS];
 290 
 291         for (i=0; i<NUMJOBS; i++) {
 292                 usem_init(&gosems[i], "g", i);
 293                 usem_init(&waitsems[i], "w", i);
 294         }
 295 
 296         for (i=0; i<NUMJOBS; i++) {
 297                 pids[i] = fork();
 298                 if (pids[i] < 0) {
 299                         err(1, "fork");
 300                 }
 301                 if (pids[i] == 0) {
 302                         child_with_own_fd(&gosems[i], &waitsems[i], i);
 303                         _exit(0);
 304                 }
 305         }
 306         baseparent(gosems, waitsems);
 307 
 308         for (i=0; i<NUMJOBS; i++) {
 309                 dowait(pids[i], i);
 310         }
 311 
 312         for (i=0; i<NUMJOBS; i++) {
 313                 usem_cleanup(&gosems[i]);
 314                 usem_cleanup(&waitsems[i]);
 315         }
 316 }

 338 static
 339 void
 340 conctest(void)
 341 {
 342         unsigned i;
 343         struct usem gosems[NUMJOBS], waitsems[NUMJOBS];
 344         pid_t pids[NUMJOBS];
 345 
 346         say("Shoot...\n");
 347 
 348         for (i=0; i<NUMJOBS; i++) {
 349                 usem_init(&gosems[i], "g", i);
 350                 usem_init(&waitsems[i], "w", i);
 351                 usem_open(&gosems[i]);
 352                 usem_open(&waitsems[i]);
 353         }
 354 
 355         for (i=0; i<NUMJOBS; i++) {
 356                 pids[i] = fork();
 357                 if (pids[i] < 0) {
 358                         err(1, "fork");
 359                 }
 360                 if (pids[i] == 0) {
 361                         child_plain(&gosems[i], &waitsems[i], i);
 362                         _exit(0);
 363                 }
 364         }
 365         concparent(gosems, waitsems);
 366 
 367         for (i=0; i<NUMJOBS; i++) {
 368                 dowait(pids[i], i);
 369         }
 370 
 371         for (i=0; i<NUMJOBS; i++) {
 372                 usem_close(&gosems[i]);
 373                 usem_close(&waitsems[i]);
 374                 usem_cleanup(&gosems[i]);
 375                 usem_cleanup(&waitsems[i]);
 376         }
