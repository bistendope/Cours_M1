  83 dumpval(const char *desc, const char *val)
  84 {
  85         size_t dlen, vlen, used;
  86 
  87         dlen = strlen(desc);
  88         vlen = strlen(val);
  89 
  90         printf("    ");
  91 
  92         printf("%s: %s", desc, val);
  93 
  94         used = dlen + 2 + vlen;
  95         for (; used < 36; used++) {
  96                 putchar(' ');
  97         }
  98 
  99         if (dumppos % 2 == 1) {
 100                 printf("\n");
 101         }
 102         dumppos++;
 103 }

 135 static
 136 uint32_t
 137 readsb(void)
 138 {
 139         struct sfs_superblock sb;
 140 
 141         diskread(&sb, SFS_SUPER_BLOCK);
 142         if (SWAP32(sb.sb_magic) != SFS_MAGIC) {
 143                 errx(1, "Not an sfs filesystem");
 144         }
 145         return SWAP32(sb.sb_nblocks);
 146 }

 148 static
 149 void
 150 dumpsb(void)
 151 {
 152         struct sfs_superblock sb;
 153         unsigned i;
 154 
 155         diskread(&sb, SFS_SUPER_BLOCK);
 156         sb.sb_volname[sizeof(sb.sb_volname)-1] = 0;
 157 
 158         printf("Superblock\n");
 159         printf("----------\n");
 160         dumpvalf("Magic", "0x%8x", SWAP32(sb.sb_magic));
 161         dumpvalf("Size", "%u blocks", SWAP32(sb.sb_nblocks));
 162         dumpvalf("Freemap size", "%u blocks",
 163                  SFS_FREEMAPBLOCKS(SWAP32(sb.sb_nblocks)));
 164         dumpvalf("Block size", "%u bytes", SFS_BLOCKSIZE);
 165         dumplval("Volume name", sb.sb_volname);
 166 
 167         for (i=0; i<ARRAYCOUNT(sb.reserved); i++) {
 168                 if (sb.reserved[i] != 0) {
 169                         printf("    Word %u in reserved area: 0x%x\n",
 170                                i, SWAP32(sb.reserved[i]));
 171                 }
 172         }
 173         printf("\n");
 174 }

 176 static
 177 void
 178 dumpfreemap(uint32_t fsblocks)
 179 {
 180         uint32_t freemapblocks = SFS_FREEMAPBLOCKS(fsblocks);
 181         uint32_t i, j, k, bn;
 182         uint8_t data[SFS_BLOCKSIZE], mask;
 183         char tmp[16];
 184 
 185         printf("Free block bitmap\n");
 186         printf("-----------------\n");
 187         for (i=0; i<freemapblocks; i++) {
 188                 diskread(data, SFS_FREEMAP_START+i);
 189                 printf("    Freemap block #%u in disk block %u: blocks %u - %u"
 190                        " (0x%x - 0x%x)\n",
 191                        i, SFS_FREEMAP_START+i,
 192                        i*SFS_BITSPERBLOCK, (i+1)*SFS_BITSPERBLOCK - 1,
 193                        i*SFS_BITSPERBLOCK, (i+1)*SFS_BITSPERBLOCK - 1);
 194                 for (j=0; j<SFS_BLOCKSIZE; j++) {
 195                         if (j % 8 == 0) {
 196                                 snprintf(tmp, sizeof(tmp), "0x%x",
 197                                          i*SFS_BITSPERBLOCK + j*8);
 198                                 printf("%-7s ", tmp);
 199                         }
 200                         for (k=0; k<8; k++) {
 201                                 bn = i*SFS_BITSPERBLOCK + j*8 + k;
 202                                 mask = 1U << k;
 203                                 if (bn >= fsblocks) {
 204                                         if (data[j] & mask) {
 205                                                 putchar('x');
 206                                         }
 207                                         else {
 208                                                 putchar('!');
 209                                         }
 210                                 }
 211                                 else {
 212                                         if (data[j] & mask) {
 213                                                 putchar('*');
 214                                         }
 215                                         else {
 216                                                 putchar('.');
 217                                         }
 218                                 }
 219                         }
 220                         if (j % 8 == 7) {
 221                                 printf("\n");
 222                         }
 223                         else {
 224                                 printf(" ");
 225                         }
 226                 }
 227         }
 228         printf("\n");
 229 }

 231 static
 232 void
 233 dumpindirect(uint32_t block)
 234 {
 235         uint32_t ib[SFS_BLOCKSIZE/sizeof(uint32_t)];
 236         char tmp[128];
 237         unsigned i;
 238 
 239         if (block == 0) {
 240                 return;
 241         }
 242         printf("Indirect block %u\n", block);
 243 
 244         diskread(ib, block);
 245         for (i=0; i<ARRAYCOUNT(ib); i++) {
 246                 if (i % 4 == 0) {
 247                         printf("@%-3u   ", i);
 248                 }
 249                 snprintf(tmp, sizeof(tmp), "%u (0x%x)",
 250                          SWAP32(ib[i]), SWAP32(ib[i]));
 251                 printf("  %-16s", tmp);
 252                 if (i % 4 == 3) {
 253                         printf("\n");
 254                 }
 255         }
 256 }

 258 static
 259 uint32_t
 260 traverse_ib(uint32_t fileblock, uint32_t numblocks, uint32_t block,
 261             void (*doblock)(uint32_t, uint32_t))
 262 {
 263         uint32_t ib[SFS_BLOCKSIZE/sizeof(uint32_t)];
 264         unsigned i;
 265 
 266         if (block == 0) {
 267                 memset(ib, 0, sizeof(ib));
 268         }
 269         else {
 270                 diskread(ib, block);
 271         }
 272         for (i=0; i<ARRAYCOUNT(ib) && fileblock < numblocks; i++) {
 273                 doblock(fileblock++, SWAP32(ib[i]));
 274         }
 275         return fileblock;
 276 }

 278 static
 279 void
 280 traverse(const struct sfs_dinode *sfi, void (*doblock)(uint32_t, uint32_t))
 281 {
 282         uint32_t fileblock;
 283         uint32_t numblocks;
 284         unsigned i;
 285 
 286         numblocks = DIVROUNDUP(SWAP32(sfi->sfi_size), SFS_BLOCKSIZE);
 287 
 288         fileblock = 0;
 289         for (i=0; i<SFS_NDIRECT && fileblock < numblocks; i++) {
 290                 doblock(fileblock++, SWAP32(sfi->sfi_direct[i]));
 291         }
 292         if (fileblock < numblocks) {
 293                 fileblock = traverse_ib(fileblock, numblocks,
 294                                         SWAP32(sfi->sfi_indirect), doblock);
 295         }
 296         assert(fileblock == numblocks);
 297 }

 299 static
 300 void
 301 dumpdirblock(uint32_t fileblock, uint32_t diskblock)
 302 {
 303         struct sfs_direntry sds[SFS_BLOCKSIZE/sizeof(struct sfs_direntry)];
 304         int nsds = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
 305         int i;
 306 
 307         (void)fileblock;
 308         if (diskblock == 0) {
 309                 printf("    [block %u - empty]\n", diskblock);
 310                 return;
 311         }
 312         diskread(&sds, diskblock);
 313 
 314         printf("    [block %u]\n", diskblock);
 315         for (i=0; i<nsds; i++) {
 316                 uint32_t ino = SWAP32(sds[i].sfd_ino);
 317                 if (ino==SFS_NOINO) {
 318                         printf("        [free entry]\n");
 319                 }
 320                 else {
 321                         sds[i].sfd_name[SFS_NAMELEN-1] = 0; /* just in case */
 322                         printf("        %u %s\n", ino, sds[i].sfd_name);
 323                 }
 324         }
 325 }

 327 static
 328 void
 329 dumpdir(uint32_t ino, const struct sfs_dinode *sfi)
 330 {
 331         int nentries;
 332 
 333         nentries = SWAP32(sfi->sfi_size) / sizeof(struct sfs_direntry);
 334         if (SWAP32(sfi->sfi_size) % sizeof(struct sfs_direntry) != 0) {
 335                 warnx("Warning: dir size is not a multiple of dir entry size");
 336         }
 337         printf("Directory contents for inode %u: %d entries\n", ino, nentries);
 338         traverse(sfi, dumpdirblock);
 339 }

 341 static
 342 void
 343 recursedirblock(uint32_t fileblock, uint32_t diskblock)
 344 {
 345         struct sfs_direntry sds[SFS_BLOCKSIZE/sizeof(struct sfs_direntry)];
 346         int nsds = SFS_BLOCKSIZE/sizeof(struct sfs_direntry);
 347         int i;
 348 
 349         (void)fileblock;
 350         if (diskblock == 0) {
 351                 return;
 352         }
 353         diskread(&sds, diskblock);
 354 
 355         for (i=0; i<nsds; i++) {
 356                 uint32_t ino = SWAP32(sds[i].sfd_ino);
 357                 if (ino==SFS_NOINO) {
 358                         continue;
 359                 }
 360                 sds[i].sfd_name[SFS_NAMELEN-1] = 0; /* just in case */
 361                 dumpinode(ino, sds[i].sfd_name);
 362         }
 363 }

 377 static
 378 void dumpfileblock(uint32_t fileblock, uint32_t diskblock)
 379 {
 380         uint8_t data[SFS_BLOCKSIZE];
 381         unsigned i, j;
 382         char tmp[128];
 383 
 384         if (diskblock == 0) {
 385                 printf("    0x%6x  [sparse]\n", fileblock * SFS_BLOCKSIZE);
 386                 return;
 387         }
 388 
 389         diskread(data, diskblock);
 390         for (i=0; i<SFS_BLOCKSIZE; i++) {
 391                 if (i % 16 == 0) {
 392                         snprintf(tmp, sizeof(tmp), "0x%x",
 393                                  fileblock * SFS_BLOCKSIZE + i);
 394                         printf("%8s", tmp);
 395                 }
 396                 if (i % 8 == 0) {
 397                         printf("  ");
 398                 }
 399                 else {
 400                         printf(" ");
 401                 }
 402                 printf("%02x", data[i]);
 403                 if (i % 16 == 15) {
 404                         printf("  ");
 405                         for (j = i-15; j<=i; j++) {
 406                                 if (data[j] < 32 || data[j] > 126) {
 407                                         putchar('.');
 408                                 }
 409                                 else {
 410                                         putchar(data[j]);
 411                                 }
 412                         }
 413                         printf("\n");
 414                 }
 415         }
 416 }

 426 static
 427 void
 428 dumpinode(uint32_t ino, const char *name)
 429 {
 430         struct sfs_dinode sfi;
 431         const char *typename;
 432         char tmp[128];
 433         unsigned i;
 434 
 435         diskread(&sfi, ino);
 436 
 437         printf("Inode %u", ino);
 438         if (name != NULL) {
 439                 printf(" (%s)", name);
 440         }
 441         printf("\n");
 442         printf("--------------\n");
 443 
 444         switch (SWAP16(sfi.sfi_type)) {
 445             case SFS_TYPE_FILE: typename = "regular file"; break;
 446             case SFS_TYPE_DIR: typename = "directory"; break;
 447             default: typename = "invalid"; break;
 448         }
 449         dumpvalf("Type", "%u (%s)", SWAP16(sfi.sfi_type), typename);
 450         dumpvalf("Size", "%u", SWAP32(sfi.sfi_size));
 451         dumpvalf("Link count", "%u", SWAP16(sfi.sfi_linkcount));
 452         printf("\n");
 453 
 454         printf("    Direct blocks:\n");
 455         for (i=0; i<SFS_NDIRECT; i++) {
 456                 if (i % 4 == 0) {
 457                         printf("@%-2u    ", i);
 458                 }
 459                 /*
 460                  * Assume the disk size might be > 64K sectors (which
 461                  * would be 32M) but is < 1024K sectors (512M) so we
 462                  * need up to 5 hex digits for a block number. And
 463                  * assume it's actually < 1 million sectors so we need
 464                  * only up to 6 decimal digits. The complete block
 465                  * number print then needs up to 16 digits.
 466                  */
 467                 snprintf(tmp, sizeof(tmp), "%u (0x%x)",
 468                          SWAP32(sfi.sfi_direct[i]), SWAP32(sfi.sfi_direct[i]));
 469                 printf("  %-16s", tmp);
 470                 if (i % 4 == 3) {
 471                         printf("\n");
 472                 }
 473         }
 474         if (i % 4 != 0) {
 475                 printf("\n");
 476         }
 477         printf("    Indirect block: %u (0x%x)\n",
 478                SWAP32(sfi.sfi_indirect), SWAP32(sfi.sfi_indirect));
 479         for (i=0; i<ARRAYCOUNT(sfi.sfi_waste); i++) {
 480                 if (sfi.sfi_waste[i] != 0) {
 481                         printf("    Word %u in waste area: 0x%x\n",
 482                                i, SWAP32(sfi.sfi_waste[i]));
 483                 }
 484         }
 485 
 486         if (doindirect) {
 487                 dumpindirect(SWAP32(sfi.sfi_indirect));
 488         }
 489 
 490         if (SWAP16(sfi.sfi_type) == SFS_TYPE_DIR && dodirs) {
 491                 dumpdir(ino, &sfi);
 492         }
 493         if (SWAP16(sfi.sfi_type) == SFS_TYPE_FILE && dofiles) {
 494                 dumpfile(ino, &sfi);
 495         }
 496         if (SWAP16(sfi.sfi_type) == SFS_TYPE_DIR && recurse) {
 497                 recursedir(ino, &sfi);
 498         }
 499 }

 504 static
 505 void
 506 usage(void)
 507 {
 508         warnx("Usage: dumpsfs [options] device/diskfile");
 509         warnx("   -s: dump superblock");
 510         warnx("   -b: dump free block bitmap");
 511         warnx("   -i ino: dump specified inode");
 512         warnx("   -I: dump indirect blocks");
 513         warnx("   -f: dump file contents");
 514         warnx("   -d: dump directory contents");
 515         warnx("   -r: recurse into directory contents");
 516         warnx("   -a: equivalent to -sbdfr -i 1");
 517         errx(1, "   Default is -i 1");

 520 int
 521 main(int argc, char **argv)
 522 {
 523         bool dosb = false;
 524         bool dofreemap = false;
 525         uint32_t dumpino = 0;
 526         const char *dumpdisk = NULL;
 527 
 528         int i, j;
 529         uint32_t nblocks;
 530 
 531 #ifdef HOST
 532         /* Don't do this; it frobs the tty and you can't pipe to less */
 533         /*hostcompat_init(argc, argv);*/
 534         hostcompat_progname = argv[0];
 535 #endif
 536 
 537         for (i=1; i<argc; i++) {
 538                 if (argv[i][0] == '-') {
 539                         for (j=1; argv[i][j]; j++) {
 540                                 switch (argv[i][j]) {
 541                                     case 's': dosb = true; break;
 542                                     case 'b': dofreemap = true; break;
 543                                     case 'i':
 544                                         if (argv[i][j+1] == 0) {
 545                                                 dumpino = atoi(argv[++i]);
 546                                         }
 547                                         else {
 548                                                 dumpino = atoi(argv[i]+j+1);
 549                                                 j = strlen(argv[i]);
 550                                         }
 551                                         /* XXX ugly */
 552                                         goto nextarg;
 553                                     case 'I': doindirect = true; break;
 554                                     case 'f': dofiles = true; break;
 555                                     case 'd': dodirs = true; break;
 556                                     case 'r': recurse = true; break;
 557                                     case 'a':
 558                                         dosb = true;
 559                                         dofreemap = true;
 560                                         if (dumpino == 0) {
 561                                                 dumpino = SFS_ROOTDIR_INO;
 562                                         }
 563                                         doindirect = true;
 564                                         dofiles = true;
 565                                         dodirs = true;
 566                                         recurse = true;
 567                                         break;
 568                                     default:
 569                                         usage();
 570                                         break;
 571                                 }
 572                         }
 573                 }
 574                 else {
 575                         if (dumpdisk != NULL) {
 576                                 usage();
 577                         }
 578                         dumpdisk = argv[i];
 579                 }
 580          nextarg:
 581                 ;
 582         }
 583         if (dumpdisk == NULL) {
 584                 usage();
 585         }
 586 
 587         if (!dosb && !dofreemap && dumpino == 0) {
 588                 dumpino = SFS_ROOTDIR_INO;
 589         }
 590 
 591         opendisk(dumpdisk);
 592         nblocks = readsb();
 593 
 594         if (dosb) {
 595                 dumpsb();
 596         }
 597         if (dofreemap) {
 598                 dumpfreemap(nblocks);
 599         }
 600         if (dumpino != 0) {
 601                 dumpinode(dumpino, NULL);
 602         }
 603 
 604         closedisk();
 605 
 606         return 0;
 607 }
