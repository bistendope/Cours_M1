  48 /*
  49  * Note: sfs_readblock is used to read the superblock
  50  * early in mount, before sfs is fully (or even mostly)
  51  * initialized, and so may not use anything from sfs
  52  * except sfs_device.

  55 /*
  56  * Read or write a block, retrying I/O errors.
  57  */
  58 static
  59 int
  60 sfs_rwblock(struct sfs_fs *sfs, struct uio *uio)
  61 {
  62         int result;
  63         int tries=0;
  64 
  65         KASSERT(vfs_biglock_do_i_hold());
  66 
  67         DEBUG(DB_SFS, "sfs: %s %llu\n",
  68               uio->uio_rw == UIO_READ ? "read" : "write",
  69               uio->uio_offset / SFS_BLOCKSIZE);
  70 
  71  retry:
  72         result = DEVOP_IO(sfs->sfs_device, uio);
  73         if (result == EINVAL) {
  74                 /*
  75                  * This means the sector we requested was out of range,
  76                  * or the seek address we gave wasn't sector-aligned,
  77                  * or a couple of other things that are our fault.
  78                  */
  79                 panic("sfs: %s: DEVOP_IO returned EINVAL\n",
  80                       sfs->sfs_sb.sb_volname);
  81         }
  82         if (result == EIO) {
  83                 if (tries == 0) {
  84                         tries++;
  85                         kprintf("sfs: %s: block %llu I/O error, retrying\n",
  86                                 sfs->sfs_sb.sb_volname,
  87                                 uio->uio_offset / SFS_BLOCKSIZE);
  88                         goto retry;
  89                 }
  90                 else if (tries < 10) {
  91                         tries++;
  92                         goto retry;
  93                 }
  94                 else {
  95                         kprintf("sfs: %s: block %llu I/O error, giving up "
  96                                 "after %d retries\n",
  97                                 sfs->sfs_sb.sb_volname,
  98                                 uio->uio_offset / SFS_BLOCKSIZE, tries);
  99                 }
 100         }
 101         return result;
 102 }

 104 /*
 105  * Read a block.
 106  */
 107 int
 108 sfs_readblock(struct sfs_fs *sfs, daddr_t block, void *data, size_t len)
 109 {
 110         struct iovec iov;
 111         struct uio ku;
 112 
 113         KASSERT(len == SFS_BLOCKSIZE);
 114 
 115         SFSUIO(&iov, &ku, data, block, UIO_READ);
 116         return sfs_rwblock(sfs, &ku);
 117 }

 127 
 128         KASSERT(len == SFS_BLOCKSIZE);

 138 /*
 139  * Do I/O to a block of a file that doesn't cover the whole block.  We
 140  * need to read in the original block first, even if we're writing, so
 141  * we don't clobber the portion of the block we're not intending to
 142  * write over.
 143  *
 144  * SKIPSTART is the number of bytes to skip past at the beginning of
 145  * the sector; LEN is the number of bytes to actually read or write.
 146  * UIO is the area to do the I/O into.
 147  */
 148 static
 149 int
 150 sfs_partialio(struct sfs_vnode *sv, struct uio *uio,
 151               uint32_t skipstart, uint32_t len)
 152 {
 153         /*
 154          * I/O buffer for handling partial sectors.
 155          *
 156          * Note: in real life (and when you've done the fs assignment)
 157          * you would get space from the disk buffer cache for this,
 158          * not use a static area.
 159          */
 160         static char iobuf[SFS_BLOCKSIZE];
 161 
 162         struct sfs_fs *sfs = sv->sv_absvn.vn_fs->fs_data;
 163         daddr_t diskblock;
 164         uint32_t fileblock;
 165         int result;
 166 
 167         /* Allocate missing blocks if and only if we're writing */
 168         bool doalloc = (uio->uio_rw==UIO_WRITE);
 169 
 170         KASSERT(skipstart + len <= SFS_BLOCKSIZE);
 171 
 172         /* We're using a global static buffer; it had better be locked */
 173         KASSERT(vfs_biglock_do_i_hold());
 174 
 175         /* Compute the block offset of this block in the file */
 176         fileblock = uio->uio_offset / SFS_BLOCKSIZE;
 177 
 178         /* Get the disk block number */
 179         result = sfs_bmap(sv, fileblock, doalloc, &diskblock);
 180         if (result) {
 181                 return result;
 182         }
 183 
 184         if (diskblock == 0) {
 185                 /*
 186                  * There was no block mapped at this point in the file.
 187                  * Zero the buffer.
 188                  */
 189                 KASSERT(uio->uio_rw == UIO_READ);
 190                 bzero(iobuf, sizeof(iobuf));
 191         }
 192         else {
 193                 /*
 194                  * Read the block.
 195                  */
 196                 result = sfs_readblock(sfs, diskblock, iobuf, sizeof(iobuf));
 197                 if (result) {
 198                         return result;
 199                 }
 200         }
 201 
 202         /*
 203          * Now perform the requested operation into/out of the buffer.
 204          */
 205         result = uiomove(iobuf+skipstart, len, uio);
 206         if (result) {
 207                 return result;
 208         }
 209 
 210         /*
 211          * If it was a write, write back the modified block.
 212          */
 213         if (uio->uio_rw == UIO_WRITE) {
 214                 result = sfs_writeblock(sfs, diskblock, iobuf, sizeof(iobuf));
 215                 if (result) {
 216                         return result;
 217                 }
 218         }
 219 
 220         return 0;
 221 }

 223 /*
 224  * Do I/O (either read or write) of a single whole block.
 225  */
 226 static
 227 int
 228 sfs_blockio(struct sfs_vnode *sv, struct uio *uio)
 229 {
 230         struct sfs_fs *sfs = sv->sv_absvn.vn_fs->fs_data;
 231         daddr_t diskblock;
 232         uint32_t fileblock;
 233         int result;
 234         bool doalloc = (uio->uio_rw==UIO_WRITE);
 235         off_t saveoff;
 236         off_t diskoff;
 237         off_t saveres;
 238         off_t diskres;
 239 
 240         /* Get the block number within the file */
 241         fileblock = uio->uio_offset / SFS_BLOCKSIZE;
 242 
 243         /* Look up the disk block number */
 244         result = sfs_bmap(sv, fileblock, doalloc, &diskblock);
 245         if (result) {
 246                 return result;
 247         }
 248 
 249         if (diskblock == 0) {
 250                 /*
 251                  * No block - fill with zeros.
 252                  *
 253                  * We must be reading, or sfs_bmap would have
 254                  * allocated a block for us.
 255                  */
 256                 KASSERT(uio->uio_rw == UIO_READ);
 257                 return uiomovezeros(SFS_BLOCKSIZE, uio);
 258         }
 259 
 260         /*
 261          * Do the I/O directly to the uio region. Save the uio_offset,
 262          * and substitute one that makes sense to the device.
 263          */
 264         saveoff = uio->uio_offset;
 265         diskoff = diskblock * SFS_BLOCKSIZE;
 266         uio->uio_offset = diskoff;
 267 
 268         /*
 269          * Temporarily set the residue to be one block size.
 270          */
 271         KASSERT(uio->uio_resid >= SFS_BLOCKSIZE);
 272         saveres = uio->uio_resid;
 273         diskres = SFS_BLOCKSIZE;
 274         uio->uio_resid = diskres;
 275 
 276         result = sfs_rwblock(sfs, uio);
 277 
 278         /*
 279          * Now, restore the original uio_offset and uio_resid and update
 280          * them by the amount of I/O done.
 281          */
 282         uio->uio_offset = (uio->uio_offset - diskoff) + saveoff;
 283         uio->uio_resid = (uio->uio_resid - diskres) + saveres;
 284 
 285         return result;
 286 }

 288 /*
 289  * Do I/O of a whole region of data, whether or not it's block-aligned.
 290  */
 291 int
 292 sfs_io(struct sfs_vnode *sv, struct uio *uio)
 293 {
 294         uint32_t blkoff;
 295         uint32_t nblocks, i;
 296         int result = 0;
 297         uint32_t origresid, extraresid = 0;
 298 
 299         origresid = uio->uio_resid;
 300 
 301         /*
 302          * If reading, check for EOF. If we can read a partial area,
 303          * remember how much extra there was in EXTRARESID so we can
 304          * add it back to uio_resid at the end.
 305          */
 306         if (uio->uio_rw == UIO_READ) {
 307                 off_t size = sv->sv_i.sfi_size;
 308                 off_t endpos = uio->uio_offset + uio->uio_resid;
 309 
 310                 if (uio->uio_offset >= size) {
 311                         /* At or past EOF - just return */
 312                         return 0;
 313                 }
 314 
 315                 if (endpos > size) {
 316                         extraresid = endpos - size;
 317                         KASSERT(uio->uio_resid > extraresid);
 318                         uio->uio_resid -= extraresid;
 319                 }
 320         }
 321 
 322         /*
 323          * First, do any leading partial block.
 324          */
 325         blkoff = uio->uio_offset % SFS_BLOCKSIZE;
 326         if (blkoff != 0) {
 327                 /* Number of bytes at beginning of block to skip */
 328                 uint32_t skip = blkoff;
 329 
 330                 /* Number of bytes to read/write after that point */
 331                 uint32_t len = SFS_BLOCKSIZE - blkoff;
 332 
 333                 /* ...which might be less than the rest of the block */
 334                 if (len > uio->uio_resid) {
 335                         len = uio->uio_resid;
 336                 }
 337 
 338                 /* Call sfs_partialio() to do it. */
 339                 result = sfs_partialio(sv, uio, skip, len);
 340                 if (result) {
 341                         goto out;
 342                 }
 343         }
 344 
 345         /* If we're done, quit. */
 346         if (uio->uio_resid==0) {
 347                 goto out;
 348         }
 349 
 350         /*
 351          * Now we should be block-aligned. Do the remaining whole blocks.
 352          */
 353         KASSERT(uio->uio_offset % SFS_BLOCKSIZE == 0);
 354         nblocks = uio->uio_resid / SFS_BLOCKSIZE;
 355         for (i=0; i<nblocks; i++) {
 356                 result = sfs_blockio(sv, uio);
 357                 if (result) {
 358                         goto out;
 359                 }
 360         }
 361 
 362         /*
 363          * Now do any remaining partial block at the end.
 364          */
 365         KASSERT(uio->uio_resid < SFS_BLOCKSIZE);
 366 
 367         if (uio->uio_resid > 0) {
 368                 result = sfs_partialio(sv, uio, 0, uio->uio_resid);
 369                 if (result) {
 370                         goto out;
 371                 }
 372         }
 373 
 374  out:
 375 
 376         /* If writing and we did anything, adjust file length */
 377         if (uio->uio_resid != origresid &&
 378             uio->uio_rw == UIO_WRITE &&
 379             uio->uio_offset > (off_t)sv->sv_i.sfi_size) {
 380                 sv->sv_i.sfi_size = uio->uio_offset;
 381                 sv->sv_dirty = true;
 382         }
 383 
 384         /* Add in any extra amount we couldn't read because of EOF */
 385         uio->uio_resid += extraresid;
 386 
 387         /* Done */
 388         return result;
 389 }

 391 ////////////////////////////////////////////////////////////
 392 // Metadata I/O
 393 
 394 /*
 395  * This is much the same as sfs_partialio, but intended for use with
 396  * metadata (e.g. directory entries). It assumes the objects being
 397  * handled are smaller than whole blocks, do not cross block
 398  * boundaries, and originate in the kernel.
 399  *
 400  * It is separate from sfs_partialio because, although there is no
 401  * such code in this version of SFS, it is often desirable when doing
 402  * more advanced things to handle metadata and user data I/O
 403  * differently.
 404  */
 405 int
 406 sfs_metaio(struct sfs_vnode *sv, off_t actualpos, void *data, size_t len,
 407            enum uio_rw rw)
 408 {
 409         struct sfs_fs *sfs = sv->sv_absvn.vn_fs->fs_data;
 410         off_t endpos;
 411         uint32_t vnblock;
 412         uint32_t blockoffset;
 413         daddr_t diskblock;
 414         bool doalloc;
 415         int result;
 416 
 417         /*
 418          * I/O buffer for metadata ops.
 419          *
 420          * Note: in real life (and when you've done the fs assignment) you
 421          * would get space from the disk buffer cache for this, not use a
 422          * static area.
 423          */
 424         static char metaiobuf[SFS_BLOCKSIZE];
 425 
 426         /* We're using a global static buffer; it had better be locked */
 427         KASSERT(vfs_biglock_do_i_hold());
 428 
 429         /* Figure out which block of the vnode (directory, whatever) this is */
 430         vnblock = actualpos / SFS_BLOCKSIZE;
 431         blockoffset = actualpos % SFS_BLOCKSIZE;
 432 
 433         /* Get the disk block number */
 434         doalloc = (rw == UIO_WRITE);
 435         result = sfs_bmap(sv, vnblock, doalloc, &diskblock);
 436         if (result) {
 437                 return result;
 438         }
 439 
 440         if (diskblock == 0) {
 441                 /* Should only get block 0 back if doalloc is false */
 442                 KASSERT(rw == UIO_READ);
 443 
 444                 /* Sparse file, read as zeros. */
 445                 bzero(data, len);
 446                 return 0;
 447         }
 448 
 449         /* Read the block */
 450         result = sfs_readblock(sfs, diskblock, metaiobuf, sizeof(metaiobuf));
 451         if (result) {
 452                 return result;
 453         }
 454 
 455         if (rw == UIO_READ) {
 456                 /* Copy out the selected region */
 457                 memcpy(data, metaiobuf + blockoffset, len);
 458         }
 459         else {
 460                 /* Update the selected region */
 461                 memcpy(metaiobuf + blockoffset, data, len);
 462 
 463                 /* Write the block back */
 464                 result = sfs_writeblock(sfs, diskblock,
 465                                         metaiobuf, sizeof(metaiobuf));
 466                 if (result) {
 467                         return result;
 468                 }
 469 
 470                 /* Update the vnode size if needed */
 471                 endpos = actualpos + len;
 472                 if (endpos > (off_t)sv->sv_i.sfi_size) {
 473                         sv->sv_i.sfi_size = endpos;
 474                         sv->sv_dirty = true;
 475                 }
 476         }
 477 
 478         /* Done */
 479         return 0;
 480 }
