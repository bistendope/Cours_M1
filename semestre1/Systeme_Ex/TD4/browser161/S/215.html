  46 /*
  47  * Test kmalloc; allocate ITEMSIZE bytes NTRIES times, freeing
  48  * somewhat later.
  49  *
  50  * The total of ITEMSIZE * NTRIES is intended to exceed the size of
  51  * available memory.
  52  *
  53  * kmallocstress does the same thing, but from NTHREADS different
  54  * threads at once.
  55  */
  56 
  57 #define NTRIES   1200
  58 #define ITEMSIZE  997
  59 #define NTHREADS  8
  60 
  61 static
  62 void
  63 kmallocthread(void *sm, unsigned long num)
  64 {
  65         struct semaphore *sem = sm;
  66         void *ptr;
  67         void *oldptr=NULL;
  68         void *oldptr2=NULL;
  69         int i;
  70 
  71         for (i=0; i<NTRIES; i++) {
  72                 ptr = kmalloc(ITEMSIZE);
  73                 if (ptr==NULL) {
  74                         if (sem) {
  75                                 kprintf("thread %lu: kmalloc returned NULL\n",
  76                                         num);
  77                                 goto done;
  78                         }
  79                         kprintf("kmalloc returned null; test failed.\n");
  80                         goto done;
  81                 }
  82                 if (oldptr2) {
  83                         kfree(oldptr2);
  84                 }
  85                 oldptr2 = oldptr;
  86                 oldptr = ptr;
  87         }
  88 done:
  89         if (oldptr2) {
  90                 kfree(oldptr2);
  91         }
  92         if (oldptr) {
  93                 kfree(oldptr);
  94         }
  95         if (sem) {
  96                 V(sem);
  97         }
  98 }

 113 int
 114 kmallocstress(int nargs, char **args)
 115 {
 116         struct semaphore *sem;
 117         int i, result;
 118 
 119         (void)nargs;
 120         (void)args;
 121 
 122         sem = sem_create("kmallocstress", 0);
 123         if (sem == NULL) {
 124                 panic("kmallocstress: sem_create failed\n");
 125         }
 126 
 127         kprintf("Starting kmalloc stress test...\n");
 128 
 129         for (i=0; i<NTHREADS; i++) {
 130                 result = thread_fork("kmallocstress", NULL,
 131                                      kmallocthread, sem, i);
 132                 if (result) {
 133                         panic("kmallocstress: thread_fork failed: %s\n",
 134                               strerror(result));
 135                 }
 136         }
 137 
 138         for (i=0; i<NTHREADS; i++) {
 139                 P(sem);
 140         }
 141 
 142         sem_destroy(sem);
 143         kprintf("kmalloc stress test done\n");
 144 
 145         return 0;
 146 }

 151 /*
 152  * Larger kmalloc test. Or at least, potentially larger. The size is
 153  * an argument.
 154  *
 155  * The argument specifies the number of objects to allocate; the size
 156  * of each allocation rotates through sizes[]. (FUTURE: should there
 157  * be a mode that allocates random sizes?) In order to hold the
 158  * pointers returned by kmalloc we first allocate a two-level radix
 159  * tree whose lower tier is made up of blocks of size PAGE_SIZE/4.
 160  * (This is so they all go to the subpage allocator rather than being
 161  * whole-page allocations.)
 162  *
 163  * Since PAGE_SIZE is commonly 4096, each of these blocks holds 1024
 164  * pointers (on a 32-bit machine) or 512 (on a 64-bit machine) and so
 165  * we can store considerably more pointers than we have memory for
 166  * before the upper tier becomes a whole page or otherwise gets
 167  * uncomfortably large.
 168  *
 169  * Having set this up, the test just allocates and then frees all the
 170  * pointers in order, setting and checking the contents.
 171  */
 172 int
 173 kmalloctest3(int nargs, char **args)
 174 {
 175 #define NUM_KM3_SIZES 5
 176         static const unsigned sizes[NUM_KM3_SIZES] = { 32, 41, 109, 86, 9 };
 177         unsigned numptrs;
 178         size_t ptrspace;
 179         size_t blocksize;
 180         unsigned numptrblocks;
 181         void ***ptrblocks;
 182         unsigned curblock, curpos, cursizeindex, cursize;
 183         size_t totalsize;
 184         unsigned i, j;
 185         unsigned char *ptr;
 186 
 187         if (nargs != 2) {
 188                 kprintf("kmalloctest3: usage: km3 numobjects\n");
 189                 return EINVAL;
 190         }
 191 
 192         /* Figure out how many pointers we'll get and the space they need. */
 193         numptrs = atoi(args[1]);
 194         ptrspace = numptrs * sizeof(void *);
 195 
 196         /* Figure out how many blocks in the lower tier. */
 197         blocksize = PAGE_SIZE / 4;
 198         numptrblocks = DIVROUNDUP(ptrspace, blocksize);
 199 
 200         kprintf("kmalloctest3: %u objects, %u pointer blocks\n",
 201                 numptrs, numptrblocks);
 202 
 203         /* Allocate the upper tier. */
 204         ptrblocks = kmalloc(numptrblocks * sizeof(ptrblocks[0]));
 205         if (ptrblocks == NULL) {
 206                 panic("kmalloctest3: failed on pointer block array\n");
 207         }
 208         /* Allocate the lower tier. */
 209         for (i=0; i<numptrblocks; i++) {
 210                 ptrblocks[i] = kmalloc(blocksize);
 211                 if (ptrblocks[i] == NULL) {
 212                         panic("kmalloctest3: failed on pointer block %u\n", i);
 213                 }
 214         }
 215 
 216         /* Allocate the objects. */
 217         curblock = 0;
 218         curpos = 0;
 219         cursizeindex = 0;
 220         totalsize = 0;
 221         for (i=0; i<numptrs; i++) {
 222                 cursize = sizes[cursizeindex];
 223                 ptr = kmalloc(cursize);
 224                 if (ptr == NULL) {
 225                         kprintf("kmalloctest3: failed on object %u size %u\n",
 226                                 i, cursize);
 227                         kprintf("kmalloctest3: pos %u in pointer block %u\n",
 228                                 curpos, curblock);
 229                         kprintf("kmalloctest3: total so far %zu\n", totalsize);
 230                         panic("kmalloctest3: failed.\n");
 231                 }
 232                 /* Fill the object with its number. */
 233                 for (j=0; j<cursize; j++) {
 234                         ptr[j] = (unsigned char) i;
 235                 }
 236                 /* Move to the next slot in the tree. */
 237                 ptrblocks[curblock][curpos] = ptr;
 238                 curpos++;
 239                 if (curpos >= blocksize / sizeof(void *)) {
 240                         curblock++;
 241                         curpos = 0;
 242                 }
 243                 /* Update the running total, and rotate the size. */
 244                 totalsize += cursize;
 245                 cursizeindex = (cursizeindex + 1) % NUM_KM3_SIZES;
 246         }
 247 
 248         kprintf("kmalloctest3: %zu bytes allocated\n", totalsize);
 249 
 250         /* Free the objects. */
 251         curblock = 0;
 252         curpos = 0;
 253         cursizeindex = 0;
 254         for (i=0; i<numptrs; i++) {
 255                 cursize = sizes[cursizeindex];
 256                 ptr = ptrblocks[curblock][curpos];
 257                 KASSERT(ptr != NULL);
 258                 for (j=0; j<cursize; j++) {
 259                         if (ptr[j] == (unsigned char) i) {
 260                                 continue;
 261                         }
 262                         kprintf("kmalloctest3: failed on object %u size %u\n",
 263                                 i, cursize);
 264                         kprintf("kmalloctest3: pos %u in pointer block %u\n",
 265                                 curpos, curblock);
 266                         kprintf("kmalloctest3: at object offset %u\n", j);
 267                         kprintf("kmalloctest3: expected 0x%x, found 0x%x\n",
 268                                 ptr[j], (unsigned char) i);
 269                         panic("kmalloctest3: failed.\n");
 270                 }
 271                 kfree(ptr);
 272                 curpos++;
 273                 if (curpos >= blocksize / sizeof(void *)) {
 274                         curblock++;
 275                         curpos = 0;
 276                 }
 277                 KASSERT(totalsize > 0);
 278                 totalsize -= cursize;
 279                 cursizeindex = (cursizeindex + 1) % NUM_KM3_SIZES;
 280         }
 281         KASSERT(totalsize == 0);
 282 
 283         /* Free the lower tier. */
 284         for (i=0; i<numptrblocks; i++) {
 285                 KASSERT(ptrblocks[i] != NULL);
 286                 kfree(ptrblocks[i]);
 287         }
 288         /* Free the upper tier. */
 289         kfree(ptrblocks);
 290 
 291         kprintf("kmalloctest3: passed\n");
 292         return 0;
 293 }

 298 static
 299 void
 300 kmalloctest4thread(void *sm, unsigned long num)
 301 {
 302 #define NUM_KM4_SIZES 5
 303         static const unsigned sizes[NUM_KM4_SIZES] = { 1, 3, 5, 2, 4 };
 304 
 305         struct semaphore *sem = sm;
 306         void *ptrs[NUM_KM4_SIZES];
 307         unsigned p, q;
 308         unsigned i;
 309 
 310         for (i=0; i<NUM_KM4_SIZES; i++) {
 311                 ptrs[i] = NULL;
 312         }
 313         p = 0;
 314         q = NUM_KM4_SIZES / 2;
 315 
 316         for (i=0; i<NTRIES; i++) {
 317                 if (ptrs[q] != NULL) {
 318                         kfree(ptrs[q]);
 319                         ptrs[q] = NULL;
 320                 }
 321                 ptrs[p] = kmalloc(sizes[p] * PAGE_SIZE);
 322                 if (ptrs[p] == NULL) {
 323                         panic("kmalloctest4: thread %lu: "
 324                               "allocating %u pages failed\n",
 325                               num, sizes[p]);
 326                 }
 327                 p = (p + 1) % NUM_KM4_SIZES;
 328                 q = (q + 1) % NUM_KM4_SIZES;
 329         }
 330 
 331         for (i=0; i<NUM_KM4_SIZES; i++) {
 332                 if (ptrs[i] != NULL) {
 333                         kfree(ptrs[i]);
 334                 }
 335         }
 336 
 337         V(sem);
 338 }

 340 int
 341 kmalloctest4(int nargs, char **args)
 342 {
 343         struct semaphore *sem;
 344         unsigned nthreads;
 345         unsigned i;
 346         int result;
 347 
 348         (void)nargs;
 349         (void)args;
 350 
 351         kprintf("Starting multipage kmalloc test...\n");
 352 #if OPT_DUMBVM
 353         kprintf("(This test will not work with dumbvm)\n");
 354 #endif
 355 
 356         sem = sem_create("kmalloctest4", 0);
 357         if (sem == NULL) {
 358                 panic("kmalloctest4: sem_create failed\n");
 359         }
 360 
 361         /* use 6 instead of 8 threads */
 362         nthreads = (3*NTHREADS)/4;
 363 
 364         for (i=0; i<nthreads; i++) {
 365                 result = thread_fork("kmalloctest4", NULL,
 366                                      kmalloctest4thread, sem, i);
 367                 if (result) {
 368                         panic("kmallocstress: thread_fork failed: %s\n",
 369                               strerror(result));
 370                 }
 371         }
 372 
 373         for (i=0; i<nthreads; i++) {
 374                 P(sem);
 375         }
 376 
 377         sem_destroy(sem);
 378         kprintf("Multipage kmalloc test done\n");
 379         return 0;
