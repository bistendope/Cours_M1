  51 /*
  52  * Tell GCC how to check printf formats. Also tell it about functions
  53  * that don't return, as this is helpful for avoiding bogus warnings
  54  * about uninitialized variables.
  55  */
  56 #ifdef __GNUC__
  57 #define __PF(a,b) __attribute__((__format__(__printf__, a, b)))
  58 #define __DEAD    __attribute__((__noreturn__))
  59 #define __UNUSED  __attribute__((__unused__))
  60 #else
  61 #define __PF(a,b)
  62 #define __DEAD
  63 #define __UNUSED
  64 #endif

  67 /*
  68  * Material for supporting inline functions.
  69  *
  70  * A function marked inline can be handled by the compiler in three
  71  * ways: in addition to possibly inlining into the code for other
  72  * functions, the compiler can (1) generate a file-static out-of-line
  73  * copy of the function, (2) generate a global out-of-line copy of the
  74  * function, or (3) generate no out-of-line copy of the function.
  75  *
  76  * None of these alone is thoroughly satisfactory. Since an inline
  77  * function may or may not be inlined at the compiler's discretion, if
  78  * no out-of-line copy exists the build may fail at link time with
  79  * undefined symbols. Meanwhile, if the compiler is told to generate a
  80  * global out-of-line copy, it will generate one such copy for every
  81  * source file where the inline definition is visible; since inline
  82  * functions tend to appear in header files, this leads to multiply
  83  * defined symbols and build failure. The file-static option isn't
  84  * really an improvement, either: one tends to get compiler warnings
  85  * about inline functions that haven't been used, which for any
  86  * particular source file tends to be at least some of the ones that
  87  * have been defined. Furthermore, this method leads to one
  88  * out-of-line copy of the inline function per source file that uses
  89  * it, which not only wastes space but makes debugging painful.
  90  *
  91  * Therefore, we use the following scheme.
  92  *
  93  * In the header file containing the inline functions for the module
  94  * "foo", we put
  95  *
  96  *      #ifndef FOO_INLINE
  97  *      #define FOO_INLINE INLINE
  98  *      #endif
  99  *
 100  * where INLINE selects the compiler behavior that does *not* generate
 101  * an out-of-line version. Then we define the inline functions
 102  * themselves as FOO_INLINE. This allows the compiler to inline the
 103  * functions anywhere it sees fit with a minimum of hassles. Then,
 104  * when compiling foo.c, before including headers we put
 105  *
 106  *      #define FOO_INLINE  // empty
 107  *
 108  * which causes the inline functions to appear as ordinary function
 109  * definitions, not inline at all, when foo.c is compiled. This
 110  * ensures that an out-of-line definition appears, and furthermore
 111  * ensures that the out-of-line definition is the same as the inline
 112  * definition.
 113  *
 114  * The situation is complicated further because gcc is historically
 115  * not compliant with the C standard. In C99, "inline" means "do not
 116  * generate an out-of-line copy" and "extern inline" means "generate a
 117  * global out-of-line copy". In gcc, going back far longer than C99,
 118  * the meanings were reversed. This eventually changed, but varies
 119  * with compiler version and options. The macro __GNUC_STDC_INLINE__
 120  * is defined if the behavior is C99-compliant.
 121  *
 122  * (Note that inline functions that appear only within a single source
 123  * file can safely be declared "static inline"; to avoid whining from
 124  * compiler in some contexts you may also want to add __UNUSED to
 125  * that.)
 126  */
 127 #if defined(__GNUC__) && !defined(__GNUC_STDC_INLINE__)
 128 /* gcc's non-C99 inline semantics */
 129 #define INLINE extern inline
 130 
 131 #elif defined(__STDC__) && __STDC_VERSION__ >= 199901L
 132 /* C99 */
 133 #define INLINE inline
 134 
 135 #else
 136 /* something else; static inline is safest */
 137 #define INLINE static __UNUSED inline
 138 #endif
