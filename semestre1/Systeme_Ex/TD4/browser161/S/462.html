  76 /*
  77  * can_bg
  78  * just checks for an open slot.
  79  */
  80 static
  81 int
  82 can_bg(void)
  83 {
  84         int i;
  85 
  86         for (i = 0; i < MAXBG; i++) {
  87                 if (bgpids[i] == 0) {
  88                         return 1;
  89                 }
  90         }
  91 
  92         return 0;
  93 }

  95 /*
  96  * remember_bg
  97  * sticks the pid in an open slot in the background array.  note the assert --
  98  * better check can_bg before calling this.
  99  */
 100 static
 101 void
 102 remember_bg(pid_t pid)
 103 {
 104         int i;
 105         for (i = 0; i < MAXBG; i++) {
 106                 if (bgpids[i] == 0) {
 107                         bgpids[i] = pid;
 108                         return;
 109                 }
 110         }
 111         assert(0);
 112 }

 127 /*
 128  * readstatus
 129  * unpack results from wait
 130  */
 131 static
 132 void
 133 readstatus(int status, struct exitinfo *ei)
 134 {
 135         if (WIFEXITED(status)) {
 136                 ei->val = WEXITSTATUS(status);
 137                 ei->signaled = 0;
 138                 ei->stopped = 0;
 139                 ei->coredump = 0;
 140         }
 141         else if (WIFSIGNALED(status) && WCOREDUMP(status)) {
 142                 ei->val = WTERMSIG(status);
 143                 ei->signaled = 1;
 144                 ei->stopped = 0;
 145                 ei->coredump = 1;
 146         }
 147         else if (WIFSIGNALED(status)) {
 148                 ei->val = WTERMSIG(status);
 149                 ei->signaled = 1;
 150                 ei->stopped = 0;
 151                 ei->coredump = 0;
 152         }
 153         else if (WIFSTOPPED(status)) {
 154                 ei->val = WSTOPSIG(status);
 155                 ei->signaled = 0;
 156                 ei->stopped = 1;
 157                 ei->coredump = 0;
 158         }
 159         else {
 160                 printf("Invalid status code %d", status);
 161                 ei->val = status;
 162                 ei->signaled = 0;
 163                 ei->stopped = 0;
 164                 ei->coredump = 0;
 165         }
 166 
 167 }

 169 /*
 170  * printstatus
 171  * print results from wait
 172  */
 173 static
 174 void
 175 printstatus(const struct exitinfo *ei, int printexitzero)
 176 {
 177         if (ei->signaled && ei->coredump) {
 178                 printf("Signal %d (core dumped)\n", ei->val);
 179         }
 180         else if (ei->signaled) {
 181                 printf("Signal %d\n", ei->val);
 182         }
 183         else if (ei->stopped) {
 184                 printf("Stopped on signal %d\n", ei->val);
 185         }
 186         else if (printexitzero || ei->val != 0) {
 187                 printf("Exit %d\n", ei->val);
 188         }
 189 }

 191 /*
 192  * dowait
 193  * just does a waitpid.
 194  */
 195 static
 196 void
 197 dowait(pid_t pid)
 198 {
 199         struct exitinfo ei;
 200         int status;
 201 
 202         if (waitpid(pid, &status, 0) < 0) {
 203                 warn("pid %d", pid);
 204         }
 205         else {
 206                 printf("pid %d: ", pid);
 207                 readstatus(status, &ei);
 208                 printstatus(&ei, 1);
 209         }
 210 }

 218 int
 219 dowaitpoll(pid_t pid)
 220 {
 221         struct exitinfo ei;
 222         pid_t foundpid;
 223         int status;
 224 
 225         foundpid = waitpid(pid, &status, WNOHANG);
 226         if (foundpid < 0) {
 227                 warn("pid %d", pid);
 228         }

 230                 printf("pid %d: ", pid);
 231                 readstatus(status, &ei);
 232                 printstatus(&ei, 1);
 233                 return 1;
 234         }
 235         return 0;

 238 /*
 239  * waitpoll
 240  * poll all background jobs for having exited.
 241  */
 242 static
 243 void
 244 waitpoll(void)
 245 {
 246         int i;
 247         for (i=0; i < MAXBG; i++) {
 248                 if (bgpids[i] != 0) {
 249                         if (dowaitpoll(bgpids[i])) {
 250                                 bgpids[i] = 0;
 251                         }
 252                 }
 253         }
 254 }
 255 #endif /* WNOHANG */

 257 /*
 258  * wait
 259  * allows the user to "foreground" a process by waiting on it.  without ps to
 260  * know the pids, this is a little tough to use with an arg, but without an
 261  * arg it will wait for all the background jobs.
 262  */
 263 static
 264 void
 265 cmd_wait(int ac, char *av[], struct exitinfo *ei)
 266 {
 267         int i;
 268         pid_t pid;
 269 
 270         if (ac == 2) {
 271                 pid = atoi(av[1]);
 272                 dowait(pid);
 273                 for (i = 0; i < MAXBG; i++) {
 274                         if (bgpids[i]==pid) {
 275                                 bgpids[i] = 0;
 276                         }
 277                 }
 278                 exitinfo_exit(ei, 0);
 279                 return;
 280         }
 281         else if (ac == 1) {
 282                 for (i=0; i < MAXBG; i++) {
 283                         if (bgpids[i] != 0) {
 284                                 dowait(bgpids[i]);
 285                                 bgpids[i] = 0;
 286                         }
 287                 }
 288                 exitinfo_exit(ei, 0);
 289                 return;
 290         }
 291         printf("Usage: wait [pid]\n");
 292         exitinfo_exit(ei, 1);
 293 }

 358 /*
 359  * docommand
 360  * tokenizes the command line using strtok.  if there aren't any commands,
 361  * simply returns.  checks to see if it's a builtin, running it if it is.
 362  * otherwise, it's a standard command.  check for the '&', try to background
 363  * the job if possible, otherwise just run it and wait on it.
 364  */
 365 static
 366 void
 367 docommand(char *buf, struct exitinfo *ei)
 368 {
 369         char *args[NARG_MAX + 1];
 370         int nargs, i;
 371         char *s;
 372         pid_t pid;
 373         int status;
 374         int bg=0;
 375         time_t startsecs, endsecs;
 376         unsigned long startnsecs, endnsecs;
 377 
 378         nargs = 0;
 379         for (s = strtok(buf, " \t\r\n"); s; s = strtok(NULL, " \t\r\n")) {
 380                 if (nargs >= NARG_MAX) {
 381                         printf("%s: Too many arguments "
 382                                "(exceeds system limit)\n",
 383                                args[0]);
 384                         exitinfo_exit(ei, 1);
 385                         return;
 386                 }
 387                 args[nargs++] = s;
 388         }
 389         args[nargs] = NULL;
 390 
 391         if (nargs==0) {
 392                 /* empty line */
 393                 exitinfo_exit(ei, 0);
 394                 return;
 395         }
 396 
 397         for (i=0; builtins[i].name; i++) {
 398                 if (!strcmp(builtins[i].name, args[0])) {
 399                         builtins[i].func(nargs, args, ei);
 400                         return;
 401                 }
 402         }
 403 
 404         /* Not a builtin; run it */
 405 
 406         if (nargs > 0 && !strcmp(args[nargs-1], "&")) {
 407                 /* background */
 408                 if (!can_bg()) {
 409                         printf("%s: Too many background jobs; wait for "
 410                                "some to finish before starting more\n",
 411                                args[0]);
 412                         exitinfo_exit(ei, 1);
 413                         return;
 414                 }
 415                 nargs--;
 416                 args[nargs] = NULL;
 417                 bg = 1;
 418         }
 419 
 420         if (timing) {
 421                 __time(&startsecs, &startnsecs);
 422         }
 423 
 424         pid = fork();
 425         switch (pid) {
 426                 case -1:
 427                         /* error */
 428                         warn("fork");
 429                         exitinfo_exit(ei, 255);
 430                         return;
 431                 case 0:
 432                         /* child */
 433                         execvp(args[0], args);
 434                         warn("%s", args[0]);
 435                         /*
 436                          * Use _exit() instead of exit() in the child
 437                          * process to avoid calling atexit() functions,
 438                          * which would cause hostcompat (if present) to
 439                          * reset the tty state and mess up our input
 440                          * handling.
 441                          */
 442                         _exit(1);
 443                 default:
 444                         break;
 445         }
 446 
 447         /* parent */
 448         if (bg) {
 449                 /* background this command */
 450                 remember_bg(pid);
 451                 printf("[%d] %s ... &\n", pid, args[0]);
 452                 exitinfo_exit(ei, 0);
 453                 return;
 454         }
 455 
 456         if (waitpid(pid, &status, 0) < 0) {
 457                 warn("waitpid");
 458                 exitinfo_exit(ei, 255);
 459         }
 460         else {
 461                 readstatus(status, ei);
 462         }
 463 
 464         if (timing) {
 465                 __time(&endsecs, &endnsecs);
 466                 if (endnsecs < startnsecs) {
 467                         endnsecs += 1000000000;
 468                         endsecs--;
 469                 }
 470                 endnsecs -= startnsecs;
 471                 endsecs -= startsecs;
 472                 warnx("subprocess time: %lu.%09lu seconds",
 473                       (unsigned long) endsecs, (unsigned long) endnsecs);
 474         }
 475 }

 477 /*
 478  * getcmd
 479  * pulls valid characters off the console, filling the buffer.
 480  * backspace deletes a character, simply by moving the position back.
 481  * a newline or carriage return breaks the loop, which terminates
 482  * the string and returns.
 483  *
 484  * if there's an invalid character or a backspace when there's nothing
 485  * in the buffer, putchars an alert (bell).
 486  */
 487 static
 488 void
 489 getcmd(char *buf, size_t len)
 490 {
 491         size_t pos = 0;
 492         int done=0, ch;
 493 
 494         /*
 495          * In the absence of a <ctype.h>, assume input is 7-bit ASCII.
 496          */
 497 
 498         while (!done) {
 499                 ch = getchar();
 500                 if ((ch == '\b' || ch == 127) && pos > 0) {
 501                         putchar('\b');
 502                         putchar(' ');
 503                         putchar('\b');
 504                         pos--;
 505                 }
 506                 else if (ch == '\r' || ch == '\n') {
 507                         putchar('\r');
 508                         putchar('\n');
 509                         done = 1;
 510                 }
 511                 else if (ch >= 32 && ch < 127 && pos < len-1) {
 512                         buf[pos++] = ch;
 513                         putchar(ch);
 514                 }
 515                 else {
 516                         /* alert (bell) character */
 517                         putchar('\a');
 518                 }
 519         }
 520         buf[pos] = 0;
 521 }

 523 /*
 524  * interactive
 525  * runs the interactive shell.  basically, just infinitely loops, grabbing
 526  * commands and running them (and printing the exit status if it's not
 527  * success.)
 528  */
 529 static
 530 void
 531 interactive(void)
 532 {
 533         char buf[CMDLINE_MAX];
 534         struct exitinfo ei;
 535 
 536         while (1) {
 537                 printf("OS/161$ ");
 538                 getcmd(buf, sizeof(buf));
 539                 docommand(buf, &ei);
 540                 printstatus(&ei, 0);
 541 #ifdef WNOHANG
 542                 waitpoll();
 543 #endif
 544         }
 545 }

 547 static
 548 void
 549 check_timing(void)
 550 {
 551         time_t secs;
 552         unsigned long nsecs;
 553         if (__time(&secs, &nsecs) != -1) {
 554                 timing = 1;
 555                 warnx("Timing enabled.");
 556         }
 557 }

 559 /*
 560  * main
 561  * if there are no arguments, run interactively, otherwise, run a program
 562  * from within the shell, but immediately exit.
 563  */
 564 int
 565 main(int argc, char *argv[])
 566 {
 567 #ifdef HOST
 568         hostcompat_init(argc, argv);
 569 #endif
 570         check_timing();
 571 
 572         /*
 573          * Allow argc to be 0 in case we're running on a broken kernel,
 574          * or one that doesn't set argv when starting the first shell.
 575          */
 576         if (argc == 0 || argc == 1) {
 577                 interactive();
 578         }
 579         else if (argc == 3 && !strcmp(argv[1], "-c")) {
 580                 struct exitinfo ei;
 581                 docommand(argv[2], &ei);
 582                 printstatus(&ei, 0);
 583                 if (ei.signaled || ei.stopped || ei.val != 0) {
 584                         exit(1);
 585                 }
 586         }
 587         else {
 588                 errx(1, "Usage: sh [-c command]");
 589         }
 590         return 0;
 591 }
