  90 /*
  91  * Use this instead of just calling printf so we know each printout
  92  * is atomic; this prevents the lines from getting intermingled.
  93  */
  94 static
  95 void
  96 say(const char *fmt, ...)
  97 {
  98         char buf[256];
  99         va_list ap;
 100         va_start(ap, fmt);
 101         vsnprintf(buf, sizeof(buf), fmt, ap);
 102         va_end(ap);
 103         write(STDOUT_FILENO, buf, strlen(buf));
 104 }

 108 static
 109 void
 110 multiply(struct matrix *res, const struct matrix *m1, const struct matrix *m2)
 111 {
 112         int i, j, k;
 113 
 114         for (i=0; i<DIM; i++) {
 115                 for (j=0; j<DIM; j++) {
 116                         int val=0;
 117                         for (k=0; k<DIM; k++) {
 118                                 val += m1->m_data[i][k]*m2->m_data[k][j];
 119                         }
 120                         res->m_data[i][j] = val;
 121                 }
 122         }
 123 }
 124 
 125 static
 126 void
 127 addeq(struct matrix *m1, const struct matrix *m2)
 128 {
 129         int i, j;
 130         for (i=0; i<DIM; i++) {
 131                 for (j=0; j<DIM; j++) {
 132                         m1->m_data[i][j] += m2->m_data[i][j];
 133                 }
 134         }
 135 }
 136 
 137 static
 138 int
 139 trace(const struct matrix *m1)
 140 {
 141         int i, t=0;
 142         for (i=0; i<DIM; i++) {
 143                 t += m1->m_data[i][i];
 144         }
 145         return t;
 146 }

 152 static
 153 void
 154 populate_initial_matrixes(int mynum)
 155 {
 156         int i,j;
 157         struct matrix *m = &mats[0];
 158         for (i=0; i<DIM; i++) {
 159                 for (j=0; j<DIM; j++) {
 160                         m->m_data[i][j] = mynum+i-2*j;
 161                 }
 162         }
 163 
 164         multiply(&mats[1], &mats[0], &mats[0]);
 165 }
 166 
 167 static
 168 void
 169 compute(int n)
 170 {
 171         struct matrix tmp;
 172         int i, j;
 173 
 174         for (i=0,j=n-1; i<j; i++,j--) {
 175                 multiply(&tmp, &mats[i], &mats[j]);
 176                 addeq(&mats[n], &tmp);
 177         }
 178 }
 179 
 180 static
 181 void
 182 computeall(int mynum)
 183 {
 184         int i;
 185         populate_initial_matrixes(mynum);
 186         for (i=2; i<NMATS; i++) {
 187                 compute(i);
 188         }
 189 }
 190 
 191 static
 192 int
 193 answer(void)
 194 {
 195         return trace(&mats[NMATS-1]);
 196 }

 198 static
 199 void
 200 go(int mynum)
 201 {
 202         int r;
 203 
 204         say("Process %d (pid %d) starting computation...\n", mynum,
 205             (int) getpid());
 206 
 207         computeall(mynum);
 208         r = answer();
 209 
 210         if (r != right_answers[mynum]) {
 211                 say("Process %d answer %d: FAILED, should be %d\n",
 212                     mynum, r, right_answers[mynum]);
 213                 exit(1);
 214         }
 215         say("Process %d answer %d: passed\n", mynum, r);
 216         exit(0);
 217 }

 222 /*
 223  * We open the semaphore separately in each process to avoid
 224  * filehandle-level locking problems. If you can't be "reading" and
 225  * "writing" the semaphore concurrently because of the open file
 226  * object lock, then using the same file handle for P and V will
 227  * deadlock. Also, if this same lock is used to protect the reference
 228  * count on the open file logic, fork will block if another process is
 229  * using the same file handle for P, and then we're deadlocked too.
 230  *
 231  * Ideally the open file / filetable code wouldn't have this problem,
 232  * as it makes e.g. console output from background jobs behave
 233  * strangely, but it's a common issue in practice and it's better for
 234  * parallelvm to be immune to it.
 235  */
 236 
 237 struct usem {
 238         char name[32];
 239         int fd;
 240 };
 241 
 242 static
 243 void
 244 semcreate(const char *tag, struct usem *sem)
 245 {
 246         int fd;
 247 
 248         snprintf(sem->name, sizeof(sem->name), "sem:parallelvm.%s.%d",
 249                  tag, (int)getpid());
 250 
 251         fd = open(sem->name, O_WRONLY|O_CREAT|O_TRUNC, 0664);
 252         if (fd < 0) {
 253                 err(1, "%s: create", sem->name);
 254         }
 255         close(fd);
 256 }
 257 
 258 static
 259 void
 260 semopen(struct usem *sem)
 261 {
 262         sem->fd = open(sem->name, O_RDWR, 0664);
 263         if (sem->fd < 0) {
 264                 err(1, "%s: open", sem->name);
 265         }
 266 }
 267 
 268 static
 269 void
 270 semclose(struct usem *sem)
 271 {
 272         close(sem->fd);
 273 }
 274 
 275 static
 276 void
 277 semdestroy(struct usem *sem)
 278 {
 279         remove(sem->name);
 280 }

 282 static
 283 void
 284 semP(struct usem *sem, size_t num)
 285 {
 286         char c[num];
 287 
 288         if (read(sem->fd, c, num) < 0) {
 289                 err(1, "%s: read", sem->name);
 290         }
 291         (void)c;
 292 }
 293 
 294 static
 295 void
 296 semV(struct usem *sem, size_t num)
 297 {
 298         char c[num];
 299 
 300         /* semfs does not use these values, but be conservative */
 301         memset(c, 0, num);
 302 
 303         if (write(sem->fd, c, num) < 0) {
 304                 err(1, "%s: write", sem->name);
 305         }
 306 }

 311 static
 312 int
 313 status_is_failure(int status)
 314 {
 315         /* Proper interpretation of Unix exit status */
 316         if (WIFSIGNALED(status)) {
 317                 return 1;
 318         }
 319         if (!WIFEXITED(status)) {
 320                 /* ? */
 321                 return 1;
 322         }
 323         status = WEXITSTATUS(status);
 324         return status != 0;
 325 }

 327 static
 328 void
 329 makeprocs(bool dowait)
 330 {
 331         int i, status, failcount;
 332         struct usem s1, s2;
 333         pid_t pids[NJOBS];
 334 
 335         if (dowait) {
 336                 semcreate("1", &s1);
 337                 semcreate("2", &s2);
 338         }
 339 
 340         printf("Job size approximately %lu bytes\n", (unsigned long) JOBSIZE);
 341         printf("Forking %d jobs; total load %luk\n", NJOBS,
 342                (unsigned long) (NJOBS * JOBSIZE)/1024);
 343 
 344         for (i=0; i<NJOBS; i++) {
 345                 pids[i] = fork();
 346                 if (pids[i]<0) {
 347                         warn("fork (process %d)", i);
 348                         if (dowait) {
 349                                 semopen(&s1);
 350                                 semV(&s1, 1);
 351                                 semclose(&s1);
 352                         }
 353                 }
 354                 if (pids[i]==0) {
 355                         /* child */
 356                         if (dowait) {
 357                                 say("Process %d forked\n", i);
 358                                 semopen(&s1);
 359                                 semopen(&s2);
 360                                 semV(&s1, 1);
 361                                 semP(&s2, 1);
 362                                 semclose(&s1);
 363                                 semclose(&s2);
 364                         }
 365                         go(i);
 366                 }
 367         }
 368 
 369         if (dowait) {
 370                 semopen(&s1);
 371                 semopen(&s2);
 372                 say("Waiting for fork...\n");
 373                 semP(&s1, NJOBS);
 374                 say("Starting computation.\n");
 375                 semV(&s2, NJOBS);
 376         }
 377 
 378         failcount=0;
 379         for (i=0; i<NJOBS; i++) {
 380                 if (pids[i]<0) {
 381                         failcount++;
 382                 }
 383                 else {
 384                         if (waitpid(pids[i], &status, 0)<0) {
 385                                 err(1, "waitpid");
 386                         }
 387                         if (status_is_failure(status)) {
 388                                 failcount++;
 389                         }
 390                 }
 391         }
 392 
 393         if (failcount>0) {
 394                 printf("%d subprocesses failed\n", failcount);
 395                 exit(1);
 396         }
 397         printf("Test complete\n");
 398 
 399         semclose(&s1);
 400         semclose(&s2);
 401         semdestroy(&s1);
 402         semdestroy(&s2);
 403 }
