  54 /*
  55  * Create a dummy struct thread that we can put on lists for testing.
  56  */
  57 static
  58 struct thread *
  59 fakethread_create(const char *name)
  60 {
  61         struct thread *t;
  62 
  63         t = kmalloc(sizeof(*t));
  64         if (t == NULL) {
  65                 panic("threadlisttest: Out of memory\n");
  66         }
  67         /* ignore most of the fields, zero everything for tidiness */
  68         bzero(t, sizeof(*t));
  69         t->t_name = kstrdup(name);
  70         if (t->t_name == NULL) {
  71                 panic("threadlisttest: Out of memory\n");
  72         }
  73         t->t_stack = FAKE_MAGIC;
  74         threadlistnode_init(&t->t_listnode, t);
  75         return t;
  76 }

  78 /*
  79  * Destroy a fake thread.
  80  */
  81 static
  82 void
  83 fakethread_destroy(struct thread *t)
  84 {
  85         KASSERT(t->t_stack == FAKE_MAGIC);
  86         threadlistnode_cleanup(&t->t_listnode);
  87         kfree(t->t_name);
  88         kfree(t);
  89 }

  94 static
  95 void
  96 check_order(struct threadlist *tl, bool rev)
  97 {
  98         const char string0[] = "...";
  99         const char stringN[] = "~~~";
 100 
 101         struct thread *t;
 102         const char *first = rev ? stringN : string0;
 103         const char *last = rev ? string0 : stringN;
 104         const char *prev;
 105         int cmp;
 106 
 107         prev = first;
 108         THREADLIST_FORALL(t, *tl) {
 109                 cmp = strcmp(prev, t->t_name);
 110                 KASSERT(rev ? (cmp > 0) : (cmp < 0));
 111                 prev = t->t_name;
 112         }
 113         cmp = strcmp(prev, last);
 114         KASSERT(rev ? (cmp > 0) : (cmp < 0));
 115 }

 134 {
 135         struct threadlist tl;
 136         struct thread *t;

 138         threadlist_init(&tl);

 140         threadlist_addhead(&tl, fakethreads[0]);
 141         check_order(&tl, false);
 142         check_order(&tl, true);
 143         KASSERT(tl.tl_count == 1);

 231 static
 232 void
 233 threadlisttest_e(void)
 234 {
 235         struct threadlist tl;
 236         struct thread *t;
 237         unsigned i;
 238 
 239         threadlist_init(&tl);
 240 
 241         threadlist_addhead(&tl, fakethreads[1]);
 242         threadlist_addtail(&tl, fakethreads[3]);
 243         KASSERT(tl.tl_count == 2);
 244         check_order(&tl, false);
 245 
 246         threadlist_insertafter(&tl, fakethreads[3], fakethreads[4]);
 247         KASSERT(tl.tl_count == 3);
 248         check_order(&tl, false);
 249 
 250         threadlist_insertbefore(&tl, fakethreads[0], fakethreads[1]);
 251         KASSERT(tl.tl_count == 4);
 252         check_order(&tl, false);
 253 
 254         threadlist_insertafter(&tl, fakethreads[1], fakethreads[2]);
 255         KASSERT(tl.tl_count == 5);
 256         check_order(&tl, false);
 257 
 258         KASSERT(fakethreads[4]->t_listnode.tln_prev->tln_self ==
 259                 fakethreads[3]);
 260         KASSERT(fakethreads[3]->t_listnode.tln_prev->tln_self ==
 261                 fakethreads[2]);
 262         KASSERT(fakethreads[2]->t_listnode.tln_prev->tln_self ==
 263                 fakethreads[1]);
 264         KASSERT(fakethreads[1]->t_listnode.tln_prev->tln_self ==
 265                 fakethreads[0]);
 266 
 267         for (i=0; i<5; i++) {
 268                 t = threadlist_remhead(&tl);
 269                 KASSERT(t == fakethreads[i]);
 270         }
 271         KASSERT(tl.tl_count == 0);
 272 
 273         threadlist_cleanup(&tl);
 274 }

 276 static
 277 void
 278 threadlisttest_f(void)
 279 {
 280         struct threadlist tl;
 281         struct thread *t;
 282         unsigned i;
 283 
 284         threadlist_init(&tl);
 285 
 286         for (i=0; i<NUMNAMES; i++) {
 287                 threadlist_addtail(&tl, fakethreads[i]);
 288         }
 289         KASSERT(tl.tl_count == NUMNAMES);
 290 
 291         i=0;
 292         THREADLIST_FORALL(t, tl) {
 293                 KASSERT(t == fakethreads[i]);
 294                 i++;
 295         }
 296         KASSERT(i == NUMNAMES);
 297 
 298         i=0;
 299         THREADLIST_FORALL_REV(t, tl) {
 300                 KASSERT(t == fakethreads[NUMNAMES - i - 1]);
 301                 i++;
 302         }
 303         KASSERT(i == NUMNAMES);
 304 
 305         for (i=0; i<NUMNAMES; i++) {
 306                 t = threadlist_remhead(&tl);
 307                 KASSERT(t == fakethreads[i]);
 308         }
 309         KASSERT(tl.tl_count == 0);
 310 }

 316 threadlisttest(int nargs, char **args)
 317 {
 318         unsigned i;
 319 
 320         (void)nargs;
 321         (void)args;
 322 
 323         kprintf("Testing threadlists...\n");
 324 
 325         for (i=0; i<NUMNAMES; i++) {
 326                 fakethreads[i] = fakethread_create(names[i]);
 327         }
 328 
 329         threadlisttest_a();
 330         threadlisttest_b();
 331         threadlisttest_c();
 332         threadlisttest_d();
 333         threadlisttest_e();
 334         threadlisttest_f();
 335 
 336         for (i=0; i<NUMNAMES; i++) {
 337                 fakethread_destroy(fakethreads[i]);
 338                 fakethreads[i] = NULL;
 339         }
 340 
 341         kprintf("Done.\n");
 342         return 0;
