  86 semtestthread(void *junk, unsigned long num)
  87 {
  88         int i;
  89         (void)junk;
  90 
  91         /*
  92          * Only one of these should print at a time.

 112         kprintf("Starting semaphore test...\n");
 113         kprintf("If this hangs, it's broken: ");
 114         P(testsem);
 115         P(testsem);
 116         kprintf("ok\n");
 117 
 118         for (i=0; i<NTHREADS; i++) {
 119                 result = thread_fork("semtest", NULL, semtestthread, NULL, i);

 121                         panic("semtest: thread_fork failed: %s\n",
 122                               strerror(result));

 124         }
 125 
 126         for (i=0; i<NTHREADS; i++) {
 127                 V(testsem);
 128                 P(donesem);
 129         }
 130 
 131         /* so we can run it again */
 132         V(testsem);
 133         V(testsem);

 152 static
 153 void
 154 locktestthread(void *junk, unsigned long num)
 155 {
 156         int i;
 157         (void)junk;
 158 
 159         for (i=0; i<NLOCKLOOPS; i++) {
 160                 lock_acquire(testlock);
 161                 testval1 = num;
 162                 testval2 = num*num;
 163                 testval3 = num%3;
 164 
 165                 if (testval2 != testval1*testval1) {
 166                         fail(num, "testval2/testval1");
 167                 }
 168 
 169                 if (testval2%3 != (testval3*testval3)%3) {
 170                         fail(num, "testval2/testval3");
 171                 }
 172 
 173                 if (testval3 != testval1%3) {
 174                         fail(num, "testval3/testval1");
 175                 }
 176 
 177                 if (testval1 != num) {
 178                         fail(num, "testval1/num");
 179                 }
 180 
 181                 if (testval2 != num*num) {
 182                         fail(num, "testval2/num");
 183                 }
 184 
 185                 if (testval3 != num%3) {
 186                         fail(num, "testval3/num");
 187                 }
 188 
 189                 lock_release(testlock);
 190         }
 191         V(donesem);
 192 }

 223 static
 224 void
 225 cvtestthread(void *junk, unsigned long num)
 226 {
 227         int i;
 228         volatile int j;
 229         struct timespec ts1, ts2;
 230 
 231         (void)junk;
 232 
 233         for (i=0; i<NCVLOOPS; i++) {
 234                 lock_acquire(testlock);
 235                 while (testval1 != num) {
 236                         gettime(&ts1);
 237                         cv_wait(testcv, testlock);
 238                         gettime(&ts2);
 239 
 240                         /* ts2 -= ts1 */
 241                         timespec_sub(&ts2, &ts1, &ts2);
 242 
 243                         /* Require at least 2000 cpu cycles (we're 25mhz) */
 244                         if (ts2.tv_sec == 0 && ts2.tv_nsec < 40*2000) {
 245                                 kprintf("cv_wait took only %u ns\n",
 246                                         ts2.tv_nsec);
 247                                 kprintf("That's too fast... you must be "
 248                                         "busy-looping\n");
 249                                 V(donesem);
 250                                 thread_exit();
 251                         }
 252 
 253                 }
 254                 kprintf("Thread %lu\n", num);
 255                 testval1 = (testval1 + NTHREADS - 1)%NTHREADS;
 256 
 257                 /*
 258                  * loop a little while to make sure we can measure the
 259                  * time waiting on the cv.
 260                  */
 261                 for (j=0; j<3000; j++);
 262 
 263                 cv_broadcast(testcv, testlock);
 264                 lock_release(testlock);
 265         }
 266         V(donesem);
 267 }

 302 /*
 303  * Try to find out if going to sleep is really atomic.
 304  *
 305  * What we'll do is rotate through NCVS lock/CV pairs, with one thread
 306  * sleeping and the other waking it up. If we miss a wakeup, the sleep
 307  * thread won't go around enough times.
 308  */
 309 
 310 #define NCVS 250
 311 #define NLOOPS 40
 312 static struct cv *testcvs[NCVS];
 313 static struct lock *testlocks[NCVS];
 314 static struct semaphore *gatesem;
 315 static struct semaphore *exitsem;
 316 
 317 static
 318 void
 319 sleepthread(void *junk1, unsigned long junk2)
 320 {
 321         unsigned i, j;
 322 
 323         (void)junk1;
 324         (void)junk2;
 325 
 326         for (j=0; j<NLOOPS; j++) {
 327                 for (i=0; i<NCVS; i++) {
 328                         lock_acquire(testlocks[i]);
 329                         V(gatesem);
 330                         cv_wait(testcvs[i], testlocks[i]);
 331                         lock_release(testlocks[i]);
 332                 }
 333                 kprintf("sleepthread: %u\n", j);
 334         }
 335         V(exitsem);
 336 }

 338 static
 339 void
 340 wakethread(void *junk1, unsigned long junk2)
 341 {
 342         unsigned i, j;
 343 
 344         (void)junk1;
 345         (void)junk2;
 346 
 347         for (j=0; j<NLOOPS; j++) {
 348                 for (i=0; i<NCVS; i++) {
 349                         P(gatesem);
 350                         lock_acquire(testlocks[i]);
 351                         cv_signal(testcvs[i], testlocks[i]);
 352                         lock_release(testlocks[i]);
 353                 }
 354                 kprintf("wakethread: %u\n", j);
 355         }
 356         V(exitsem);
 357 }

 359 int
 360 cvtest2(int nargs, char **args)
 361 {
 362         unsigned i;
 363         int result;
 364 
 365         (void)nargs;
 366         (void)args;
 367 
 368         for (i=0; i<NCVS; i++) {
 369                 testlocks[i] = lock_create("cvtest2 lock");
 370                 testcvs[i] = cv_create("cvtest2 cv");
 371         }
 372         gatesem = sem_create("gatesem", 0);
 373         exitsem = sem_create("exitsem", 0);
 374 
 375         kprintf("cvtest2...\n");
 376 
 377         result = thread_fork("cvtest2", NULL, sleepthread, NULL, 0);
 378         if (result) {
 379                 panic("cvtest2: thread_fork failed\n");
 380         }
 381         result = thread_fork("cvtest2", NULL, wakethread, NULL, 0);
 382         if (result) {
 383                 panic("cvtest2: thread_fork failed\n");
 384         }
 385 
 386         P(exitsem);
 387         P(exitsem);
 388 
 389         sem_destroy(exitsem);
 390         sem_destroy(gatesem);
 391         exitsem = gatesem = NULL;
 392         for (i=0; i<NCVS; i++) {
 393                 lock_destroy(testlocks[i]);
 394                 cv_destroy(testcvs[i]);
 395                 testlocks[i] = NULL;
 396                 testcvs[i] = NULL;
 397         }
 398 
 399         kprintf("cvtest2 done\n");
 400         return 0;
 401 }
