  30 #ifndef _ERR_H_

  33 #include <sys/cdefs.h>   /* for __DEAD */
  34 #include <kern/types.h>  /* for __va_list */

  36 /*
  37  * 4.4BSD error-printing functions.
  38  *
  39  * These print the program name and the supplied message, and
  40  * (non-*x versions only) the string for the error currently
  41  * stored in "errno", and a newline. The err* versions then
  42  * exit with the supplied exitcode.
  43  *
  44  * The v* versions are to the non-v* versions like vprintf is to
  45  * printf.
  46  */
  47 
  48 void warn(const char *fmt, ...);
  49 void warnx(const char *fmt, ...);
  50 __DEAD void err(int exitcode, const char *fmt, ...);
  51 __DEAD void errx(int exitcode, const char *fmt, ...);
  52 
  53 void vwarn(const char *fmt, __va_list);
  54 void vwarnx(const char *fmt, __va_list);
  55 __DEAD void verr(int exitcode, const char *fmt, __va_list);
  56 __DEAD void verrx(int exitcode, const char *fmt, __va_list);
  57 
  58 #endif /* _ERR_H_ */
