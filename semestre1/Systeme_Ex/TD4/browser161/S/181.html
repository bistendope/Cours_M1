  40 /*
  41  * Machine-independent interface to interrupt enable/disable.
  42  *
  43  * "spl" stands for "set priority level", and was originally the name of
  44  * a VAX assembler instruction.
  45  *
  46  * The idea is that one can block less important interrupts while
  47  * processing them, but still allow more urgent interrupts to interrupt
  48  * that processing.
  49  *
  50  * Ordinarily there would be a whole bunch of defined interrupt
  51  * priority levels and functions for setting them - spltty(),
  52  * splbio(), etc., etc. But we don't support interrupt priorities in
  53  * OS/161, so there are only three:
  54  *
  55  *      spl0()       sets IPL to 0, enabling all interrupts.
  56  *      splhigh()    sets IPL to the highest value, disabling all interrupts.
  57  *      splx(s)      sets IPL to S, enabling whatever state S represents.
  58  *
  59  * All three return the old interrupt state. Thus, these are commonly used
  60  * as follows:
  61  *
  62  *      int s = splhigh();
  63  *      [ code ]
  64  *      splx(s);
  65  *
  66  * Note that these functions only affect interrupts on the current
  67  * processor.
  68  */
  69 
  70 SPL_INLINE int spl0(void);
  71 SPL_INLINE int splhigh(void);
  72 int splx(int);
  73 
  74 /*
  75  * Integer interrupt priority levels.
  76  */
  77 #define IPL_NONE   0
  78 #define IPL_HIGH   1
  79 
  80 /*
  81  * Lower-level functions for explicitly raising and lowering
  82  * particular interrupt levels. These are used by splx() and by the
  83  * spinlock code.
  84  *
  85  * A previous setting of OLDIPL is cancelled and replaced with NEWIPL.
  86  *
  87  * For splraise, NEWIPL > OLDIPL, and for spllower, NEWIPL < OLDIPL.
  88  */
  89 void splraise(int oldipl, int newipl);
  90 void spllower(int oldipl, int newipl);
