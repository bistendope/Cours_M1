  33 /*
  34  * On the mips there's only one memory barrier instruction, so these
  35  * are all the same. This is not true on many other CPUs (x86, arm,
  36  * sparc, powerpc, etc.) We also mark the instruction as a compiler-
  37  * level barrier by telling gcc that it destroys memory; this prevents
  38  * gcc from reordering loads and stores around it.
  39  *
  40  * See include/membar.h for further information.
  41  */
  42 
  43 MEMBAR_INLINE
  44 void
  45 membar_any_any(void)
  46 {
  47         __asm volatile(
  48                 ".set push;"            /* save assembler mode */
  49                 ".set mips32;"          /* allow MIPS32 instructions */
  50                 "sync;"                 /* do it */
  51                 ".set pop"              /* restore assembler mode */
  52                 :                       /* no outputs */
  53                 :                       /* no inputs */
  54                 : "memory");            /* "changes" memory */
  55 }
  56 
  57 MEMBAR_INLINE void membar_load_load(void) { membar_any_any(); }
  58 MEMBAR_INLINE void membar_store_store(void) { membar_any_any(); }
  59 MEMBAR_INLINE void membar_store_any(void) { membar_any_any(); }
  60 MEMBAR_INLINE void membar_any_store(void) { membar_any_any(); }
