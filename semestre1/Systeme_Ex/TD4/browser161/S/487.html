  33 /*
  34  * Get struct stat and all the #defines from the kernel
  35  */
  36 #include <kern/stat.h>
  37 #include <kern/stattypes.h>

  39 /*
  40  * Test macros for object types.
  41  */
  42 #define S_ISDIR(mode) ((mode & _S_IFMT) == _S_IFDIR)
  43 #define S_ISREG(mode) ((mode & _S_IFMT) == _S_IFREG)
  44 #define S_ISDIR(mode) ((mode & _S_IFMT) == _S_IFDIR)
  45 #define S_ISLNK(mode) ((mode & _S_IFMT) == _S_IFLNK)
  46 #define S_ISIFO(mode) ((mode & _S_IFMT) == _S_IFIFO)
  47 #define S_ISSOCK(mode) ((mode & _S_IFMT) ==_S_IFSOCK)
  48 #define S_ISCHR(mode) ((mode & _S_IFMT) == _S_IFCHR)
  49 #define S_ISBLK(mode) ((mode & _S_IFMT) == _S_IFBLK)
  50 
  51 /*
  52  * Provide non-underscore names. These are not actually standard; for
  53  * some reason only the test macros are.
  54  */
  55 #define S_IFMT   _S_IFMT
  56 #define S_IFREG  _S_IFREG
  57 #define S_IFDIR  _S_IFDIR
  58 #define S_IFLNK  _S_IFLNK
  59 #define S_IFIFO  _S_IFIFO
  60 #define S_IFSOCK _S_IFSOCK
  61 #define S_IFCHR  _S_IFCHR
  62 #define S_IFBLK  _S_IFBLK
  63 
  64 /*
  65  * stat is the same as fstat, only on a file that isn't already
  66  * open. lstat is the same as stat, only if the name passed names a
  67  * symlink, information about the symlink is returned rather than
  68  * information about the file it points to. You don't need to
  69  * implement lstat unless you're implementing symbolic links.
  70  */
  71 int fstat(int filehandle, struct stat *buf);
  72 int stat(const char *path, struct stat *buf);
  73 int lstat(const char *path, struct stat *buf);
  74 
  75 /*
  76  * The second argument to mkdir is the mode for the new directory.
  77  * Unless you're implementing security and permissions, you can
  78  * (and should) ignore it. See notes in unistd.h.
  79  */
  80 int mkdir(const char *dirname, int ignore);
