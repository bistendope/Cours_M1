  30 #ifndef _MAINBUS_H_
  31 #define _MAINBUS_H_
  32 
  33 /*
  34  * Abstract system bus interface.
  35  */
  36 
  37 
  38 struct cpu;       /* from <cpu.h> */
  39 struct trapframe; /* from <machine/trapframe.h> */
  40 
  41 
  42 /* Initialize the system bus and probe and attach hardware devices. */
  43 void mainbus_bootstrap(void);
  44 
  45 /* Start up secondary CPUs, once their cpu structures are set up */
  46 void mainbus_start_cpus(void);
  47 
  48 /* Bus-level interrupt handler, called from cpu-level trap/interrupt code */
  49 void mainbus_interrupt(struct trapframe *);
  50 
  51 /* Find the size of main memory. */
  52 /* XXX this interface is not adequately MI */
  53 size_t mainbus_ramsize(void);
  54 
  55 /* Switch on an inter-processor interrupt. (Low-level.) */
  56 void mainbus_send_ipi(struct cpu *target);
  57 
  58 /* Request breaking into the debugger, where available. */
  59 void mainbus_debugger(void);
  60 
  61 /*
  62  * The various ways to shut down the system. (These are very low-level
  63  * and should generally not be called directly - md_poweroff, for
  64  * instance, unceremoniously turns the power off without doing
  65  * anything else.)
  66  */
  67 void mainbus_halt(void);
  68 void mainbus_poweroff(void);
  69 void mainbus_reboot(void);
  70 void mainbus_panic(void);
  71 
  72 
  73 #endif /* _MAINBUS_H_ */
