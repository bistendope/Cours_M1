  61 /*
  62  * Workload sizing.
  63  *
  64  * We fork numprocs processes. Each one works on WORKNUM integers at a
  65  * time, so the total VM load is WORKNUM * sizeof(int) * numprocs. For
  66  * the best stress testing, this should be substantially larger than
  67  * your available RAM.
  68  *
  69  * Meanwhile we generate and process numkeys integers, so the total
  70  * filesystem load is numkeys * sizeof(int). For the best stress
  71  * testing this should be substantially larger than your buffer cache.
  72  *
  73  * This test is supposed to adjust to arbitrary settings of WORKNUM,
  74  * numprocs, and numkeys. For a small test try these settings:
  75  *      WORKNUM         (16*1024)
  76  *      numprocs        4, or 2
  77  *      numkeys         10000
  78  *
  79  * This should fit in memory (memory footprint is 256K) and generate
  80  * only small files (~40K) which will work on the pre-largefiles SFS.
  81  *
  82  * The default settings are:
  83  *      WORKNUM         (96*1024)
  84  *      numprocs        4
  85  *      numkeys         128*1024
  86  *
  87  * so the memory footprint is 1.5M (comparable to the VM tests) and
  88  * the base file size is 512K. Note that because there are a lot of
  89  * temporary files, it pumps a good deal more than 512K of data. For
  90  * the record, the solution set takes about 7.5 minutes of virtual
  91  * time to run it, in either 1M or 2M of RAM and with 4 CPUs; this
  92  * currently corresponds to about 13-14 minutes of real time.
  93  *
  94  * Note that the parent psort process serves as a director and doesn't
  95  * itself compute; it has a workspace (because it's a static buffer)
  96  * but doesn't use it. A VM system that doesn't do zerofill
  97  * optimization will be a lot slower because it has to copy this space
  98  * for every batch of forks.
  99  *
 100  * Also note that you can set numprocs and numkeys on the command
 101  * line, but not WORKNUM.
 102  *
 103  * FUTURE: maybe make a build option to malloc the work space instead
 104  * of using a static buffer, which would allow choosing WORKNUM on the
 105  * command line too, at the cost of depending on malloc working.
 106  */

 130 static
 131 void
 132 sortints(int *v, int num)
 133 {
 134         int pivotval, pivotpoint, pivotcount;
 135         int frontpos, readpos, endpos, i, j;
 136         int tmp;
 137 
 138         if (num < 2) {
 139                 return;
 140         }
 141 
 142         pivotpoint = num/2;
 143         pivotval = v[pivotpoint];
 144         pivotcount = 0;
 145 
 146         frontpos = 0;
 147         readpos = 0;
 148         endpos = num;
 149         while (readpos < endpos) {
 150                 if (v[readpos] < pivotval) {
 151                         v[frontpos++] = v[readpos++];
 152                 }
 153                 else if (v[readpos] == pivotval) {
 154                         readpos++;
 155                         pivotcount++;
 156                 }
 157                 else {
 158                         tmp = v[--endpos];
 159                         v[endpos] = v[readpos];
 160                         v[readpos] = tmp;
 161                 }
 162         }
 163         assert(readpos == endpos);
 164         assert(frontpos + pivotcount == readpos);
 165 
 166         for (i=frontpos; i<endpos; i++) {
 167                 v[i] = pivotval;
 168         }
 169 
 170         for (i=endpos, j=num-1; i<j; i++,j--) {
 171                 tmp = v[i];
 172                 v[i] = v[j];
 173                 v[j] = tmp;
 174         }
 175 
 176         sortints(v, frontpos);
 177         sortints(&v[endpos], num-endpos);
 178 }

 325 static
 326 off_t
 327 getsize(const char *path)
 328 {
 329         struct stat buf;
 330         int fd;
 331         static int no_stat, no_fstat;
 332 
 333         if (!no_stat) {
 334                 if (stat(path, &buf) == 0) {
 335                         return buf.st_size;
 336                 }
 337                 if (errno != ENOSYS) {
 338                         complain("%s: stat", path);
 339                         exit(1);
 340                 }
 341                 /* Avoid further "Unknown syscall 81" messages */
 342                 no_stat = 1;
 343         }
 344 
 345         fd = doopen(path, O_RDONLY, 0);
 346         if (!no_fstat) {
 347                 if (fstat(fd, &buf) == 0) {
 348                         close(fd);
 349                         return buf.st_size;
 350                 }
 351                 if (errno != ENOSYS) {
 352                         complain("%s: stat", path);
 353                         exit(1);
 354                 }
 355                 /* Avoid further "Unknown syscall 82" messages */
 356                 no_fstat = 1;
 357         }
 358 
 359         /* otherwise, lseek */
 360         if (lseek(fd, 0, SEEK_END) >= 0) {
 361                 buf.st_size = lseek(fd, 0, SEEK_CUR);
 362                 if (buf.st_size >= 0) {
 363                         return buf.st_size;
 364                 }
 365         }
 366         complain("%s: getting file size with lseek", path);
 367         close(fd);
 368         exit(1);
 369 }

 464 static
 465 int
 466 dowait(int guy, pid_t pid)
 467 {
 468         int status, result;
 469 
 470         result = waitpid(pid, &status, 0);
 471         if (result < 0) {
 472                 complain("waitpid");
 473                 return -1;
 474         }
 475         if (WIFSIGNALED(status)) {
 476                 complainx("proc %d: signal %d", guy, WTERMSIG(status));
 477                 return -1;
 478         }
 479         assert(WIFEXITED(status));
 480         status = WEXITSTATUS(status);
 481         if (status) {
 482                 complainx("proc %d: exit %d", guy, status);
 483                 return -1;
 484         }
 485         return 0;
 486 }

 488 static
 489 void
 490 doforkall(const char *phasename, void (*func)(void))
 491 {
 492         int i, bad = 0;
 493         pid_t pids[numprocs];
 494 
 495         for (i=0; i<numprocs; i++) {
 496                 pids[i] = dofork();
 497                 if (pids[i] < 0) {
 498                         bad = 1;
 499                 }
 500                 else if (pids[i] == 0) {
 501                         /* child */
 502                         me = i;
 503                         func();
 504                         exit(0);
 505                 }
 506         }
 507 
 508         for (i=0; i<numprocs; i++) {
 509                 if (pids[i] > 0 && dowait(i, pids[i])) {
 510                         bad = 1;
 511                 }
 512         }
 513 
 514         if (bad) {
 515                 complainx("%s failed.", phasename);
 516                 exit(1);
 517         }
 518 }

 537 {
 538         int keys_per, myfirst, mykeys;
 539

 541         myfirst = me*keys_per;

 575 static
 576 void
 577 genkeys_sub(void)
 578 {
 579         int fd, i, mykeys, keys_done, keys_to_do, value;
 580 
 581         fd = doopen(PATH_KEYS, O_WRONLY, 0);
 582 
 583         mykeys = getmykeys();
 584         seekmyplace(PATH_KEYS, fd);
 585 
 586         srandom(seeds[me]);
 587         keys_done = 0;
 588         while (keys_done < mykeys) {
 589                 keys_to_do = mykeys - keys_done;
 590                 if (keys_to_do > WORKNUM) {
 591                         keys_to_do = WORKNUM;
 592                 }
 593 
 594                 for (i=0; i<keys_to_do; i++) {
 595                         value = random();
 596 
 597                         // check bounds of value
 598                         assert(value >= 0);
 599                         assert(value <= RANDOM_MAX);
 600 
 601                         // do not allow the value to be zero or RANDOM_MAX
 602                         while (value == 0 || value == RANDOM_MAX) {
 603                                 value = random();
 604                         }
 605 
 606                         workspace[i] = value;
 607                 }
 608 
 609                 dowrite(PATH_KEYS, fd, workspace, keys_to_do*sizeof(int));
 610                 keys_done += keys_to_do;
 611         }
 612 
 613         doclose(PATH_KEYS, fd);
 614 }

 616 static
 617 void
 618 genkeys(void)
 619 {
 620         long seedspace[numprocs];
 621         int i;
 622 
 623         /* Create the file. */
 624         docreate(PATH_KEYS);
 625 
 626         /* Generate random seeds for each subprocess. */
 627         srandom(randomseed);
 628         for (i=0; i<numprocs; i++) {
 629                 seedspace[i] = random();
 630         }
 631 
 632         /* Do it. */
 633         complainx("Generating %d integers using %d procs", numkeys, numprocs);
 634         seeds = seedspace;
 635         doforkall("Initialization", genkeys_sub);
 636         seeds = NULL;
 637 
 638         /* Cross-check the size of the output. */
 639         if (getsize(PATH_KEYS) != correctsize) {
 640                 complainx("%s: file is wrong size", PATH_KEYS);
 641                 exit(1);
 642         }
 643 
 644         /* Checksum the output. */
 645         complainx("Checksumming the data (using one proc)");
 646         checksum = checksum_file(PATH_KEYS);
 647         complainx("Checksum of unsorted keys: %ld", checksum);
 648 }

 670 static
 671 void
 672 bin(void)
 673 {
 674         int infd, outfds[numprocs];
 675         const char *name;
 676         int i, mykeys, keys_done, keys_to_do;
 677         int key, pivot, binnum;
 678 
 679         infd = doopen(PATH_KEYS, O_RDONLY, 0);
 680 
 681         mykeys = getmykeys();
 682         seekmyplace(PATH_KEYS, infd);
 683 
 684         for (i=0; i<numprocs; i++) {
 685                 name = binname(me, i);
 686                 outfds[i] = doopen(name, O_WRONLY|O_CREAT|O_TRUNC, 0664);
 687         }
 688 
 689         pivot = (RANDOM_MAX / numprocs);
 690 
 691         keys_done = 0;
 692         while (keys_done < mykeys) {
 693                 keys_to_do = mykeys - keys_done;
 694                 if (keys_to_do > WORKNUM) {
 695                         keys_to_do = WORKNUM;
 696                 }
 697 
 698                 doexactread(PATH_KEYS, infd, workspace,
 699                             keys_to_do * sizeof(int));
 700 
 701                 for (i=0; i<keys_to_do; i++) {
 702                         key = workspace[i];
 703 
 704                         binnum = key / pivot;
 705                         if (key <= 0) {
 706                                 complainx("proc %d: garbage key %d", me, key);
 707                                 key = 0;
 708                         }
 709                         assert(binnum >= 0);
 710                         assert(binnum < numprocs);
 711                         dowrite("bin", outfds[binnum], &key, sizeof(key));
 712                 }
 713 
 714                 keys_done += keys_to_do;
 715         }
 716         doclose(PATH_KEYS, infd);
 717 
 718         for (i=0; i<numprocs; i++) {
 719                 doclose(binname(me, i), outfds[i]);
 720         }
 721 }

 723 static
 724 void
 725 sortbins(void)
 726 {
 727         const char *name;
 728         int i, fd;
 729         off_t binsize;
 730 
 731         for (i=0; i<numprocs; i++) {
 732                 name = binname(me, i);
 733                 binsize = getsize(name);
 734                 if (binsize % sizeof(int) != 0) {
 735                         complainx("%s: bin size %ld no good", name,
 736                                   (long) binsize);
 737                         exit(1);
 738                 }
 739                 if (binsize > (off_t) sizeof(workspace)) {
 740                         complainx("proc %d: %s: bin too large", me, name);
 741                         exit(1);
 742                 }
 743 
 744                 fd = doopen(name, O_RDWR, 0);
 745                 doexactread(name, fd, workspace, binsize);
 746 
 747                 sortints(workspace, binsize/sizeof(int));
 748 
 749                 dolseek(name, fd, 0, SEEK_SET);
 750                 dowrite(name, fd, workspace, binsize);
 751                 doclose(name, fd);
 752         }
 753 }

 755 static
 756 void
 757 mergebins(void)
 758 {
 759         int infds[numprocs], outfd;
 760         int values[numprocs], ready[numprocs];
 761         const char *name, *outname;
 762         int i, result;
 763         int numready, place, val, worknum;
 764 
 765         outname = mergedname(me);
 766         outfd = doopen(outname, O_WRONLY|O_CREAT|O_TRUNC, 0664);
 767 
 768         for (i=0; i<numprocs; i++) {
 769                 name = binname(i, me);
 770                 infds[i] = doopen(name, O_RDONLY, 0);
 771                 values[i] = 0;
 772                 ready[i] = 0;
 773         }
 774 
 775         worknum = 0;
 776 
 777         while (1) {
 778                 numready = 0;
 779                 for (i=0; i<numprocs; i++) {
 780                         if (infds[i] < 0) {
 781                                 continue;
 782                         }
 783 
 784                         if (!ready[i]) {
 785                                 result = doread("bin", infds[i],
 786                                                 &val, sizeof(int));
 787                                 if (result == 0) {
 788                                         doclose("bin", infds[i]);
 789                                         infds[i] = -1;
 790                                         continue;
 791                                 }
 792                                 if ((size_t) result != sizeof(int)) {
 793                                         complainx("%s: read: short count",
 794                                                   binname(i, me));
 795                                         exit(1);
 796                                 }
 797                                 values[i] = val;
 798                                 ready[i] = 1;
 799                         }
 800                         numready++;
 801                 }
 802                 if (numready == 0) {
 803                         break;
 804                 }
 805 
 806                 /* find the smallest */
 807                 place = -1;
 808                 for (i=0; i<numprocs; i++) {
 809                         if (!ready[i]) {
 810                                 continue;
 811                         }
 812                         if (place < 0 || values[i] < val) {
 813                                 val = values[i];
 814                                 place = i;
 815                         }
 816                 }
 817                 assert(place >= 0);
 818 
 819                 workspace[worknum++] = val;
 820                 if (worknum >= WORKNUM) {
 821                         assert(worknum == WORKNUM);
 822                         dowrite(outname, outfd, workspace,
 823                                 worknum * sizeof(int));
 824                         worknum = 0;
 825                 }
 826                 ready[place] = 0;
 827         }
 828 
 829         dowrite(outname, outfd, workspace, worknum * sizeof(int));
 830         doclose(outname, outfd);
 831 
 832         for (i=0; i<numprocs; i++) {
 833                 assert(infds[i] < 0);
 834         }
 835 }

 906 static
 907 void
 908 sort(void)
 909 {
 910         unsigned long sortedsum;
 911         int i, j;
 912 
 913         /* Step 1. Toss into bins. */
 914         complainx("Tossing into %d bins using %d procs",
 915                   numprocs*numprocs, numprocs);
 916         doforkall("Tossing", bin);
 917         checksize_bins();
 918         complainx("Done tossing into bins.");
 919 
 920         /* Step 2: Sort the bins. */
 921         complainx("Sorting %d bins using %d procs",
 922                   numprocs*numprocs, numprocs);
 923         doforkall("Sorting", sortbins);
 924         checksize_bins();
 925         complainx("Done sorting the bins.");
 926 
 927         /* Step 3: Merge corresponding bins. */
 928         complainx("Merging %d bins using %d procs",
 929                   numprocs*numprocs, numprocs);
 930         doforkall("Merging", mergebins);
 931         checksize_merge();
 932         complainx("Done merging the bins.");
 933 
 934         /* Step 3a: delete the bins */
 935         for (i=0; i<numprocs; i++) {
 936                 for (j=0; j<numprocs; j++) {
 937                         doremove(binname(i, j));
 938                 }
 939         }
 940 
 941         /* Step 4: assemble output file */
 942         complainx("Assembling output file using %d procs", numprocs);
 943         docreate(PATH_SORTED);
 944         doforkall("Final assembly", assemble);
 945         if (getsize(PATH_SORTED) != correctsize) {
 946                 complainx("%s: file is wrong size", PATH_SORTED);
 947                 exit(1);
 948         }
 949 
 950         /* Step 4a: delete the merged bins */
 951         for (i=0; i<numprocs; i++) {
 952                 doremove(mergedname(i));
 953         }
 954 
 955         /* Step 5: Checksum the result. */
 956         complainx("Checksumming the output (using one proc)");
 957         sortedsum = checksum_file(PATH_SORTED);
 958         complainx("Checksum of sorted keys: %ld", sortedsum);
 959 
 960         if (sortedsum != checksum) {
 961                 complainx("Sums do not match");
 962                 exit(1);
 963         }
 964 }

 998 static
 999 void
1000 dovalidate(void)
1001 {
1002         const char *name;
1003         int fd, i, mykeys, keys_done, keys_to_do;
1004         int key, smallest, largest;
1005 
1006         name = PATH_SORTED;
1007         fd = doopen(name, O_RDONLY, 0);
1008 
1009         mykeys = getmykeys();
1010         seekmyplace(name, fd);
1011 
1012         smallest = RANDOM_MAX;
1013         largest = 0;
1014 
1015         keys_done = 0;
1016         while (keys_done < mykeys) {
1017                 keys_to_do = mykeys - keys_done;
1018                 if (keys_to_do > WORKNUM) {
1019                         keys_to_do = WORKNUM;
1020                 }
1021 
1022                 doexactread(name, fd, workspace, keys_to_do * sizeof(int));
1023 
1024                 for (i=0; i<keys_to_do; i++) {
1025                         key = workspace[i];
1026 
1027                         if (key < 0) {
1028                                 complain("%s: found negative key", name);
1029                                 exit(1);
1030                         }
1031                         if (key == 0) {
1032                                 complain("%s: found zero key", name);
1033                                 exit(1);
1034                         }
1035                         if (key >= RANDOM_MAX) {
1036                                 complain("%s: found too-large key", name);
1037                                 exit(1);
1038                         }
1039 
1040                         if (key < smallest) {
1041                                 smallest = key;
1042                         }
1043                         if (key > largest) {
1044                                 largest = key;
1045                         }
1046                 }
1047 
1048                 keys_done += keys_to_do;
1049         }
1050         doclose(name, fd);
1051 
1052         name = validname(me);
1053         fd = doopen(name, O_WRONLY|O_CREAT|O_TRUNC, 0664);
1054         dowrite(name, fd, &smallest, sizeof(smallest));
1055         dowrite(name, fd, &largest, sizeof(largest));
1056         doclose(name, fd);
1057 }

1059 static
1060 void
1061 validate(void)
1062 {
1063         int smallest, largest, prev_largest;
1064         int i, fd;
1065         const char *name;
1066 
1067         complainx("Validating the sorted data using %d procs", numprocs);
1068         doforkall("Validation", dovalidate);
1069         checksize_valid();
1070 
1071         prev_largest = 1;
1072 
1073         for (i=0; i<numprocs; i++) {
1074                 name = validname(i);
1075                 fd = doopen(name, O_RDONLY, 0);
1076 
1077                 doexactread(name, fd, &smallest, sizeof(int));
1078                 doexactread(name, fd, &largest, sizeof(int));
1079 
1080                 if (smallest < 1) {
1081                         complainx("Validation: block %d: bad SMALLEST", i);
1082                         exit(1);
1083                 }
1084                 if (largest >= RANDOM_MAX) {
1085                         complainx("Validation: block %d: bad LARGEST", i);
1086                         exit(1);
1087                 }
1088                 if (smallest > largest) {
1089                         complainx("Validation: block %d: SMALLEST > LARGEST",
1090                                   i);
1091                         exit(1);
1092                 }
1093 
1094                 if (smallest < prev_largest) {
1095                         complain("Validation: block %d smallest key %d",
1096                                  i, smallest);
1097                         complain("Validation: previous block largest key %d",
1098                                  prev_largest);
1099                         complain("Validation failed");
1100                         exit(1);
1101                 }
1102         }
1103 
1104 
1105         for (i=0; i<numprocs; i++) {
1106                 doremove(validname(i));
1107         }
1108 }
